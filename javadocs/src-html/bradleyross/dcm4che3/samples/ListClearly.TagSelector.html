053        /**
054         * Test to see if tag is in range.
055         * <p>Can group numbers be greater than 0x7FFF?</p>
056         * @author Bradley Ross
057         *
058         */
059        protected class GroupSelector implements TagSelector {
060                public int minValue = 0x00000000;
061                public int maxValue = 0x7FFFFFFF;
062                public boolean accept(int value) {
063                        if (value >= minValue && value <= maxValue) {
064                                return true;
065                        } else {
066                                return false;
067                        }
068                }
069                public GroupSelector() {
070                        minValue = 0;
071                        maxValue = 0x7FFFFFFF;
072                }
073                public GroupSelector(int group) {
074                        if (group > 0x7FFF) {
075                                minValue = 0x7FFF0000;
076                        } else {
077                        minValue = group * 0x10000;
078                        }
079                        maxValue = minValue + 0xFFFF;
080                }
081                public GroupSelector(int start, int stop) {
082                        minValue = start;
083                        maxValue = stop;
084                }
085        }

210        /**
211         * Responds to actions involving elements of user interface.
212         * @param event action event
213         */
214        public void actionPerformed(ActionEvent event) {
215                String command = event.getActionCommand();
216                if (command.equalsIgnoreCase("READ")) {
217                        JFileChooser chooser = new JFileChooser();
218                        chooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
219                        int returnValue = chooser.showOpenDialog(frame);
220                        if (returnValue == JFileChooser.APPROVE_OPTION) {
221                                dicomFile = chooser.getSelectedFile();
222                                try {
223                                        fileName = dicomFile.getCanonicalPath();
224                                        log(fileName);
225                                } catch (IOException e) {
226                                        System.out.println("Unable to read file name");
227                                        e.printStackTrace();
228                                }
229                                System.out.println("Processing " + fileName);
230                                readFile();
231                        }
232                } else if (command.equalsIgnoreCase("SAVE")) {
233                        JFileChooser chooser = new JFileChooser();
234                        int returnValue = chooser.showSaveDialog(frame);
235                        if (returnValue == JFileChooser.APPROVE_OPTION) {
236                                dicomFile = chooser.getSelectedFile();
237                                try {
238                                        fileName = dicomFile.getCanonicalPath();
239                                } catch (IOException e) {
240                                        System.out.println("Unable to read file name");
241                                        e.printStackTrace();
242                                }
243                                System.out.println("Saving to " + fileName);    
244                        }
245                } else if (command.equalsIgnoreCase("CLEAR")) {
246                        textArea.setText(new String());
247                }else if (command.equalsIgnoreCase("EXIT")) {
248                
249                        frame.setVisible(false);
250                        frame.removeAll();
251                        frame.dispose();
252                } else {
253                        System.out.println(command + " is not recognized command");
254                }
255        }

262         * Write a log message.
263         * @param value text to be passed in message
264         */
265        protected void log(String value) {
266                for (int i = 0; i < indentLevel; i++) {
267                        textArea.append("     ");

271        /**
272         * Read a file containing a Dicom dataset.
273         */
274        protected void readFile() {
275                try {
276
277                        DicomInputStream dis = new DicomInputStream(dicomFile);
278                        dicomObject = dis.readDataset(-1,  -1);
279                        dis.close();
280                        // processObject(dicomObject);
281                } catch (IOException e) {
282                        log("Error while reading Dicom dataset");
283                }
284        }
285        protected void processObject(Attributes dicomObject) {
286                TagSelector use = new GroupSelector();
287                processObject(dicomObject, use);
288        }
289        /**
290         * Print the contents of a Dicom object.
291         * @param dicomObject object whose contents are to be displayed
292         * <p>This method will recursively call itself to handle
293         *    embedded objects.</p>
294         */
295        protected void processObject (Attributes dicomObject, TagSelector selector) {
296                int[] tags = dicomObject.tags();
297                // log("There are " + Integer.toString(tags.length) + " tags");
298                for (int i = 0; i < tags.length; i++) {
299                        int tag = tags[i];
300                        if (selector != null && !selector.accept(tags[i])) { continue; }
301                        VR vr = dicomObject.getVR(tag);
302                        showValue(dicomObject, tag);
303                        if (vr == VR.SQ) {
304                                indentLevel++;
305                                Sequence seq = dicomObject.getSequence(tag);
306                                log("There are " + Integer.toString(seq.size()) + " items in the sequence");
307                                for (int i2 = 0; i2 < seq.size(); i2++) {
308                                        log("Start of item " + Integer.toString(i2));
309                                        indentLevel++;
310                                        processObject(seq.get(i2));
311                                        indentLevel--;
312                                        log("End of item " + Integer.toString(i2));
313                                }
314                                indentLevel--;
315                        }
316                }
317        }
318        /**
319         * Display some of the fields used in Standardized Uptake
320         * Values.
321         * <p>Current activity is in Becquerels per milliliter while
322         *    the injected dose is in Becquerels.</p>
323         *    <p>see <a href="http://www.clearcanvas.ca/dnn/Portals/0/ClearCanvasFiles/Documentation/UsersGuide/Personal/7_0/index.html?suv.htm" target="_blank">
324         *       ClearCanvas SUV discussion</a></p>
325         *    <p>Highest current radiation levels on MIM data is around 8000
326         *       Becquerels per milliliter.</p>
327         *    <p>See section C.8.9.2 PET Isotope Module in part 3 of the Dicom standards.</p>
328         * @param dicomObject Dicom SOP object that will be read
329         */
330        public void processSuv(Attributes dicomObject) {
331                log("*****  *****");
332                log("Processing SUV fields");
333                showValue(dicomObject, tag.Manufacturer);
334                if (dicomObject.containsValue(tag.RadiopharmaceuticalInformationSequence)) {
335                        Sequence seq = dicomObject.getSequence(tag.RadiopharmaceuticalInformationSequence);
336                        Attributes contents = seq.get(0);
337                        log("Display of RadiopharmaceuticalInformationSequence");
338                        processObject(contents);
339                        log("Display of group 0x7053 - Philips only");
340                        processObject(dicomObject, new GroupSelector(0x7053));
341                        log("Display of group 0x0009 - GE only");
342                        processObject(dicomObject, new GroupSelector(0x0009));
343                        log("Display of individual tags");
344                        showValue(dicomObject, tag.Manufacturer);
345                        showValue(dicomObject, tag.RescaleIntercept);
346                        showValue(dicomObject, tag.RescaleSlope);
347                        showValue(contents, tag.RadiopharmaceuticalStartDateTime);
348                        showValue(contents, tag.RadiopharmaceuticalStartTime);
349                        showValue(dicomObject, tag.PatientWeight);
350                        showValue(dicomObject, tag.AcquisitionTime);
351                        showValue(dicomObject, tag.SeriesTime);
352                        showValue(contents, tag.RadionuclideTotalDose);
353                        showValue(contents, tag.RadionuclideHalfLife);
354                        showValue(dicomObject, tag.DecayCorrection);
355                        showValue(dicomObject, tag.Units);
356                        float bodyWeight; // g
357                        float startClock;
358                        float endClock;
359                        float halfLife;
360                        float totalDose; // Bq
361                        float attenuation;
362                        float halfLives;
363                        float suvMultiplier;
364                        log("Display calculations");
365                        if (dicomObject.containsValue(tag.AcquisitionTime)) {
366                                endClock = (float) dicomObject.getDate(tag.AcquisitionTime).getTime() / 1000.0f;
367                        } else if (dicomObject.containsValue(tag.SeriesTime)) {
368                                endClock = (float) dicomObject.getDate(tag.SeriesTime).getTime() / 1000.0f;
369                        } else {
370                                log("Unable to find end time");
371                                return;
372                        }
373                        if (contents.containsValue(tag.RadiopharmaceuticalStartTime) ) {
374                                        startClock = (float) contents.getDate(tag.RadiopharmaceuticalStartTime).getTime() /
375                                                        1000.0f;
376                        } else {
377                                log("Unable to find start time");
378                                return;
379                        }
380                        log("Duration since radiation measurement is " + Float.toString(endClock - startClock) + " seconds");
381                        if (contents.containsValue(tag.RadionuclideHalfLife)) {
382                                halfLife = contents.getFloat(tag.RadionuclideHalfLife, 0.0f);
383                        } else {
384                                log ("Unable to find half life");
385                                return;
386                        }
387                        if (contents.containsValue(tag.RadionuclideTotalDose)) {
388                                totalDose = contents.getFloat(tag.RadionuclideTotalDose, 0.0f);
389                        } else {
390                                log ("Unable to find total dose");
391                                return;
392                        }
393                        log("Radionuclide total dose in Becquerels: " + Float.toString(totalDose));
394                        if (dicomObject.containsValue(tag.PatientWeight)) {
395                                bodyWeight = dicomObject.getFloat(tag.PatientWeight, 0.0f) * 1000.0f;
396                        } else {
397                                log ("Unable to find patient weight");
398                                return;
399                        }
400                        log("Body mass is " + Float.toString(bodyWeight) + " grams");
401                        halfLives = (endClock - startClock)/halfLife;
402                        log(Float.toString(halfLives) + " half lives");
403                        attenuation = (float) Math.pow(0.5d, halfLives);
404                        log("attenuation fraction = " + Float.toString(attenuation));
405                        suvMultiplier = bodyWeight / totalDose / attenuation;
406                        log("SUV multiplier is " + Float.toString(suvMultiplier));
407                }
408        }
409        /**
410         * Write information about a tag.
411         * @param dicomObject Dicom object containing the tag
412         * @param tag integer value of the tag to be displayed
413         */
414        public void showValue(Attributes dicomObject, int tag) {
415                String name;
416                if (TagUtils.isPrivateGroup(tag)) {
417                        name = " - private group - ";
418                } else {
419                name = ElementDictionary.keywordOf(tag, null);
420                }
421                VR vr = dicomObject.getVR(tag);
422                if (!dicomObject.containsValue(tag)) {
423                        log(TagUtils.toString(tag) + " " + name + " not found");
424                } else if (vr == VR.LT || vr == VR.UI || vr == VR.ST ||
425                                vr == VR.LO || vr == VR.SH || vr == VR.UT || vr == VR.CS) {
426                        String value = dicomObject.getString(tag);
427                        StringBuilder sb = new StringBuilder();
428                        sb.append(TagUtils.toString(tag) + " " + name + " " + vr.toString() + " :: ");
429                        if (value.length() < 60) {
430                                sb.append(value);
431                        } else {
432                                sb.append(value.substring(0, 60));
433                        }
434                        log(sb.toString());
435                } else if (vr == VR.DS || vr == VR.FD || vr == VR.FL) {
436                        StringBuilder sb = new StringBuilder();
437                        sb.append(TagUtils.toString(tag) + " " + name);
438                        float[] values = dicomObject.getFloats(tag);
439                        if (values.length > 1) {
440                                sb.append(" - " + Integer.toString(values.length) + " values :: ");
441                        } else {
442                                sb.append(" :: ");
443                        }
444                        sb.append(Float.toString(values[0]));
445                        log(sb.toString());
446                } else if (vr == VR.US || vr == VR.UL || vr == VR.SS || vr == VR.SL) {
447                        StringBuilder sb = new StringBuilder();
448                        sb.append(TagUtils.toString(tag) + " " + name);
449                        int[] values = dicomObject.getInts(tag);
450                        if (values.length > 1) {
451                                sb.append(" - " + Integer.toString(values.length) + " values - ");
452                        } else {
453                                sb.append(" - ");
454                        }
455                        sb.append(Integer.toString(values[0]));
456                        log(sb.toString());
457                } else if (vr == VR.TM) {
458                        StringBuilder sb = new StringBuilder();
459                        sb.append(TagUtils.toString(tag) + " " + name + " :: ");
460                        float value = (float) dicomObject.getDate(tag).getTime() / 1000.0f;
461                        sb.append(Float.toString(value));
462                        log(sb.toString());
463                } else if (vr == VR.DA || vr == VR.DT) {
464                        StringBuilder sb = new StringBuilder();
465                        sb.append(TagUtils.toString(tag) + " " + name + " ");
466                        Date value = dicomObject.getDate(tag);
467                        sb.append(value.toString());
468                        log(sb.toString());
469                } else {
470                        StringBuilder sb = new StringBuilder();
471                        sb.append(TagUtils.toString(tag) + " " + name + " " + vr.toString() + " ");
472                        log(sb.toString());
473                }
474        }
