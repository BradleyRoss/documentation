001package bradleyross.library.dcm4che3;
002/**
003 * Represents a remote DICOM system.
004 * 
005 * @author Bradley Ross
006 *
007 */
008public class DicomSystem {
009        /**
010         * Default constructor.
011         */
012        public DicomSystem()
013        { ; }
014        /**
015         * IP address or domain name for target system.
016         */
017        private String address = null;
018        /**
019         * Getter for address property
020         * @see #address
021         * @return IP address or domain name
022         */
023        public String getAddress()
024        {
025                return address;
026        }
027        /**
028         * Setter for IP address or domain name.
029         * @see #address
030         * @param value IP address or domain name
031         */
032        public void setAddress(String value)
033        {
034                address = value;
035        }
036        /**
037         * Port number for target system.
038         * 
039         * <p>A negative value indicates that the value 
040         *    has not been set.</p>
041         */
042        private int port = -1;
043        /**
044         * Getter for port number.
045         * @see #port
046         * @return port number
047         */
048        public int getPort()
049        {
050                return port;
051        }
052        /**
053         * Setter for port number.
054         * @see #port
055         * @param value port number
056         */
057        public void setPort(int value)
058        {
059                port = value;
060        }
061        /**
062         * Application entity title for remote system.
063         */
064        private String title = null;
065        /**
066         * Getter for title property.
067         * @see #title
068         * @return title property
069         */
070        public String getTitle()
071        {
072                return title;
073        }
074        /** 
075         * Setter for title property
076         * @see #title
077         * @param value value for title property
078         */
079        public void setTitle(String value)
080        {
081                title = value;
082        }
083        /**
084         * Display name for remote system.  
085         * 
086         * By setting the
087         * display name different from the application entity
088         * title, it is possible to connect to multiple servers
089         * with the same name.
090         */
091        private String displayName = null;
092        /**
093         * Getter for displayName property.
094         * @see #displayName
095         * @return displayName property
096         */
097        public String getDisplayName()
098        {
099                return displayName;
100        }
101        /** 
102         * Setter for displayName property.
103         * @see #displayName
104         * @param value displayName property value
105         */
106        public void setDisplayName(String value)
107        {
108                displayName = value;
109        }
110        /**
111         * This would be an alternate Application Entity Title to be used
112         * for the local system when dealing with the remote system.
113         */
114        private String localName = null;
115        /**
116         * Getter for localName property.
117         * @return value of localName property
118         * @see #localName
119         */
120        public String getLocalName()
121        {
122                return localName;
123        }
124        /**
125         * Setter for localName property.
126         * @param value value of localName property
127         * @see #localName
128         */
129        public void setLocalName(String value)
130        {
131                localName = value;
132        }
133        private String description = new String();
134        public String getDescription() {
135                return description;
136        }
137        public void setDescription(String value) {
138                description = value;
139        }
140        private String sendTransferSyntax = null;
141        public String getSendTransferSyntax() {
142                return sendTransferSyntax;
143        }
144        public void setSendTransferSyntax(String value) {
145                sendTransferSyntax = value;
146                if (sendTransferSyntax == null) {
147                        sendTransferSyntax = new String();
148                }
149        }
150        /**
151         * Copy information on the system to a String object which
152         * can be used in diagnostic messages.
153         * 
154         * @return Information on server
155         */
156        public String dump()
157        {
158                StringBuffer working = new StringBuffer("System information:");
159                if (displayName != null)
160                {
161                        working.append("\n   Display Name: " + displayName);
162                }
163                if (title != null)
164                {
165                        working.append("\n   Title: " + title);
166                }
167                if (address != null)
168                {
169                        working.append("\n   Address: " + address);
170                }
171                if (port >= 0)
172                {
173                        working.append("\n   Port: " + Integer.toString(port));
174                }
175                return new String(working);
176        }
177        /**
178         * Returns the data for the Medical Connections
179         * public Dicom server.
180         * @return system description
181         */
182        public DicomSystem getMedicalConnections() {
183                DicomSystem working = new DicomSystem();
184                working.setAddress("www.dicomserver.co.uk");
185                working.setPort(104);
186                working.setTitle("DICOMSERVER");
187                working.setDisplayName("Medical Connections");
188                return working;
189        }
190        /**
191         * Provides the data for the PixelMed public Dicom server.
192         * @return system description
193         */
194        public DicomSystem getPixelMed() {
195                DicomSystem working = new DicomSystem();
196                working.setAddress("184.73.255.26");
197                working.setPort(11112);
198                working.setTitle("AWSPIXELMEDPUB");
199                working.setDisplayName("PixelMed");
200                return working;
201                
202        }
203}
