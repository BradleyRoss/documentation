001package bradleyross.library.dcm4che3;
002import java.util.ArrayList;
003/**
004 * Handles information on the local server and the remote servers
005 * that can be use for processing.
006 * 
007 * @author Bradley Ross
008 *
009 */
010public class Configuration {
011        /**
012         * List of DICOM servers to which connections can be made.
013         */
014        private static ArrayList<DicomSystem> systemList = new ArrayList<DicomSystem>();
015        /**
016         * Add a DICOM server to the list of DICOM servers.
017         * @param value properties of DICOM server
018         */
019        public static void addSystem(DicomSystem value)
020        {
021                systemList.add(value);
022        }
023        /**
024         * Return list of defined DICOM servers
025         * @return list of servers
026         */
027        public static ArrayList<DicomSystem> getSystems()
028        {
029                return systemList;
030        }
031        /**
032         * Get the properties for a specific DICOM server
033         * @param value identifies DICOM server
034         * @return properties of server
035         */
036        public static DicomSystem getSystem(String value)
037        {
038                for (int i = 0; i < systemList.size(); i++)
039                {
040                        if (systemList.get(i).getDisplayName().equalsIgnoreCase(value))
041                        {
042                                return systemList.get(i);
043                        }
044                }
045                return new DicomSystem();
046        }
047        /**
048         * IP address or domain name to be used for
049         * local system.
050         */
051        private static String localAddress = "127.0.0.1";
052        /**
053         * Getter for localAddress property.
054         * @return localAddress property
055         * @see #localAddress
056         */
057        public static String getLocalAddress()
058        {
059                return localAddress;
060        }
061        /**
062         * Setter for localAddress property.
063         * @param value localAddress property
064         * @see #localAddress
065         */
066        public static void setLocalAddress(String value)
067        {
068                localAddress = value;
069        }
070        /**
071         * Port address to be used for local system.
072         */
073        private static int localPort = -1;
074        /**
075         * Getter for localPort property.
076         * @return local port number
077         * @see #localPort
078         */
079        public static int getLocalPort()
080        {
081                return localPort;
082        }
083        /**
084         * Setter for localPort property.
085         * @param value localPort property
086         * @see #localPort
087         */
088        public static void setLocalPort(int value)
089        {
090                localPort = value;
091        }
092        /**
093         * Application entity title to be used for local system.
094         */
095        private static String localTitle = "TEST";
096        /**
097         * Getter for localTitle property
098         * @return localTitle property
099         * @see #localTitle
100         */
101        public static String getLocalTitle()
102        {
103                return localTitle;
104        }
105        /**
106         * Setter for localTitle property
107         * @param value localTitle property
108         * @see #localTitle
109         */
110        public static void setLocalTitle(String value)
111        {
112                localTitle = value;
113        }
114        /**
115         * Set up a  DICOM servers that can be used for requests.
116         * <table border="1">
117         * <tr><td>DICOMSERVER</td><td>DICOMSERVER@www.dicomserver.co.uk:104</td></tr>
118         * </table>
119         * <p>An additional entry was set up with the name BADENTRY for testing 
120         *    purposes.  This entry deliberately points to a nonexistent
121         *    server.</p>
122         * 
123         */
124        public static void testLoad()
125        {
126                setLocalTitle("TEST");
127                DicomSystem working = null;
128                working = new DicomSystem();
129                working.setAddress("www.dicomserver.co.uk");
130                working.setPort(104);
131                working.setTitle("DICOMSERVER");
132                working.setDisplayName("DICOMSERVER");
133                addSystem(working);
134        }
135        /**
136         * Add DICOMSERVER to the system list
137         * and then print its configuration.
138         * 
139         * @param args Not used in this case
140         */
141        public static void main(String[] args) {
142                testLoad();
143
144                System.out.println(getSystem("DICOMSERVER").dump());
145        }
146}
