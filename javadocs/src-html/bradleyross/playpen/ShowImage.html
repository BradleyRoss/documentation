001package bradleyross.playpen;
002import java.awt.image.BufferedImage;
003import java.io.File;
004// import java.io.FileNotFoundException;
005// import java.io.IOException;
006import java.io.FileInputStream;
007import javax.imageio.ImageIO;
008import javax.swing.JFileChooser;
009import javax.swing.JFrame;
010import javax.swing.JPanel;
011// import java.awt.GridBagLayout;
012// import java.awt.GridBagConstraints;
013import java.awt.Graphics;
014import java.awt.Color;
015/**
016 * Allow the user to specify a graphics file and display it
017 * in a panel.
018 * <p>Doesn't seem to work right when run from Eclipse on 
019 *    Mac.</p>
020 * @author Bradley Ross
021 *
022 */
023public class ShowImage {
024        protected File file;
025        protected BufferedImage image;
026        protected JFrame frame;
027        protected JPanel panel;
028        @SuppressWarnings("serial")
029        protected class PictureFrame extends JPanel {
030                public PictureFrame() {
031                this.setSize(300,300);
032                this.setBackground(Color.RED);
033                }
034                public void paint(Graphics g) {
035                        if (image != null) {
036                        g.drawImage(image, 0, 0, null);
037                        } 
038                }
039        }
040        protected void run() throws Exception {
041                frame = new JFrame();
042                frame.setSize(300, 500);
043                panel = new PictureFrame();
044                frame.add(panel);
045                frame.setVisible(true);
046                JFileChooser chooser = new JFileChooser();
047                System.out.println("Opening dialog");
048                int retval = chooser.showOpenDialog(panel);
049                if (retval == JFileChooser.APPROVE_OPTION) {
050                        file = chooser.getSelectedFile();
051                }
052
053                FileInputStream input = new FileInputStream(file);
054                image = ImageIO.read(input);
055                System.out.println(Integer.toString(image.getWidth()));
056                System.out.println(Integer.toString(image.getHeight()));
057                                panel.repaint();
058
059        }
060        /**
061         * @param args
062         */
063        public static void main(String[] args) {
064                try {
065                ShowImage instance = new ShowImage();
066                instance.run();
067                } catch (Exception e) {
068                        e.printStackTrace();
069                }
070        }
071
072}
