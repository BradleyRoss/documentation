001package bradleyross.playpen;
002import java.lang.Cloneable;
003/**
004 * Test of cloning operation.
005 * @author Bradley Ross
006 * @see Object#clone()
007 * @see Cloneable
008 *
009 */
010public class CloneTest {
011
012        private class Dummy implements Cloneable{
013                protected int id;
014                public Dummy(int value) {
015                        id = value;
016                }
017                public Dummy() { ; }
018                public int getId() {
019                        return id;
020                }
021                public void setId(int value) {
022                        id = value;
023                }
024                public Dummy clone() throws CloneNotSupportedException {
025                        Dummy newItem = new Dummy();
026                        newItem.setId(this.getId());
027                        return newItem;
028                }
029        }
030        Dummy first = null;
031        Dummy second = null;
032        Dummy third = null;
033        protected void display(String value) {
034                System.out.println(value);
035                if (first == null) {
036                        System.out.println("     first is null");
037                } else {
038                        System.out.println("     first: " + Integer.toString(first.getId()));
039                }
040                if (second == null) {
041                        System.out.println("     second is null");
042                } else {
043                        System.out.println("     second: " + Integer.toString(second.getId()));
044                }
045                if (third == null) {
046                        System.out.println("     third is null");
047                } else {
048                        System.out.println("     third: " + Integer.toString(third.getId()));
049                }
050        }
051        public void test() {
052                try {
053                        display("starting condition");
054                        first = new Dummy(5);
055                        display("first created with value of 5");
056                        second =  first.clone();
057                        display("second created as clone of first");
058                        third = first;
059                        display("third set equal to first");
060                        second.setId(4);
061                        display("second changed to 4");
062                        first.setId(3);
063                        display("first changed to 3");
064                        third.setId(6);
065                        display("third changed to 6");
066
067                } catch (Exception e) {
068                        e.printStackTrace();
069                }
070        }
071        /**
072         * @param args
073         */
074        public static void main(String[] args) {
075                // TODO Auto-generated method stub
076                CloneTest instance = new CloneTest();
077                instance.test();
078
079        }
080
081}
