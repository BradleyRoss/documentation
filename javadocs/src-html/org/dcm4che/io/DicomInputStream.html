379    public void readHeader() throws IOException {
380        byte[] buf = buffer;
381        tagPos = pos; 
382        readFully(buf, 0, 8);
383        switch(tag = ByteUtils.bytesToTag(buf, 0, bigEndian)) {
384        case Tag.Item:
385        case Tag.ItemDelimitationItem:
386        case Tag.SequenceDelimitationItem:
387           vr = null;
388           break;
389        default:
390            if (explicitVR) {
391                vr = VR.valueOf(ByteUtils.bytesToVR(buf, 4));
392                if (vr.headerLength() == 8) {
393                    length = ByteUtils.bytesToUShort(buf, 6, bigEndian);
394                    return;
395                }
396                readFully(buf, 4, 4);
397            } else {
398                vr = VR.UN;
399            }
400        }
401        length = ByteUtils.bytesToInt(buf, 4, bigEndian);
402    }

404    public Attributes readCommand() throws IOException {
405        if (bigEndian || explicitVR)
406            throw new IllegalStateException(
407                    "bigEndian=" + bigEndian + ", explicitVR=" + explicitVR );
408        Attributes attrs = new Attributes(9);
409        readAttributes(attrs, -1, -1);
410        return attrs;
411    }
412    /**
413     * Creates an {@link Attributes} object by reading from a file.
414     * @param len maximum number of bytes to be read - a value of -1 indicates that
415     *        that there is no maximum length
416     * @param stopTag value for the tag that will halt reading of the file - a
417     *        value of -1 indicates that there is no tag that will halt
418     *        reading
419     * @return {@link Attributes} object containing the contents of the dataset
420     * @throws IOException
421     */
422    public Attributes readDataset(int len, int stopTag) throws IOException {
423        handler.startDataset(this);
424        readFileMetaInformation();
425        Attributes attrs = new Attributes(bigEndian, 64);
426        readAttributes(attrs, len, stopTag);
427        attrs.trimToSize();
428        handler.endDataset(this);
429        return attrs;
430    }

432    public Attributes readFileMetaInformation() throws IOException {
433        if (!hasfmi)
434            return null;  // No File Meta Information
435        if (fileMetaInformation != null)
436            return fileMetaInformation;  // already read
437
438        Attributes attrs = new Attributes(bigEndian, 9);
439        while (pos != fmiEndPos) {
440            mark(12);
441            readHeader();
442            if (TagUtils.groupNumber(tag) != 2) {
443                LOG.warn(MISSING_FMI_LENGTH);
444                reset();
445                break;
446            }
447             if (vr != null) {
448                if (vr == VR.UN)
449                    vr = ElementDictionary.getStandardElementDictionary()
450                            .vrOf(tag);
451                handler.readValue(this, attrs);
452            } else
453                skipAttribute(UNEXPECTED_ATTRIBUTE);
454        }
455        fileMetaInformation = attrs;
456
457        String tsuid = attrs.getString(Tag.TransferSyntaxUID, null);
458        if (tsuid == null) {
459            LOG.warn(MISSING_TRANSFER_SYNTAX);
460            tsuid = UID.ExplicitVRLittleEndian;
461        }
462        switchTransferSyntax(tsuid);
463        return attrs;
464    }

466    public void readAttributes(Attributes attrs, int len, int stopTag)
467            throws IOException {
468        boolean undeflen = len == -1;
469        boolean hasStopTag = stopTag != -1;
470        long endPos =  pos + (len & 0xffffffffL);
471        while (undeflen || this.pos < endPos) {
472            try {
473                readHeader();
474            } catch (EOFException e) {
475                if (undeflen && pos == tagPos)
476                    break;
477                throw e;
478            }
479            if (hasStopTag && tag == stopTag)
480                break;
481            if (vr != null) {
482                boolean prevBigEndian = bigEndian;
483                boolean prevExplicitVR = explicitVR;
484                try {
485                    if (vr == VR.UN) {
486                        bigEndian = false;
487                        explicitVR = false;
488                        vr = ElementDictionary.vrOf(tag,
489                                attrs.getPrivateCreator(tag));
490                        if (vr == VR.UN && length == -1)
491                            vr = VR.SQ; // assumes UN with undefined length are SQ,
492                                        // will fail on UN fragments!
493                    }
494                    handler.readValue(this, attrs);
495                } finally {
496                    bigEndian = prevBigEndian;
497                    explicitVR = prevExplicitVR;
498                }
499            } else
500                skipAttribute(UNEXPECTED_ATTRIBUTE);
501        }
502    }

504    @Override
505    public void readValue(DicomInputStream dis, Attributes attrs)
506            throws IOException {
507        checkIsThis(dis);
508        if (includeBulkData == IncludeBulkData.NO && length != -1 && isBulkData(attrs)) {
509            skipFully(length);
510        } else if (length == 0) {
511            attrs.setNull(tag, vr);
512        } else if (vr == VR.SQ) {
513            readSequence(length, attrs, tag);
514        } else if (length == -1) {
515            readFragments(attrs, tag, vr);
516        } else if (length == BulkDataLocator.MAGIC_LEN
517                && super.in instanceof ObjectInputStream) {
518            attrs.setValue(tag, vr, BulkDataLocator.deserializeFrom(
519                    (ObjectInputStream) super.in));
520        } else if (includeBulkData == IncludeBulkData.LOCATOR && isBulkData(attrs)) {
521            attrs.setValue(tag, vr, createBulkDataLocator());
522        } else {
523            byte[] b = readValue();
524            if (!TagUtils.isGroupLength(tag)) {
525                if (bigEndian != attrs.bigEndian())
526                    vr.toggleEndian(b, false);
527                attrs.setBytes(tag, vr, b);
528            } else if (tag == Tag.FileMetaInformationGroupLength)
529                setFileMetaInformationGroupLength(b);
530        }
531    }

533    public BulkDataLocator createBulkDataLocator() throws IOException {
534            BulkDataLocator locator;
535        if (uri != null && !(super.in instanceof InflaterInputStream)) {
536            locator = new BulkDataLocator(uri, tsuid, pos, length);
537            skipFully(length);
538        } else {
539            if (blkOut == null) {
540                File blkfile = File.createTempFile(blkFilePrefix,
541                        blkFileSuffix, blkDirectory);
542                if (blkFiles == null)
543                    blkFiles = new ArrayList<File>();
544                blkFiles.add(blkfile);
545                blkURI = blkfile.toURI().toString();
546                blkOut = new FileOutputStream(blkfile);
547                blkOutPos = 0L;
548            }
549            try {
550                StreamUtils.copy(this, blkOut, length);
551            } finally {
552                if (!catBlkFiles) {
553                    SafeClose.close(blkOut);
554                    blkOut = null;
555                }
556            }
557            locator = new BulkDataLocator(blkURI,
558                    (super.in instanceof InflaterInputStream)
559                            ? UID.ExplicitVRLittleEndian
560                            : tsuid,
561                     blkOutPos, length);
562            blkOutPos += length;
563        }
564        return locator;
565    }

567    public boolean isBulkData(Attributes attrs) {
568        if (TagUtils.isPrivateCreator(tag))
569            return false;
570        int grtag = TagUtils.groupNumber(tag);
571        if (grtag < 8)
572            return false;
573        Attributes item = bulkData;
574        if (item == null)
575            bulkData = item = DicomInputStream.defaultBulkData();
576        for (int i = 0; i < level; i++) {
577            ItemPointer ip = itemPointers[i];
578            item = item.getNestedDataset(ip.privateCreator, ip.sequenceTag);
579            if (item == null)
580                return false;
581        }
582        int tag0 = ((grtag &= 0xff01) == 0x5000 || grtag == 0x6000)
583                ? tag & 0xff00ffff : tag;
584        return item.contains(attrs.getPrivateCreator(tag), tag0);
585    }

587    public boolean isBulkDataFragment() {
588        if (tag != Tag.Item)
589            return false;
590        if (bulkData == null)
591            bulkData = DicomInputStream.defaultBulkData();
592        
593        Attributes item = bulkData;
594        for (int i = 0; i < level; i++) {
595            ItemPointer ip = itemPointers[i];
596            Object value = item.getValue(ip.privateCreator, ip.sequenceTag);
597            if (value instanceof Sequence)
598                item = ((Sequence) value).get(0);
599            else
600                return value != null;
601        }
602        return false;
603    }

605    @Override
606    public void readValue(DicomInputStream dis, Sequence seq)
607            throws IOException {
608        checkIsThis(dis);
609        if (length == 0) {
610            seq.add(new Attributes(seq.getParent().bigEndian(), 0));
611            return;
612        }
613        Attributes attrs = new Attributes(seq.getParent().bigEndian());
614        seq.add(attrs);
615        readAttributes(attrs, length, Tag.ItemDelimitationItem);
616        attrs.trimToSize();
617    }

619    @Override
620    public void readValue(DicomInputStream dis, Fragments frags)
621            throws IOException {
622        checkIsThis(dis);
623        if (includeBulkData == IncludeBulkData.NO && isBulkDataFragment()) {
624            skipFully(length);
625        } else if (length == 0) {
626            frags.add(ByteUtils.EMPTY_BYTES);
627        } else if (length == BulkDataLocator.MAGIC_LEN
628                && super.in instanceof ObjectInputStream) {
629            frags.add(BulkDataLocator.deserializeFrom((ObjectInputStream) super.in));
630        } else if (includeBulkData == IncludeBulkData.LOCATOR && isBulkDataFragment()) {
631            frags.add(createBulkDataLocator());
632        } else {
633            byte[] b = readValue();
634            if (bigEndian != frags.bigEndian())
635                vr.toggleEndian(b, false);
636            frags.add(b);
637        }
638    }

653    private void skipAttribute(String message) throws IOException {
654        LOG.warn(message,
655                 new Object[] { TagUtils.toString(tag), length, tagPos });
656        skip(length);
657    }

659    private void readSequence(int len, Attributes attrs, int sqtag)
660            throws IOException {
661        if (len == 0) {
662            attrs.setNull(sqtag, VR.SQ);
663            return;
664        }
665        Sequence seq = attrs.newSequence(sqtag, 10);
666        String privateCreator = attrs.getPrivateCreator(sqtag);
667        boolean undefLen = len == -1;
668        long endPos = pos + (len & 0xffffffffL);
669        for (int i = 0; undefLen || pos < endPos; ++i) {
670            readHeader();
671            if (tag == Tag.Item) {
672                addItemPointer(sqtag, privateCreator, i);
673                handler.readValue(this, seq);
674                removeItemPointer();
675            } else if (tag == Tag.SequenceDelimitationItem) {
676                if (length != 0)
677                    skipAttribute(UNEXPECTED_NON_ZERO_ITEM_LENGTH);
678                break;
679            } else
680                skipAttribute(UNEXPECTED_ATTRIBUTE);
681        }
682        if (seq.isEmpty())
683            attrs.setNull(sqtag, VR.SQ);
684        else
685            seq.trimToSize();
686    }
687
688    private void addItemPointer(int sqtag, String privateCreator, int itemIndex) {
689        if (itemPointers.length <= level)
690            itemPointers = Arrays.copyOf(itemPointers, level + 1);
691        itemPointers[level++] = new ItemPointer(sqtag, privateCreator, itemIndex);
692    }
693
694    private void removeItemPointer() {
695        itemPointers[--level] = null;
696    }

698    public Attributes readItem() throws IOException {
699        readHeader();
700        if (tag != Tag.Item)
701            throw new IOException("Unexpected attribute "
702                    + TagUtils.toString(tag) + " #" + length + " @ " + pos);
703        Attributes attrs = new Attributes(bigEndian);
704        attrs.setItemPosition(tagPos);
705        readAttributes(attrs, length, Tag.ItemDelimitationItem);
706        attrs.trimToSize();
707        return attrs;
708    }

710    private void readFragments(Attributes attrs, int fragsTag, VR vr)
711            throws IOException {
712        Fragments frags = new Fragments(vr, attrs.bigEndian(), 10);
713        String privateCreator = attrs.getPrivateCreator(fragsTag);
714        for (int i = 0; true; ++i) {
715            readHeader();
716            if (tag == Tag.Item) {
717                addItemPointer(fragsTag, privateCreator, i);
718                handler.readValue(this, frags);
719                removeItemPointer();
720            } else if (tag == Tag.SequenceDelimitationItem) {
721                if (length != 0)
722                    skipAttribute(UNEXPECTED_NON_ZERO_ITEM_LENGTH);
723                break;
724            } else
725                skipAttribute(UNEXPECTED_ATTRIBUTE);
726        }
727        if (frags.isEmpty())
728            attrs.setNull(fragsTag, vr);
729        else {
730            frags.trimToSize();
731            attrs.setValue(fragsTag, vr, frags);
732        }
733    }

735    public byte[] readValue() throws IOException {
736        int valLen = length;
737        if (valLen < 0)
738            throw new EOFException(); // assume InputStream length < 2 GiB
739        int allocLen = allocateLimit >= 0
740                ? Math.min(valLen, allocateLimit)
741                : valLen;
742        byte[] value = new byte[allocLen];
743        readFully(value, 0, allocLen);
744        while (allocLen < valLen) {
745            int newLength = Math.min(valLen, allocLen << 1);
746            value = Arrays.copyOf(value, newLength);
747            readFully(value, allocLen, newLength - allocLen);
748            allocLen = newLength;
749        }
750        return value;
751    }

753    private void switchTransferSyntax(String tsuid) throws IOException {
754        this.tsuid = tsuid;
755        bigEndian = tsuid.equals(UID.ExplicitVRBigEndian);
756        explicitVR = !tsuid.equals(UID.ImplicitVRLittleEndian);
757        if (tsuid.equals(UID.DeflatedExplicitVRLittleEndian)
758                        || tsuid.equals(UID.JPIPReferencedDeflate)) {
759            if (hasZLIBHeader()) {
760                LOG.warn(DEFLATED_WITH_ZLIB_HEADER);
761                super.in = new InflaterInputStream(super.in);
762            } else
763                super.in = new InflaterInputStream(super.in,
764                        new Inflater(true));
765        }
766    }

768    private boolean hasZLIBHeader() throws IOException {
769        if (!markSupported())
770            return false;
771        byte[] buf = buffer;
772        mark(2);
773        read(buf, 0, 2);
774        reset();
775        return ByteUtils.bytesToUShortBE(buf, 0) == ZLIB_HEADER;
776    }

778    private void guessTransferSyntax() throws IOException {
779        byte[] b128 = new byte[128];
780        byte[] buf = buffer;
781        mark(132);
782        int rlen = read(b128);
783        if (rlen == 128) {
784            read(buf, 0, 4);
785            if (buf[0] == 'D' && buf[1] == 'I'
786                    && buf[2] == 'C' && buf[3] == 'M') {
787                preamble = b128.clone();
788                if (!markSupported()) {
789                    hasfmi = true;
790                    tsuid = UID.ExplicitVRLittleEndian;
791                    bigEndian = false;
792                    explicitVR = true;
793                    return;
794                }
795                mark(128);
796                read(b128);
797            }
798        }
799        if (rlen < 8
800                || !guessTransferSyntax(b128, rlen, false)
801                && !guessTransferSyntax(b128, rlen, true))
802            throw new DicomStreamException(NOT_A_DICOM_STREAM);
803        reset();
804        hasfmi = TagUtils.isFileMetaInformation(
805                ByteUtils.bytesToTag(b128, 0, bigEndian));
806    }

808    private boolean guessTransferSyntax(byte[] b128, int rlen, boolean bigEndian)
809            throws DicomStreamException {
810        int tag1 = ByteUtils.bytesToTag(b128, 0, bigEndian);
811        VR vr = ElementDictionary.vrOf(tag1, null);
812        if (vr == VR.UN)
813            return false;
814        if (ByteUtils.bytesToVR(b128, 4) == vr.code()) {
815            this.tsuid = bigEndian ? UID.ExplicitVRBigEndian 
816                                   : UID.ExplicitVRLittleEndian;
817            this.bigEndian = bigEndian;
818            this.explicitVR = true;
819            return true;
820        }
821        int len = ByteUtils.bytesToInt(b128, 4, bigEndian);
822        if (len < 0 || 8 + len > rlen)
823            return false;
824
825        if (bigEndian)
826            throw new DicomStreamException(IMPLICIT_VR_BIG_ENDIAN);
827
828        this.tsuid = UID.ImplicitVRLittleEndian;
829        this.bigEndian = false;
830        this.explicitVR = false;
831        return true;
832    }
