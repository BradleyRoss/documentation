039package org.dcm4che.io;
040
041import java.io.ByteArrayOutputStream;
042import java.io.IOException;
043import java.util.ArrayList;
044import java.util.LinkedList;
045
046import org.dcm4che.data.Attributes;
047import org.dcm4che.data.BulkDataLocator;
048import org.dcm4che.data.ElementDictionary;
049import org.dcm4che.data.Fragments;
050import org.dcm4che.data.PersonName;
051import org.dcm4che.data.Sequence;
052import org.dcm4che.data.VR;
053import org.dcm4che.util.Base64;
054import org.dcm4che.util.ByteUtils;
055import org.dcm4che.util.TagUtils;
056import org.xml.sax.SAXException;
057import org.xml.sax.helpers.DefaultHandler;
058
059/**
060 * @author Gunter Zeilinger <gunterze@gmail.com>
061 */
062public class ContentHandlerAdapter extends DefaultHandler {
063
064    private Attributes fmi;
065    private final boolean bigEndian;
066    private final LinkedList<Attributes> items = new LinkedList<Attributes>();
067    private final LinkedList<Sequence> seqs = new LinkedList<Sequence>();
068
069    private final ByteArrayOutputStream bout = new ByteArrayOutputStream(64);
070    private final char[] carry = new char[4];
071    private int carryLen;
072    private final StringBuilder sb = new StringBuilder(64);
073    private final ArrayList<String> values = new ArrayList<String>();
074    private PersonName pn;
075    private PersonName.Group pnGroup;
076    private int tag;
077    private String privateCreator;
078    private VR vr;
079    private String blkts;
080    private String blkuri;
081    private long blkoffset;
082    private int blklen;
083    private BulkDataLocator bulkDataLocator;
084    private Fragments dataFragments;
085    private boolean processCharacters;
086    private boolean base64;
087
088    public ContentHandlerAdapter(Attributes attrs) {
089        if (attrs == null)
090            throw new NullPointerException();
091        items.add(attrs);
092        bigEndian = attrs.bigEndian();
093    }
094
095    public Attributes getFileMetaInformation() {
096        return fmi;
097    }
098
099    @Override
100    public void startElement(String uri, String localName, String qName,
101            org.xml.sax.Attributes atts) throws SAXException {
102        switch (qName.charAt(0)) {
103        case 'A':
104            if (qName.equals("Alphabetic"))
105                startPNGroup(PersonName.Group.Alphabetic);
106            break;
107        case 'D':
108            if (qName.equals("DicomAttribute"))
109                startDicomAttribute(
110                        (int) Long.parseLong(atts.getValue("tag"), 16),
111                        atts.getValue("privateCreator"),
112                        atts.getValue("vr"));
113            else if (qName.equals("DataFragment"))
114                startDataFragment(Integer.parseInt(atts.getValue("number")));
115            break;
116        case 'F':
117            if (qName.equals("FamilyName"))
118                startText();
119            break;
120        case 'G':
121            if (qName.equals("GivenName"))
122                startText();
123            break;
124        case 'I':
125            if (qName.equals("Item"))
126                startItem(Integer.parseInt(atts.getValue("number")));
127            else if (qName.equals("Ideographic"))
128                startPNGroup(PersonName.Group.Ideographic);
129            break;
130        case 'L':
131            if (qName.equals("Length"))
132                startText();
133            break;
134        case 'M':
135            if (qName.equals("MiddleName"))
136                startText();
137            break;
138        case 'N':
139            if (qName.equals("NamePrefix") || qName.equals("NameSuffix"))
140                startText();
141            break;
142        case 'O':
143            if (qName.equals("Offset"))
144                startText();
145            break;
146        case 'P':
147            if (qName.equals("PersonName")) {
148                startPersonName(Integer.parseInt(atts.getValue("number")));
149            } else if (qName.equals("Phonetic"))
150                startPNGroup(PersonName.Group.Phonetic);
151            break;
152        case 'T':
153            if (qName.equals("TransferSyntax"))
154                startText();
155            break;
156        case 'U':
157            if (qName.equals("URI"))
158                startText();
159            break;
160        case 'V':
161            if (qName.equals("Value")) {
162                startValue(Integer.parseInt(atts.getValue("number")));
163                if (this.vr.isXMLBase64())
164                    startBase64();
165                else
166                    startText();
167            }
168            break;
169        }
170   }
171
172    private void startBase64() {
173        processCharacters = true;
174        base64 = true;
175        bout.reset();
176    }
177
178    private void startText() {
179        processCharacters = true;
180        base64 = false;
181        sb.setLength(0);
182    }
183
184    private void startDicomAttribute(int tag, String privateCreator,
185            String vr) {
186        this.tag = tag;
187        this.privateCreator = privateCreator;
188        this.vr = vr != null ? VR.valueOf(vr)
189                             : ElementDictionary.vrOf(tag, privateCreator);
190        if (this.vr == VR.SQ)
191            seqs.add(items.getLast().newSequence(privateCreator, tag, 10));
192    }
193
194    private void startDataFragment(int number) {
195        if (dataFragments == null)
196            dataFragments = items.getLast()
197                    .newFragments(privateCreator, tag, vr,  10);
198        while (dataFragments.size() < number-1)
199            dataFragments.add(ByteUtils.EMPTY_BYTES);
200    }
201
202    private void startItem(int number) {
203        Sequence seq = seqs.getLast();
204        while (seq.size() < number-1)
205            seq.add(new Attributes(0));
206        Attributes item = new Attributes();
207        seq.add(item);
208        items.add(item);
209    }
210
211    private void startValue(int number) {
212        while (values.size() < number-1)
213            values.add(null);
214    }
215
216    private void startPersonName(int number) {
217        startValue(number);
218        pn = new PersonName();
219    }
220
221    private void startPNGroup(PersonName.Group pnGroup) {
222        this.pnGroup = pnGroup;
223    }
224
225    @Override
226    public void characters(char[] ch, int offset, int len)
227            throws SAXException {
228        if (processCharacters)
229            if (base64)
230                try {
231                    if (carryLen != 0) {
232                        int copy = 4 - carryLen;
233                        System.arraycopy(ch, offset, carry, carryLen, copy);
234                        Base64.decode(carry, 0, 4, bout);
235                        offset += copy;
236                        len -= copy;
237                    }
238                    if ((carryLen = len & 3) != 0) {
239                        len -= carryLen;
240                        System.arraycopy(ch, offset + len, carry, 0, carryLen);
241                    }
242                    Base64.decode(ch, offset, len, bout);
243                } catch (IOException e) {
244                    throw new RuntimeException(e);
245                }
246            else
247                sb.append(ch, offset, len);
248    }
249
250    @Override
251    public void endElement(String uri, String localName, String qName)
252            throws SAXException {
253        switch (qName.charAt(0)) {
254        case 'B':
255            if (qName.equals("BulkDataLocator"))
256                endBulkDataLocator();
257            break;
258        case 'D':
259            if (qName.equals("DicomAttribute"))
260                endDicomAttribute();
261            else if (qName.equals("DataFragment"))
262                endDataFragment();
263            break;
264        case 'F':
265            if (qName.equals("FamilyName"))
266                endPNComponent(PersonName.Component.FamilyName);
267            break;
268        case 'G':
269            if (qName.equals("GivenName"))
270                endPNComponent(PersonName.Component.GivenName);
271            break;
272        case 'I':
273            if (qName.equals("Item"))
274                endItem();
275            break;
276        case 'L':
277            if (qName.equals("Length"))
278                blklen = Integer.parseInt(getString());
279            break;
280        case 'M':
281            if (qName.equals("MiddleName"))
282                endPNComponent(PersonName.Component.MiddleName);
283            break;
284        case 'N':
285            if (qName.equals("NamePrefix"))
286                endPNComponent(PersonName.Component.NamePrefix);
287            else if (qName.equals("NameSuffix"))
288                endPNComponent(PersonName.Component.NameSuffix);
289            break;
290        case 'O':
291            if (qName.equals("Offset"))
292                blkoffset = Long.parseLong(getString());
293            break;
294        case 'P':
295            if (qName.equals("PersonName"))
296                endPersonName();
297            break;
298        case 'T':
299            if (qName.equals("TransferSyntax")) {
300                blkts = getString();
301            }
302            break;
303        case 'U':
304            if (qName.equals("URI")) {
305                blkuri = getString();
306            }
307            break;
308        case 'V':
309            if (qName.equals("Value")) {
310                endValue();
311            }
312            break;
313        }
314        processCharacters = false;
315    }
316
317    @Override
318    public void endDocument() throws SAXException {
319        if (fmi != null)
320            fmi.trimToSize();
321        items.getFirst().trimToSize();
322    }
323
324    private void endDataFragment() {
325        if (bulkDataLocator != null) {
326            dataFragments.add(bulkDataLocator);
327            bulkDataLocator = null;
328        } else {
329            dataFragments.add(getBytes());
330        }
331    }
332
333    private void endDicomAttribute() {
334        if (vr == VR.SQ) {
335            seqs.removeLast().trimToSize();
336            return;
337        }
338        if (dataFragments != null) {
339            dataFragments.trimToSize();
340            dataFragments = null;
341            return;
342        }
343        Attributes attrs = items.getLast();
344        if (TagUtils.isFileMetaInformation(tag)) {
345            if (fmi == null)
346                fmi = new Attributes();
347            attrs = fmi;
348        }
349        if (bulkDataLocator != null) {
350            attrs.setValue(privateCreator, tag, vr, bulkDataLocator);
351            bulkDataLocator = null;
352        } else if (base64) {
353            attrs.setBytes(privateCreator, tag, vr, getBytes());
354        } else {
355            attrs.setString(privateCreator, tag, vr, getStrings());
356        }
357    }
358
359    private void endBulkDataLocator() {
360        bulkDataLocator = new BulkDataLocator(blkuri, blkts, blkoffset, blklen);
361    }
362
363    private void endItem() {
364        items.removeLast().trimToSize();
365        vr = VR.SQ;
366    }
367
368    private void endPersonName() {
369        values.add(pn.toString());
370        pn = null;
371    }
372
373    private void endValue() {
374        if (!base64)
375            values.add(getString());
376    }
377
378    private void endPNComponent(PersonName.Component pnComp) {
379        pn.set(pnGroup, pnComp, getString());
380    }
381
382    private String getString() {
383        return sb.toString();
384    }
385
386    private byte[] getBytes() {
387        byte[] b = bout.toByteArray();
388        return bigEndian ? vr.toggleEndian(b, false) : b;
389    }
390
391    private String[] getStrings() {
392        try {
393            return values.toArray(new String[values.size()]);
394        } finally {;
395            values.clear();
396        }
397    }
