039package org.dcm4che.soundex;
040
041/**
042 * @author Gunter Zeilinger <gunterze@gmail.com>
043 */
044public class Soundex implements FuzzyStr {
045
046    protected static final java.lang.String MAP_6 =
047        // A BCD  E FG  H   I JKLMN  O PQRST  U V  W X  Y Z
048        "\000123\00012\001\00022455\00012623\0001\0012\0002";
049    protected static final java.lang.String MAP_9 = 
050        // A BCD  E FG  H   I JKLMN  O PQRST  U V  W X  Y Z
051        "\000136\00024\001\00043788\00015936\0002\0015\0005";
052
053    private final boolean encodeFirst;
054    private final int codeLength;
055    private final int padLength;
056    private final char[] map;
057
058    public Soundex() {
059        this(false, 4, 4, MAP_6);
060    }
061
062    public Soundex(boolean encodeFirst, int codeLength, int padLength, String map) {
063        this.encodeFirst = encodeFirst;
064        this.codeLength = codeLength;
065        this.padLength = padLength;
066        this.map = map.toCharArray();
067    }
068
069    @Override
070    public String toFuzzy(String s) {
071        if (s == null || s.length() == 0)
072            return "";
073
074        char[] in = s.toCharArray();
075        char[] out = in.length < padLength ? new char[padLength] : in;
076        int i = 0;
077        int j = 0;
078        char prevout = 0;
079        if (!encodeFirst) {
080            while (!Character.isLetter(in[i]))
081                if (++i >= in.length)
082                    return "";
083            prevout = map(out[j++] = Character.toUpperCase(in[i++]));
084        }
085
086        char curout = 0;
087        for (; i < in.length && j < codeLength; i++) {
088            curout = map(in[i]);
089            switch (curout) {
090            case '\0':
091                prevout = curout;
092            case '\1':
093                break;
094            default:
095                if (curout != prevout)
096                    out[j++] = prevout = curout;
097            }
098        }
099        while (j < padLength)
100            out[j++] = '0';
101        return new String(out, 0, j);
102    }
103
104    private char map(char c) {
105        try {
106            return map[c >= 'a' ? c - 'a' : c - 'A'];
107        } catch (IndexOutOfBoundsException e) {
108            return (c == '?' || c == '?' || c == '?') ? map['c' - 'a'] : '\u0000';
109        }
110    }
111
112    public static void main(String[] args) {
113        Soundex inst = new Soundex();
114        for (String arg : args)
115            System.out.println(inst.toFuzzy(arg));
116    }
117
118}
