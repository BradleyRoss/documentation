038package org.dcm4che.net;
039
040import java.io.Serializable;
041
042/**
043 * @author Gunter Zeilinger <gunterze@gmail.com>
044 *
045 */
046public class AEExtension implements Serializable {
047
048    private static final long serialVersionUID = -8287458555186708798L;
049
050    protected ApplicationEntity ae;
051
052    public final ApplicationEntity getApplicationEntity() {
053        return ae;
054    }
055
056    void setApplicationEntity(ApplicationEntity ae) {
057        if (ae != null && this.ae != null)
058            throw new IllegalStateException(
059                    "already owned by AE: " + ae.getAETitle());
060        this.ae = ae;
061    }
062
063    public void reconfigure(AEExtension from) { }
064}
