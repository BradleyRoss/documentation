070 * Network AE will have the 16 same functional capability regardless of the

073 * are 18 functional differences based on other internal structures, then these

129        if (device != null) {

131                throw new IllegalStateException("already owned by " + 
132                        this.device.getDeviceName());
133            for (Connection conn : conns)
134                if (conn.getDevice() != device)

136                            device.getDeviceName());
137        }
138        this.device = device;

395    public void addConnection(Connection conn) {
396        if (conn.getProtocol() != Connection.Protocol.DICOM)
397            throw new IllegalArgumentException(
398                    "protocol != DICOM - " + conn.getProtocol());
399
400        if (device != null && device != conn.getDevice())
401            throw new IllegalStateException(conn + " not contained by Device: " + 
402                    device.getDeviceName());
403        conns.add(conn);
404    }

414    public TransferCapability addTransferCapability(TransferCapability tc) {
415        tc.setApplicationEntity(this);
416        TransferCapability prev = (tc.getRole() == TransferCapability.Role.SCU
417                ? scuTCs : scpTCs).put(tc.getSopClass(), tc);
418        if (prev != null && prev != tc)
419            prev.setApplicationEntity(null);
420        return prev;
421    }

450    protected PresentationContext negotiate(AAssociateRQ rq, AAssociateAC ac,
451           PresentationContext rqpc) {
452       String as = rqpc.getAbstractSyntax();
453       TransferCapability tc = roleSelection(rq, ac, as);
454       int pcid = rqpc.getPCID();
455       if (tc == null)
456           return new PresentationContext(pcid,
457                   PresentationContext.ABSTRACT_SYNTAX_NOT_SUPPORTED,
458                   rqpc.getTransferSyntax());
459
460       for (String ts : rqpc.getTransferSyntaxes())
461           if (tc.containsTransferSyntax(ts)) {
462               byte[] info = negotiate(rq.getExtNegotiationFor(as), tc);
463               if (info != null)
464                   ac.addExtendedNegotiation(new ExtendedNegotiation(as, info));
465               return new PresentationContext(pcid,
466                       PresentationContext.ACCEPTANCE, ts);
467           }
468
469       return new PresentationContext(pcid,
470                PresentationContext.TRANSFER_SYNTAX_NOT_SUPPORTED,
471                rqpc.getTransferSyntax());
472    }

474    private TransferCapability roleSelection(AAssociateRQ rq,
475            AAssociateAC ac, String asuid) {
476        RoleSelection rqrs = rq.getRoleSelectionFor(asuid);
477        if (rqrs == null)
478            return getTC(scpTCs, asuid, rq);
479
480        RoleSelection acrs = ac.getRoleSelectionFor(asuid);
481        if (acrs != null)
482            return getTC(acrs.isSCU() ? scpTCs : scuTCs, asuid, rq);
483
484        TransferCapability tcscu = null;
485        TransferCapability tcscp = null;
486        boolean scu = rqrs.isSCU()
487                && (tcscp = getTC(scpTCs, asuid, rq)) != null;
488        boolean scp = rqrs.isSCP()
489                && (tcscu = getTC(scuTCs, asuid, rq)) != null;
490        ac.addRoleSelection(new RoleSelection(asuid, scu, scp));
491        return scu ? tcscp : tcscu;
492    }
493
494    private TransferCapability getTC(HashMap<String, TransferCapability> tcs,
495            String asuid, AAssociateRQ rq) {
496        TransferCapability tc = tcs.get(asuid);
497        if (tc != null)
498            return tc;
499
500        CommonExtendedNegotiation commonExtNeg =
501                rq.getCommonExtendedNegotiationFor(asuid);
502        if (commonExtNeg != null) {
503            for (String cuid : commonExtNeg.getRelatedGeneralSOPClassUIDs()) {
504                tc = tcs.get(cuid);
505                if (tc != null)
506                    return tc;
507            }
508            tc = tcs.get(commonExtNeg.getServiceClassUID());
509            if (tc != null)
510                return tc;
511        }
512
513        return tcs.get("*");
514    }

516    private byte[] negotiate(ExtendedNegotiation exneg, TransferCapability tc) {
517        if (exneg == null)
518            return null;
519
520        StorageOptions storageOptions = tc.getStorageOptions();
521        if (storageOptions != null)
522            return storageOptions.toExtendedNegotiationInformation();
523
524        EnumSet<QueryOption> queryOptions = tc.getQueryOptions();
525        if (queryOptions != null) {
526            EnumSet<QueryOption> commonOpts = QueryOption.toOptions(exneg);
527            commonOpts.retainAll(queryOptions);
528            return QueryOption.toExtendedNegotiationInformation(commonOpts);
529        }
530        return null;
531    }

533    public Association connect(Connection local, Connection remote, AAssociateRQ rq)
534            throws IOException, InterruptedException, IncompatibleConnectionException, GeneralSecurityException {
535        checkDevice();
536        checkInstalled();
537        if (rq.getCallingAET() == null)
538            rq.setCallingAET(aet);
539        rq.setMaxOpsInvoked(local.getMaxOpsInvoked());
540        rq.setMaxOpsPerformed(local.getMaxOpsPerformed());
541        rq.setMaxPDULength(local.getReceivePDULength());
542        Socket sock = local.connect(remote);
543        Association as = new Association(this, local, sock);
544        as.write(rq);
545        as.waitForLeaving(State.Sta5);
546        return as;
547    }

554    public Connection findCompatibelConnection(Connection remoteConn)
555            throws IncompatibleConnectionException {
556        for (Connection conn : conns)
557            if (conn.isInstalled() && conn.isCompatible(remoteConn))
558                return conn;
559        throw new IncompatibleConnectionException(
560                "No compatible connection to " + remoteConn + " available on " + this);
561    }
562
563    public CompatibleConnection findCompatibelConnection(ApplicationEntity remote)
564            throws IncompatibleConnectionException {
565        for (Connection remoteConn : remote.conns)
566            if (remoteConn.isInstalled() && remoteConn.isServer())
567                for (Connection conn : conns)
568                    if (conn.isInstalled() && conn.isCompatible(remoteConn))
569                        return new CompatibleConnection(conn, remoteConn);
570        throw new IncompatibleConnectionException(
571                "No compatible connection to " + remote + " available on " + this);
572    }
573
574    public Association connect(ApplicationEntity remote, AAssociateRQ rq)
575        throws IOException, InterruptedException, IncompatibleConnectionException, GeneralSecurityException {
576        CompatibleConnection cc = findCompatibelConnection(remote);
577        if (rq.getCalledAET() == null)
578            rq.setCalledAET(remote.getAETitle());
579        return connect(cc.getLocalConnection(), cc.getRemoteConnection(), rq);
580    }

587    public StringBuilder promptTo(StringBuilder sb, String indent) {
588        String indent2 = indent + "  ";
589        StringUtils.appendLine(sb, indent, "ApplicationEntity[title: ", aet);
590        StringUtils.appendLine(sb, indent2,"desc: ", description);
591        StringUtils.appendLine(sb, indent2,"acceptor: ", acceptor);
592        StringUtils.appendLine(sb, indent2,"initiator: ", initiator);
593        StringUtils.appendLine(sb, indent2,"installed: ", getInstalled());
594        for (Connection conn : conns)
595            conn.promptTo(sb, indent2).append(StringUtils.LINE_SEPARATOR);
596        for (TransferCapability tc : getTransferCapabilities())
597            tc.promptTo(sb, indent2).append(StringUtils.LINE_SEPARATOR);
598        return sb.append(indent).append(']');
599    }

601    void reconfigure(ApplicationEntity src) {

604        reconfigureTransferCapabilities(src);
605        reconfigureAEExtensions(src);
606    }

608    private void reconfigureTransferCapabilities(ApplicationEntity src) {
609        scuTCs.clear();
610        scuTCs.putAll(src.scuTCs);
611        scpTCs.clear();
612        scpTCs.putAll(src.scpTCs);
613    }

615    private void reconfigureAEExtensions(ApplicationEntity from) {
616        for (Iterator<AEExtension> it = extensions.iterator();
617                it.hasNext();) {
618            AEExtension ext = it.next();
619            if (from.getAEExtension(ext.getClass()) == null) {
620                it.remove();
621            }
622        }
623        for (AEExtension src : from.extensions) {
624            Class<? extends AEExtension> clazz = src.getClass();
625            AEExtension ext = getAEExtension(clazz);
626            if (ext == null)
627                try {
628                    addAEExtension(ext = clazz.newInstance());
629                } catch (Exception e) {
630                    throw new RuntimeException(
631                            "Failed to instantiate " + clazz.getName(), e);
632                }
633            ext.reconfigure(src);
634        }
635    }

673    @SuppressWarnings("unchecked")
674    public <T extends AEExtension> T getAEExtension(Class<T> clazz) {
675        for (AEExtension ext : extensions)

677                return (T) ext;
