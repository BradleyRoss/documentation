039package org.dcm4che.net.service;
040
041import java.io.IOException;
042import java.util.HashMap;
043
044import org.dcm4che.data.Attributes;
045import org.dcm4che.data.Tag;
046import org.dcm4che.net.Association;
047import org.dcm4che.net.AssociationStateException;
048import org.dcm4che.net.Dimse;
049import org.dcm4che.net.DimseRQHandler;
050import org.dcm4che.net.PDVInputStream;
051import org.dcm4che.net.Status;
052import org.dcm4che.net.pdu.CommonExtendedNegotiation;
053import org.dcm4che.net.pdu.PresentationContext;
054
055/**
056 * @author Gunter Zeilinger <gunterze@gmail.com>
057 *
058 */
059public class DicomServiceRegistry implements DimseRQHandler {
060
061    private final HashMap<String, DicomService> services =
062            new HashMap<String, DicomService>();
063
064    public synchronized void addDicomService(DicomService service) {
065        for (String uid : service.getSOPClasses())
066            services.put(uid, service);
067    }
068
069    public synchronized void removeDicomService(DicomService service) {
070        for (String uid : service.getSOPClasses())
071            services.remove(uid);
072    }
073
074    @Override
075    public void onDimseRQ(Association as, PresentationContext pc,
076            Dimse dimse, Attributes cmd, PDVInputStream data) throws IOException {
077        try {
078            lookupService(as, dimse, cmd).onDimseRQ(as, pc, dimse, cmd, data);
079        } catch (DicomServiceException e) {
080            Attributes rsp = e.mkRSP(dimse.commandFieldOfRSP(), cmd.getInt(Tag.MessageID, 0));
081            try {
082                as.writeDimseRSP(pc, rsp, e.getDataset());
083            } catch (AssociationStateException ase) {
084                Association.LOG.warn("{} << DIMSE-RSP failed: {}",
085                        as, ase.getMessage());
086            }
087        }
088    }
089
090    private DicomService lookupService(Association as, Dimse dimse, Attributes cmd)
091            throws DicomServiceException {
092        String cuid = cmd.getString(dimse.tagOfSOPClassUID());
093        if (cuid == null)
094            throw new DicomServiceException(Status.MistypedArgument);
095
096        DicomService service = services.get(cuid);
097        if (service != null)
098            return service;
099
100        if (dimse == Dimse.C_STORE_RQ) {
101            CommonExtendedNegotiation commonExtNeg = as
102                    .getCommonExtendedNegotiationFor(cuid);
103            if (commonExtNeg != null) {
104                for (String uid : commonExtNeg.getRelatedGeneralSOPClassUIDs()) {
105                    service = services.get(uid);
106                    if (service != null)
107                        return service;
108                }
109                service = services.get(commonExtNeg.getServiceClassUID());
110                if (service != null)
111                    return service;
112            }
113            service = services.get("*");
114            if (service != null)
115                return service;
116        }
117        throw new DicomServiceException(dimse.isCService()
118                ? Status.SOPclassNotSupported
119                : Status.NoSuchSOPclass);
120    }
121
122
123    @Override
124    public void onClose(Association as) {
125        for (DicomService service : services.values())
126            service.onClose(as);
127    }
128
129}
