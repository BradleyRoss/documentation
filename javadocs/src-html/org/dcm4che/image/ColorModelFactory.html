039package org.dcm4che.image;
040
041import java.awt.Transparency;
042import java.awt.color.ColorSpace;
043import java.awt.color.ICC_ColorSpace;
044import java.awt.color.ICC_Profile;
045import java.awt.image.ColorModel;
046import java.awt.image.ComponentColorModel;
047
048import org.dcm4che.data.Attributes;
049import org.dcm4che.data.Tag;
050
051/**
052 * @author Gunter Zeilinger <gunterze@gmail.com>
053 *
054 */
055public class ColorModelFactory {
056
057    public static ColorModel createMonochromeColorModel(int bits, int dataType) {
058        return new ComponentColorModel(
059                ColorSpace.getInstance(ColorSpace.CS_GRAY),
060                new int[] { bits },
061                false, // hasAlpha
062                false, // isAlphaPremultiplied
063                Transparency.OPAQUE,
064                dataType);
065    }
066
067    public static ColorModel createPaletteColorModel(int bits, int dataType,
068            Attributes ds) {
069        return new PaletteColorModel(bits, dataType, createRGBColorSpace(ds), ds);
070    }
071
072    public static ColorModel createRGBColorModel(int bits, int dataType,
073            Attributes ds) {
074        return new ComponentColorModel(
075                createRGBColorSpace(ds),
076                new int[] { bits, bits, bits },
077                false, // hasAlpha
078                false, // isAlphaPremultiplied
079                Transparency.OPAQUE,
080                dataType);
081    }
082
083
084    public static ColorModel createYBRFullColorModel(int bits, int dataType,
085            Attributes ds) {
086        return new ComponentColorModel(
087                new YBRColorSpace(createRGBColorSpace(ds),  YBR.FULL),
088                new int[] { bits, bits, bits },
089                false, // hasAlpha
090                false, // isAlphaPremultiplied
091                Transparency.OPAQUE,
092                dataType);
093    }
094
095    public static ColorModel createYBRColorModel(int bits, int dataType,
096            Attributes ds, YBR ybr, ColorSubsampling subsampling) {
097        return new SampledComponentColorModel(
098                new YBRColorSpace(createRGBColorSpace(ds), ybr),
099                subsampling);
100    }
101
102    private static ColorSpace createRGBColorSpace(Attributes ds) {
103        return createRGBColorSpace(ds.getSafeBytes(Tag.ICCProfile));
104    }
105
106    private static ColorSpace createRGBColorSpace(byte[] iccProfile) {
107        if (iccProfile != null)
108            return new ICC_ColorSpace(ICC_Profile.getInstance(iccProfile));
109
110        return ColorSpace.getInstance(ColorSpace.CS_sRGB);
111    }
112
113}
