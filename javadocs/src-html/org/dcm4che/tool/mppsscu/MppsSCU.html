285    public final void setDiscontinuationReason(String codeValue) {
286        if (codes == null)
287            throw new IllegalStateException("codes not initialized");
288        String codeMeaning = codes.getProperty(codeValue);
289        if (codeMeaning == null)
290            throw new IllegalArgumentException("undefined code value: "
291                        + codeValue);
292        int endDesignator = codeValue.indexOf('-');
293        Attributes attrs = new Attributes(3);
294        attrs.setString(Tag.CodeValue, VR.SH,
295                endDesignator >= 0
296                    ? codeValue.substring(endDesignator + 1)
297                    : codeValue);
298        attrs.setString(Tag.CodingSchemeDesignator, VR.SH,
299                endDesignator >= 0
300                    ? codeValue.substring(0, endDesignator)
301                    : "DCM");
302        attrs.setString(Tag.CodeMeaning, VR.LO, codeMeaning);
303        this.discontinuationReason = attrs;
304    }

316    @SuppressWarnings("unchecked")
317    public static void main(String[] args) {
318        try {
319            CommandLine cl = parseComandLine(args);
320            Device device = new Device("mppsscu");
321            Connection conn = new Connection();
322            device.addConnection(conn);
323            ApplicationEntity ae = new ApplicationEntity("MPPSSCU");
324            device.addApplicationEntity(ae);
325            ae.addConnection(conn);
326            final MppsSCU main = new MppsSCU(ae);
327            configureMPPS(main, cl);
328            CLIUtils.configureConnect(main.remote, main.rq, cl);
329            CLIUtils.configureBind(conn, main.ae, cl);
330            CLIUtils.configure(conn, cl);
331            main.remote.setTlsProtocols(conn.getTlsProtocols());
332            main.remote.setTlsCipherSuites(conn.getTlsCipherSuites());
333            main.setTransferSyntaxes(CLIUtils.transferSyntaxesOf(cl));
334            main.setAttributes(new Attributes());
335            CLIUtils.addAttributes(main.attrs, cl.getOptionValues("s"));
336            main.setUIDSuffix(cl.getOptionValue("uid-suffix"));
337            List<String> argList = cl.getArgList();
338            boolean echo = argList.isEmpty();
339            if (!echo) {
340                System.out.println(rb.getString("scanning"));
341                DicomFiles.scan(argList, new DicomFiles.Callback() {
342                    
343                    @Override
344                    public boolean dicomFile(File f, Attributes fmi, 
345                            long dsPos, Attributes ds) {
346                        if (UID.ModalityPerformedProcedureStepSOPClass.equals(
347                                fmi.getString(Tag.MediaStorageSOPClassUID))) {
348                            return main.addMPPS(
349                                    fmi.getString(Tag.MediaStorageSOPInstanceUID),
350                                    ds);
351                        }
352                        return main.addInstance(ds);
353                    }
354                });
355            }
356            ExecutorService executorService =
357                    Executors.newSingleThreadExecutor();
358            ScheduledExecutorService scheduledExecutorService =
359                    Executors.newSingleThreadScheduledExecutor();
360            device.setExecutor(executorService);
361            device.setScheduledExecutor(scheduledExecutorService);
362            try {
363                main.open();
364                if (echo)
365                    main.echo();
366                else {
367                    main.createMpps();
368                    main.updateMpps();
369                }
370            } finally {
371                main.close();
372                executorService.shutdown();
373                scheduledExecutorService.shutdown();
374            }
375        } catch (ParseException e) {
376            System.err.println("mppsscu: " + e.getMessage());
377            System.err.println(rb.getString("try"));
378            System.exit(2);
379        } catch (Exception e) {
380            System.err.println("mppsscu: " + e.getMessage());
381            e.printStackTrace();
382            System.exit(2);
383        }
384    }

386    private static void configureMPPS(MppsSCU main, CommandLine cl)
387            throws Exception {
388        main.setNewPPSID(cl.hasOption("ppsid-new"));
389        main.setPPSUID(cl.getOptionValue("ppsuid"));
390        main.setPPSID(cl.getOptionValue("ppsid"));
391        if (cl.hasOption("ppsid-start"))
392            main.setPPSIDStart(Integer.parseInt(cl.getOptionValue("ppsid-start")));
393        main.setPPSIDFormat(cl.getOptionValue("ppsid-format", "PPS-0000000000"));
394        main.setProtocolName(cl.getOptionValue("protocol", "UNKNOWN"));
395        if (cl.hasOption("archive"))
396            main.setArchiveRequested(cl.getOptionValue("archive"));
397        main.setCodes(CLIUtils.loadProperties(
398                cl.getOptionValue("code-config", "resource:code.properties"),
399                null));
400        if (cl.hasOption("dc"))
401            main.setFinalStatus(DISCONTINUED);
402        if (cl.hasOption("dc-reason"))
403            main.setDiscontinuationReason(cl.getOptionValue("dc-reason"));
404    }

508    private void createMpps(final MppsWithIUID mppsWithUID)
509            throws IOException, InterruptedException {
510        final String iuid = mppsWithUID.iuid;
511        Attributes mpps = mppsWithUID.mpps;
512        mppsWithUID.mpps = new Attributes(mpps, FINAL_MPPS_TOP_LEVEL_ATTRS);
513        mpps.setString(Tag.PerformedProcedureStepStatus, VR.CS, IN_PROGRESS);
514        for (int tag : CREATE_MPPS_TOP_LEVEL_EMPTY_ATTRS)
515            mpps.setNull(tag, dict.vrOf(tag));
516
517        DimseRSPHandler rspHandler = new DimseRSPHandler(as.nextMessageID()) {
518
519            @Override
520            public void onDimseRSP(Association as, Attributes cmd, Attributes data) {
521                switch(cmd.getInt(Tag.Status, -1)) {
522                    case Status.Success:
523                    case Status.AttributeListError:
524                    case Status.AttributeValueOutOfRange:
525                        mppsWithUID.iuid = cmd.getString(
526                                Tag.AffectedSOPInstanceUID, mppsWithUID.iuid);
527                        created.add(mppsWithUID);
528                }
529                super.onDimseRSP(as, cmd, data);
530            }
531
532        };
533        as.ncreate(UID.ModalityPerformedProcedureStepSOPClass,
534                iuid, mpps, null, rspHandler);
535    }

549    public boolean addInstance(Attributes inst) {
550        CLIUtils.updateAttributes(inst, attrs, uidSuffix);
551        String suid = inst.getString(Tag.StudyInstanceUID);
552        if (suid == null)
553            return false;
554        MppsWithIUID mppsWithIUID = map.get(suid);
555        if (mppsWithIUID == null)
556            map.put(suid, mppsWithIUID = new MppsWithIUID(ppsuid(null), createMPPS(inst)));
557        updateMPPS(mppsWithIUID.mpps, inst);
558        return true;
559    }

566    private String ppsuid(String defval) {
567        if (ppsuid == null)
568            return defval;
569        
570        int size = map.size();
571        switch (size) {
572        case 0:
573            return ppsuid;
574        case 1:
575            map.values().iterator().next().iuid += ".1";
576        }
577        return ppsuid + '.' + (size + 1);
578    }

581    private String mkPPSID() {
582        if (ppsid != null)
583            return ppsid;
584        String id = ppsidFormat.format(serialNo);
585        if (++serialNo < 0)
586            serialNo = 0;
587        return id;

590    private Attributes createMPPS(Attributes inst) {
591        Attributes mpps = new Attributes();
592        mpps.setString(Tag.PerformedStationAETitle, VR.AE, ae.getAETitle());
593        mpps.setString(Tag.PerformedProcedureStepStartDate, VR.DA,
594                inst.getString(Tag.StudyDate, ppsStartDate));
595        mpps.setString(Tag.PerformedProcedureStepStartTime, VR.TM,
596                inst.getString(Tag.StudyTime, ppsStartTime));
597        for (int tag : MPPS_TOP_LEVEL_TYPE_2_ATTRS)
598            mpps.setNull(tag, dict.vrOf(tag));
599        mpps.addSelected(inst, MPPS_TOP_LEVEL_ATTRS);
600        if (newPPSID || !mpps.containsValue(Tag.PerformedProcedureStepID))
601            mpps.setString(Tag.PerformedProcedureStepID, VR.CS, mkPPSID());
602        mpps.setString(Tag.PerformedProcedureStepEndDate, VR.DA,
603                mpps.getString(Tag.PerformedProcedureStepStartDate));
604        mpps.setString(Tag.PerformedProcedureStepEndTime, VR.TM,
605                mpps.getString(Tag.PerformedProcedureStepStartTime));
606        mpps.setString(Tag.PerformedProcedureStepStatus, VR.CS, finalStatus);
607        Sequence dcrSeq = mpps.newSequence(Tag.PerformedProcedureStepDiscontinuationReasonCodeSequence, 1);
608        if (discontinuationReason != null)
609            dcrSeq.add(new Attributes(discontinuationReason));
610
611        Sequence raSeq = inst.getSequence(Tag.RequestAttributesSequence);
612        if (raSeq == null || raSeq.isEmpty()) {
613            Sequence ssaSeq = 
614                    mpps.newSequence(Tag.ScheduledStepAttributesSequence, 1);
615            Attributes ssa = new Attributes();
616            ssaSeq.add(ssa);
617            for (int tag : SSA_TYPE_2_ATTRS)
618                ssa.setNull(tag, dict.vrOf(tag));
619            ssa.addSelected(inst, SSA_ATTRS);
620        } else {
621            Sequence ssaSeq =
622                    mpps.newSequence(Tag.ScheduledStepAttributesSequence, raSeq.size());
623            for (Attributes ra : raSeq) {
624                Attributes ssa = new Attributes();
625                ssaSeq.add(ssa);
626                for (int tag : SSA_TYPE_2_ATTRS)
627                    ssa.setNull(tag, dict.vrOf(tag));
628                ssa.addSelected(inst, SSA_ATTRS);
629                ssa.addSelected(ra, SSA_ATTRS);
630            }
631        }
632        mpps.newSequence(Tag.PerformedSeriesSequence, 1);
633        return mpps ;
634    }

636    private void updateMPPS(Attributes mpps, Attributes inst) {
637        String endTime = inst.getString(Tag.AcquisitionTime);
638        if (endTime == null) {
639            endTime = inst.getString(Tag.ContentTime);
640            if (endTime == null)
641                endTime = inst.getString(Tag.SeriesTime);
642        }
643        if (endTime != null && endTime.compareTo(
644                mpps.getString(Tag.PerformedProcedureStepEndTime)) > 0)
645            mpps.setString(Tag.PerformedProcedureStepEndTime, VR.TM, endTime);
646        Sequence prefSeriesSeq = mpps.getSequence(Tag.PerformedSeriesSequence);
647        Attributes prefSeries = getPerfSeries(prefSeriesSeq, inst);
648        Sequence refSOPSeq = prefSeries.getSequence(Tag.ReferencedImageSequence);
649        Attributes refSOP = new Attributes();
650        refSOPSeq.add(refSOP);
651        refSOP.setString(Tag.ReferencedSOPClassUID, VR.UI,
652                inst.getString(Tag.SOPClassUID));
653        refSOP.setString(Tag.ReferencedSOPInstanceUID, VR.UI,
654                inst.getString(Tag.SOPInstanceUID));
655    }
656
657    private Attributes getPerfSeries(Sequence prefSeriesSeq, Attributes inst) {
658        String suid = inst.getString(Tag.SeriesInstanceUID);
659        for (Attributes prefSeries : prefSeriesSeq) {
660            if (suid.equals(prefSeries.getString(Tag.SeriesInstanceUID)))
661                return prefSeries;
662        }
663        Attributes prefSeries = new Attributes();
664        prefSeriesSeq.add(prefSeries);
665        for (int tag : PERF_SERIES_TYPE_2_ATTRS)
666            prefSeries.setNull(tag, dict.vrOf(tag));
667        prefSeries.setString(Tag.ProtocolName, VR.LO, protocolName);
668        prefSeries.addSelected(inst, PERF_SERIES_ATTRS);
669        prefSeries.newSequence(Tag.ReferencedImageSequence, 10);
670        if (archiveRequested != null)
671            prefSeries.setString(Tag.ArchiveRequested, VR.CS, archiveRequested);
672        return prefSeries;
673    }
