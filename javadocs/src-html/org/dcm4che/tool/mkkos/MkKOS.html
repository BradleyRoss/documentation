158    @SuppressWarnings("unchecked")
159    public static void main(String[] args) throws Exception {
160        try {
161            CommandLine cl = parseComandLine(args);
162            final MkKOS main = new MkKOS();
163            configure(main, cl);
164            System.out.println(rb.getString("scanning"));
165            DicomFiles.scan(cl.getArgList(), new DicomFiles.Callback() {
166                
167                @Override
168                public boolean dicomFile(File f, Attributes fmi,
169                        long dsPos, Attributes ds) {
170                    return main.addInstance(ds);
171                }
172            });
173            System.out.println();
174            main.writeKOS();
175            System.out.println(
176                    MessageFormat.format(rb.getString("stored"), main.fname));
177        } catch (ParseException e) {
178            System.err.println("mkkos: " + e.getMessage());
179            System.err.println(rb.getString("try"));
180            System.exit(2);
181        }
182    }

184    private static CommandLine parseComandLine(String[] args)
185            throws ParseException{
186        Options opts = new Options();
187        CLIUtils.addCommonOptions(opts);
188        addOptions(opts);
189        CommandLine cl = CLIUtils.parseComandLine(args, opts, rb, MkKOS.class);
190        if (cl.getArgList().isEmpty())
191            throw new ParseException(rb.getString("missing"));
192        return cl;
193    }

265    private static void configure(MkKOS main, CommandLine cl) throws Exception {
266        main.setCodes(CLIUtils.loadProperties(
267                cl.getOptionValue("code-config", "resource:code.properties"),
268                null));
269        main.setDocumentTitle(main.toCodeItem(documentTitleOf(cl)));
270        if (cl.hasOption("modifier"))
271            main.setDocumentTitleModifier(
272                    main.toCodeItem(cl.getOptionValue("modifier")));
273        main.setKeyObjectDescription(cl.getOptionValue("desc"));
274        main.setSeriesNumber(cl.getOptionValue("series-no", "999"));
275        main.setInstanceNumber(cl.getOptionValue("inst-no", "1"));
276        main.setOutputFile(outputFileOf(cl));
277        main.setNoFileMetaInformation(cl.hasOption("F"));
278        main.setTransferSyntax(cl.getOptionValue("t", UID.ExplicitVRLittleEndian));
279        main.setEncodingOptions(CLIUtils.encodingOptionsOf(cl));
280        CLIUtils.addAttributes(main.attrs, cl.getOptionValues("s"));
281        main.setUIDSuffix(cl.getOptionValue("uid-suffix"));
282    }

296    public Attributes toCodeItem(String codeValue) {
297        if (codes == null)
298            throw new IllegalStateException("codes not initialized");
299        String codeMeaning = codes.getProperty(codeValue);
300        if (codeMeaning == null)
301            throw new IllegalArgumentException("undefined code value: "
302                        + codeValue);
303        int endDesignator = codeValue.indexOf('-');
304        Attributes attrs = new Attributes(3);
305        attrs.setString(Tag.CodeValue, VR.SH,
306                endDesignator >= 0
307                    ? codeValue.substring(endDesignator + 1)
308                    : codeValue);
309        attrs.setString(Tag.CodingSchemeDesignator, VR.SH,
310                endDesignator >= 0
311                    ? codeValue.substring(0, endDesignator)
312                    : "DCM");
313        attrs.setString(Tag.CodeMeaning, VR.LO, codeMeaning);
314        return attrs;
315    }

317    public boolean addInstance(Attributes inst) {
318        CLIUtils.updateAttributes(inst, attrs, uidSuffix);
319        String studyIUID = inst.getString(Tag.StudyInstanceUID);
320        String seriesIUID = inst.getString(Tag.SeriesInstanceUID);
321        String iuid = inst.getString(Tag.SOPInstanceUID);
322        String cuid = inst.getString(Tag.SOPClassUID);
323        if (studyIUID == null || seriesIUID == null || iuid == null || cuid == null)
324            return false;
325        if (kos == null)
326            kos = createKOS(inst);
327        refSOPSeq(studyIUID, seriesIUID).add(refSOP(cuid, iuid));
328        contentSeq.add(contentItem(valueTypeOf(inst), refSOP(cuid, iuid)));
329        return true;
330    }

332    public void writeKOS() throws IOException {
333        DicomOutputStream dos = new DicomOutputStream(
334                new BufferedOutputStream(fname != null 
335                        ? new FileOutputStream(fname)
336                        : new FileOutputStream(FileDescriptor.out)),
337                nofmi ? UID.ImplicitVRLittleEndian
338                      : UID.ExplicitVRLittleEndian);
339        dos.setEncodingOptions(encOpts);
340        try {
341            dos.writeDataset(
342                    nofmi ? null : kos.createFileMetaInformation(tsuid),
343                    kos);
344        } finally {
345            dos.close();
346        }
347    }

349    private Sequence refSOPSeq(String studyIUID, String seriesIUID) {
350        Attributes refStudy = getOrAddItem(evidenceSeq, Tag.StudyInstanceUID, studyIUID);
351        Sequence refSeriesSeq = refStudy.ensureSequence(Tag.ReferencedSeriesSequence, 10);
352        Attributes refSeries = getOrAddItem(refSeriesSeq,Tag.SeriesInstanceUID, seriesIUID);
353        return refSeries.ensureSequence(Tag.ReferencedSOPSequence, 100);
354    }
355
356    private Attributes getOrAddItem(Sequence seq, int tag, String value) {
357        for (Attributes item : seq)
358            if (value.equals(item.getString(tag)))
359                return item;
360        
361        Attributes item = new Attributes(2);
362        item.setString(tag, VR.UI, value);
363        seq.add(item);
364        return item;
365    }

367    private String valueTypeOf(Attributes inst) {
368        return inst.contains(Tag.PhotometricInterpretation) ? "IMAGE"
369                      : inst.contains(Tag.WaveformSequence) ? "WAVEFORM"
370                                                            : "COMPOSITE";
371    }

373    private Attributes refSOP(String cuid, String iuid) {
374        Attributes item = new Attributes(2);
375        item.setString(Tag.ReferencedSOPClassUID, VR.UI, cuid);
376        item.setString(Tag.ReferencedSOPInstanceUID, VR.UI, iuid);
377        return item;
378    }

380    private Attributes createKOS(Attributes inst) {
381        Attributes attrs = new Attributes(inst, PATIENT_AND_STUDY_ATTRS);
382        attrs.setString(Tag.SOPClassUID, VR.UI, UID.KeyObjectSelectionDocumentStorage);
383        attrs.setString(Tag.SOPInstanceUID, VR.UI, UIDUtils.createUID());
384        attrs.setDate(Tag.ContentDateAndTime, new Date());
385        attrs.setString(Tag.Modality, VR.CS, "KO");
386        attrs.setNull(Tag.ReferencedPerformedProcedureStepSequence, VR.SQ);
387        attrs.setString(Tag.SeriesInstanceUID, VR.UI, UIDUtils.createUID());
388        attrs.setString(Tag.SeriesNumber, VR.IS, seriesNumber);
389        attrs.setString(Tag.InstanceNumber, VR.IS, instanceNumber);
390        attrs.setString(Tag.ValueType, VR.CS, "CONTAINER");
391        attrs.setString(Tag.ContinuityOfContent, VR.CS, "SEPARATE");
392        attrs.newSequence(Tag.ConceptNameCodeSequence, 1).add(documentTitle);
393        evidenceSeq = attrs.newSequence(Tag.CurrentRequestedProcedureEvidenceSequence, 1);
394        attrs.newSequence(Tag.ContentTemplateSequence, 1).add(templateIdentifier());
395        contentSeq = attrs.newSequence(Tag.ContentSequence, 1);
396        if (documentTitleModifier != null)
397            contentSeq.add(documentTitleModifier());
398        if (keyObjectDescription != null)
399            contentSeq.add(keyObjectDescription());
400        return attrs;
401    }

403    private Attributes templateIdentifier() {
404        Attributes attrs = new Attributes(2);
405        attrs.setString(Tag.MappingResource, VR.CS, "DCMR");
406        attrs.setString(Tag.TemplateIdentifier, VR.CS, "2010");
407        return attrs ;
408    }

410    private Attributes documentTitleModifier() {
411        Attributes item = new Attributes(4);
412        item.setString(Tag.RelationshipType, VR.CS, "HAS CONCEPT MOD");
413        item.setString(Tag.ValueType, VR.CS, "CODE");
414        item.newSequence(Tag.ConceptNameCodeSequence, 1).add(toCodeItem("DCM-113011"));
415        item.newSequence(Tag.ConceptCodeSequence, 1).add(documentTitleModifier);
416        return item;
417    }

419    private Attributes keyObjectDescription() {
420        Attributes item = new Attributes(4);
421        item.setString(Tag.RelationshipType, VR.CS, "CONTAINS");
422        item.setString(Tag.ValueType, VR.CS, "TEXT");
423        item.newSequence(Tag.ConceptNameCodeSequence, 1).add(toCodeItem("DCM-113012"));
424        item.setString(Tag.TextValue, VR.UT, keyObjectDescription);
425        return item;
426    }

428    private Attributes contentItem(String valueType, Attributes refSOP) {
429        Attributes item = new Attributes(3);
430        item.setString(Tag.RelationshipType, VR.CS, "CONTAINS");
431        item.setString(Tag.ValueType, VR.CS, valueType);
432        item.newSequence(Tag.ReferencedSOPSequence, 1).add(refSOP);
433        return item;
434    }
