340    public static CommandLine parseComandLine(String[] args, Options opts, 
341            ResourceBundle rb2, Class<?> clazz) throws ParseException {
342        CommandLineParser parser = new PosixParser();
343        CommandLine cl = parser.parse(opts, args);
344        if (cl.hasOption("h")) {
345            HelpFormatter formatter = new HelpFormatter();
346            formatter.printHelp(
347                    rb2.getString("usage"),
348                    rb2.getString("description"), opts,
349                    rb2.getString("example"));
350            System.exit(0);
351        }
352        if (cl.hasOption("V")) {
353            Package p = clazz.getPackage();
354            String s = p.getName();
355            System.out.println(s.substring(s.lastIndexOf('.')+1) + ": " +
356                   p.getImplementationVersion());
357            System.exit(0);
358        }
359        return cl;
360    }

362    public static void configureConnect(Connection conn,
363            AAssociateRQ rq, CommandLine cl) throws ParseException {
364        if (!cl.hasOption("c"))
365            throw new MissingOptionException(
366                    rb.getString("missing-connect-opt"));
367        String aeAtHostPort = cl.getOptionValue("c");
368        String[] aeHostPort = split(aeAtHostPort , '@', 0);
369        if (aeHostPort[1] == null)
370            throw new ParseException(rb.getString("invalid-connect-opt"));
371        
372        String[] hostPort = split(aeHostPort[1], ':', 0);
373        if (hostPort[1] == null)
374            throw new ParseException(rb.getString("invalid-connect-opt"));
375
376        rq.setCalledAET(aeHostPort[0]);
377        conn.setHostname(hostPort[0]);
378        conn.setPort(Integer.parseInt(hostPort[1]));
379
380        conn.setHttpProxy(cl.getOptionValue("proxy"));
381
382        if (cl.hasOption("user"))
383            rq.setUserIdentityRQ(cl.hasOption("user-pass")
384                    ? new UserIdentityRQ(cl.getOptionValue("user"),
385                            cl.getOptionValue("user-pass").toCharArray())
386                    : new UserIdentityRQ(cl.getOptionValue("user"),
387                            cl.hasOption("user-rsp")));
388    }

390    public static void configureBind(Connection conn,
391            ApplicationEntity ae, CommandLine cl) throws ParseException {
392        if (cl.hasOption("b")) {
393            String aeAtHostPort = cl.getOptionValue("b");
394            String[] aeAtHostAndPort = split(aeAtHostPort, ':', 0);
395            String[] aeHost = split(aeAtHostAndPort[0], '@', 0);
396            ae.setAETitle(aeHost[0]);

398                conn.setHostname(aeHost[1]);
399            if (aeAtHostAndPort[1] != null)
400                conn.setPort(Integer.parseInt(aeAtHostAndPort[1]));
401        }

419    private static String[] split(String s, char delim, int defPos) {
420        String[] s2 = new String[2];
421        int pos = s.indexOf(delim);
422        if (pos != -1) {
423            s2[0] = s.substring(0, pos);
424            s2[1] = s.substring(pos + 1);
425        } else {
426            s2[defPos] = s;
427        }
428        return s2;
429    }

440        String optVal = cl.getOptionValue(opt);

446                : Integer.parseInt(optVal);

449    public static void configure(Connection conn, CommandLine cl)
450            throws ParseException, IOException {
451        if (cl.hasOption("max-pdulen-rcv"))
452            conn.setReceivePDULength(Integer.parseInt(
453                    cl.getOptionValue("max-pdulen-rcv")));
454        if (cl.hasOption("max-pdulen-snd"))
455            conn.setSendPDULength(Integer.parseInt(
456                    cl.getOptionValue("max-pdulen-snd")));
457        if(cl.hasOption("not-async")) {
458            conn.setMaxOpsInvoked(1);
459            conn.setMaxOpsPerformed(1);
460        } else {
461            int maxOpsInvoked = 0;
462            if (cl.hasOption("max-ops-invoked"))
463                maxOpsInvoked = Integer.parseInt(
464                        cl.getOptionValue("max-ops-invoked"));
465            conn.setMaxOpsInvoked(maxOpsInvoked);
466            int maxOpsPerformed = 0;
467            if (cl.hasOption("max-ops-performed"))
468                maxOpsPerformed = Integer.parseInt(
469                        cl.getOptionValue("max-ops-performed"));
470            conn.setMaxOpsPerformed(maxOpsPerformed);
471        }
472        conn.setPackPDV(!cl.hasOption("not-pack-pdv"));
473        if (cl.hasOption("connect-timeout"))
474            conn.setConnectTimeout(
475                    Integer.parseInt(cl.getOptionValue("connect-timeout")));
476        if (cl.hasOption("request-timeout"))
477            conn.setRequestTimeout(
478                    Integer.parseInt(cl.getOptionValue("request-timeout")));
479        if (cl.hasOption("accept-timeout"))
480            conn.setAcceptTimeout(
481                    Integer.parseInt(cl.getOptionValue("accept-timeout")));
482        if (cl.hasOption("release-timeout"))
483            conn.setReleaseTimeout(
484                    Integer.parseInt(cl.getOptionValue("release-timeout")));
485        if (cl.hasOption("response-timeout"))
486            conn.setResponseTimeout(
487                    Integer.parseInt(cl.getOptionValue("response-timeout")));
488        if (cl.hasOption("retrieve-timeout"))
489            conn.setRetrieveTimeout(
490                    Integer.parseInt(cl.getOptionValue("retrieve-timeout")));
491        if (cl.hasOption("idle-timeout"))
492            conn.setIdleTimeout(
493                    Integer.parseInt(cl.getOptionValue("idle-timeout")));
494        if (cl.hasOption("soclose-delay"))
495            conn.setSocketCloseDelay(
496                    Integer.parseInt(cl.getOptionValue("soclose-delay")));
497        if (cl.hasOption("sosnd-buffer"))
498            conn.setSendBufferSize(
499                    Integer.parseInt(cl.getOptionValue("sosnd-buffer")));
500        if (cl.hasOption("sorcv-buffer"))
501            conn.setReceiveBufferSize(
502                    Integer.parseInt(cl.getOptionValue("sorcv-buffer")));
503        conn.setTcpNoDelay(!cl.hasOption("tcp-delay"));
504        configureTLS(conn, cl);
505    }

507    private static void configureTLS(Connection conn, CommandLine cl)
508            throws ParseException, IOException {
509        if (cl.hasOption("tls"))
510            conn.setTlsCipherSuites(
511                    "SSL_RSA_WITH_NULL_SHA",
512                    "TLS_RSA_WITH_AES_128_CBC_SHA",
513                    "SSL_RSA_WITH_3DES_EDE_CBC_SHA");
514        else if (cl.hasOption("tls-null"))
515            conn.setTlsCipherSuites("SSL_RSA_WITH_NULL_SHA");
516        else if (cl.hasOption("tls-3des"))
517            conn.setTlsCipherSuites("SSL_RSA_WITH_3DES_EDE_CBC_SHA");
518        else if (cl.hasOption("tls-aes"))
519            conn.setTlsCipherSuites(
520                    "TLS_RSA_WITH_AES_128_CBC_SHA",
521                    "SSL_RSA_WITH_3DES_EDE_CBC_SHA");
522        else if (cl.hasOption("tls-cipher"))
523            conn.setTlsCipherSuites(cl.getOptionValues("tls-cipher"));
524        else
525            return;
526
527        if (cl.hasOption("tls1"))
528            conn.setTlsProtocols("TLSv1");
529        else if (cl.hasOption("ssl3"))
530            conn.setTlsProtocols("SSLv3");
531        else if (cl.hasOption("ssl2Hello"))
532            conn.setTlsProtocols("SSLv2Hello", "SSLv3", "TLSv1");
533        else if (cl.hasOption("tls-protocol"))
534            conn.setTlsProtocols(cl.getOptionValues("tls-protocol"));
535
536        conn.setTlsNeedClientAuth(!cl.hasOption("tls-noauth"));
537
538        String keyStoreURL = cl.getOptionValue("key-store", "resource:key.jks");
539        String keyStoreType =  cl.getOptionValue("key-store-type", "JKS");
540        String keyStorePass = cl.getOptionValue("key-store-pass", "secret");
541        String keyPass = cl.getOptionValue("key-pass", keyStorePass);
542        String trustStoreURL = cl.getOptionValue("trust-store", "resource:cacerts.jks");
543        String trustStoreType =  cl.getOptionValue("trust-store-type", "JKS");
544        String trustStorePass = cl.getOptionValue("trust-store-pass", "secret");
545
546        Device device = conn.getDevice();
547        try {
548            device.setKeyManager(SSLManagerFactory.createKeyManager(
549                    keyStoreType, keyStoreURL, keyStorePass, keyPass));
550            device.setTrustManager(SSLManagerFactory.createTrustManager(
551                    trustStoreType, trustStoreURL, trustStorePass));
552        } catch (GeneralSecurityException e) {
553            throw new IOException(e);
554        }
555    }

557    public static Properties loadProperties(String url, Properties p)
558            throws IOException {
559        if (p == null)
560            p = new Properties();
561        InputStream in = StreamUtils.openFileOrURL(url);
562        try {
563            p.load(in);
564        } finally {
565            SafeClose.close(in);
566        }
567        return p;
568    }

616    public static int toTag(String tagOrKeyword) {
617        try {
618            return Integer.parseInt(tagOrKeyword, 16);
619        } catch (IllegalArgumentException e) {
620            int tag = ElementDictionary.tagForKeyword(tagOrKeyword, null);
621            if (tag == -1)
622                throw new IllegalArgumentException(tagOrKeyword);
623            return tag;
624        }

713    public static void addAttributes(Attributes attrs, int[] tags, String... ss) {
714        Attributes item = attrs;
715        for (int i = 0; i < tags.length-1; i++) {
716            int tag = tags[i];
717            Sequence sq = item.getSequence(tag);
718            if (sq == null)
719                sq = item.newSequence(tag, 1);
720            if (sq.isEmpty())
721                sq.add(new Attributes());
722            item = sq.get(0);
723        }
724        int tag = tags[tags.length-1];
725        VR vr = ElementDictionary.vrOf(tag,
726                item.getPrivateCreator(tag));
727        if (ss.length == 0)
728            if (vr == VR.SQ)
729                item.newSequence(tag, 1).add(new Attributes(0));
730            else
731                item.setNull(tag, vr);
732        else
733            item.setString(tag, vr, ss);
734    }

752    public static boolean updateAttributes(Attributes data, Attributes attrs,
753            String uidSuffix) {
754        if (attrs.isEmpty() && uidSuffix == null)
755            return false;
756        if (uidSuffix != null ) {
757            data.setString(Tag.StudyInstanceUID, VR.UI,
758                    data.getString(Tag.StudyInstanceUID) + uidSuffix);
759            data.setString(Tag.SeriesInstanceUID, VR.UI,
760                    data.getString(Tag.SeriesInstanceUID) + uidSuffix);
761            data.setString(Tag.SOPInstanceUID, VR.UI, 
762                    data.getString(Tag.SOPInstanceUID) + uidSuffix);
763        }
764        data.update(attrs, null);
765        return true;
766    }
