064public class LdapAuditLoggerConfiguration extends LdapDicomConfigurationExtension {
065
066    private static final String CN_AUDIT_LOGGER = "cn=Audit Logger,";
067
068    private static final Logger LOG = LoggerFactory.getLogger(
069            LdapAuditLoggerConfiguration.class);
070
071    @Override
072    protected void storeChilds(String deviceDN, Device device) throws NamingException {
073        AuditLogger logger = device.getDeviceExtension(AuditLogger.class);
074        if (logger != null)
075            store(deviceDN, logger);
076    }
077
078    private void store(String deviceDN, AuditLogger logger)
079            throws NamingException {
080        config.createSubcontext(CN_AUDIT_LOGGER + deviceDN,
081                storeTo(logger, deviceDN, new BasicAttributes(true)));
082    }
083
084    private Attributes storeTo(AuditLogger logger, String deviceDN, Attributes attrs) {
085        attrs.put(new BasicAttribute("objectclass", "dcmAuditLogger"));
086        LdapUtils.storeNotDef(attrs, "dcmAuditFacility",
087                logger.getFacility().ordinal(), 10);
088        LdapUtils.storeNotDef(attrs, "dcmAuditSuccessSeverity",
089                logger.getSuccessSeverity().ordinal(), 5);
090        LdapUtils.storeNotDef(attrs, "dcmAuditMinorFailureSeverity",
091                logger.getMinorFailureSeverity().ordinal(), 4);
092        LdapUtils.storeNotDef(attrs, "dcmAuditSeriousFailureSeverity",
093                logger.getSeriousFailureSeverity().ordinal(), 3);
094        LdapUtils.storeNotDef(attrs, "dcmAuditMajorFailureSeverity",
095                logger.getMajorFailureSeverity().ordinal(), 2);
096        LdapUtils.storeNotNull(attrs, "dcmAuditSourceID",
097                logger.getAuditSourceID());
098        LdapUtils.storeNotNull(attrs, "dcmAuditEnterpriseSiteID",
099                logger.getAuditEnterpriseSiteID());
100        LdapUtils.storeNotEmpty(attrs, "dcmAuditSourceTypeCode",
101                logger.getAuditSourceTypeCodes());
102        LdapUtils.storeNotNull(attrs, "dcmAuditApplicationName",
103                logger.getApplicationName());
104        LdapUtils.storeNotNull(attrs, "dcmAuditMessageID",
105                StringUtils.nullify(logger.getMessageID(), AuditLogger.MESSAGE_ID));
106        LdapUtils.storeNotNull(attrs, "dcmAuditMessageEncoding",
107                StringUtils.nullify(logger.getEncoding(), "UTF-8"));
108        LdapUtils.storeNotNull(attrs, "dcmAuditMessageSchemaURI",
109                logger.getSchemaURI());
110        LdapUtils.storeNotDef(attrs, "dcmAuditMessageBOM",
111                logger.isIncludeBOM(), true);
112        LdapUtils.storeNotDef(attrs, "dcmAuditMessageFormatXML",
113                logger.isFormatXML(), false);
114        LdapUtils.storeNotDef(attrs, "dcmAuditTimestampInUTC",
115                logger.isTimestampInUTC(), false);
116        LdapUtils.storeConnRefs(attrs, logger.getConnections(), deviceDN);
117        LdapUtils.storeNotNull(attrs, "dcmAuditRecordRepositoryDeviceReference",
118                config.deviceRef(logger.getAuditRecordRepositoryDeviceName()));
119        LdapUtils.storeNotDef(attrs, "dcmAuditIncludeInstanceUID", 
120                logger.isIncludeInstanceUID(), false);
121        LdapUtils.storeNotNull(attrs, "dicomInstalled",
122                logger.getInstalled());
123        return attrs;
124    }
125
126    @Override
127    protected void loadChilds(Device device, String deviceDN)
128            throws NamingException, ConfigurationException {
129        Attributes attrs;
130        try {
131            attrs = config.getAttributes(CN_AUDIT_LOGGER + deviceDN);
132        } catch (NameNotFoundException e) {
133            return;
134        }
135        AuditLogger logger = new AuditLogger();
136        loadFrom(logger, attrs);
137        for (String connDN : LdapUtils.stringArray(
138                attrs.get("dicomNetworkConnectionReference")))
139            logger.addConnection(
140                    LdapUtils.findConnection(connDN, deviceDN, device));
141        String arrDeviceDN = LdapUtils.stringValue(
142                attrs.get("dcmAuditRecordRepositoryDeviceReference"), null);
143        logger.setAuditRecordRepositoryDevice(deviceDN.equals(arrDeviceDN)
144                ? device
145                : loadAuditRecordRepository(arrDeviceDN));
146        device.addDeviceExtension(logger);
147    }
148
149    private Device loadAuditRecordRepository(String arrDeviceRef) {
150        try {
151            return config.loadDevice(arrDeviceRef);
152        } catch (ConfigurationException e) {
153            LOG.info("Failed to load Audit Record Repository "
154                    + arrDeviceRef + " referenced by Audit Logger", e);
155            return null;
156        }
157    }
158
159    private void loadFrom(AuditLogger logger, Attributes attrs) throws NamingException {
160        logger.setFacility(AuditLogger.Facility.values()
161                [LdapUtils.intValue(attrs.get("dcmAuditFacility"), 10)]);
162        logger.setSuccessSeverity(AuditLogger.Severity.values()
163                [LdapUtils.intValue(attrs.get("dcmAuditSuccessSeverity"), 5)]);
164        logger.setMinorFailureSeverity(AuditLogger.Severity.values()
165                [LdapUtils.intValue(attrs.get("dcmAuditMinorFailureSeverity"), 4)]);
166        logger.setSeriousFailureSeverity(AuditLogger.Severity.values()
167                [LdapUtils.intValue(attrs.get("dcmAuditSeriousFailureSeverity"), 3)]);
168        logger.setMajorFailureSeverity(AuditLogger.Severity.values()
169                [LdapUtils.intValue(attrs.get("dcmAuditMajorFailureSeverity"), 2)]);
170        logger.setAuditSourceID(LdapUtils.stringValue(attrs.get("dcmAuditSourceID"), null));
171        logger.setAuditEnterpriseSiteID(
172                LdapUtils.stringValue(attrs.get("dcmAuditEnterpriseSiteID"), null));
173        logger.setAuditSourceTypeCodes(
174                LdapUtils.stringArray(attrs.get("dcmAuditSourceTypeCode")));
175        logger.setApplicationName(
176                LdapUtils.stringValue(attrs.get("dcmAuditApplicationName"), null));
177        logger.setMessageID(
178                LdapUtils.stringValue(attrs.get("dcmAuditMessageID"), AuditLogger.MESSAGE_ID));
179        logger.setEncoding(
180                LdapUtils.stringValue(attrs.get("dcmAuditMessageEncoding"), "UTF-8"));
181        logger.setSchemaURI(
182                LdapUtils.stringValue(attrs.get("dcmAuditMessageSchemaURI"), null));
183        logger.setIncludeBOM(
184                LdapUtils.booleanValue(attrs.get("dcmAuditMessageBOM"), true));
185        logger.setFormatXML(
186                LdapUtils.booleanValue(attrs.get("dcmAuditMessageFormatXML"), false));
187        logger.setTimestampInUTC(
188                LdapUtils.booleanValue(attrs.get("dcmAuditTimestampInUTC"), false));
189        logger.setIncludeInstanceUID(
190                LdapUtils.booleanValue(attrs.get("dcmAuditIncludeInstanceUID"), false));
191        logger.setInstalled(
192                LdapUtils.booleanValue(attrs.get("dicomInstalled"), null));
193    }
194
195    @Override
196    protected void mergeChilds(Device prev, Device device, String deviceDN)
197            throws NamingException {
198        AuditLogger prevLogger = prev.getDeviceExtension(AuditLogger.class);
199        AuditLogger logger = device.getDeviceExtension(AuditLogger.class);
200        if (logger == null) {
201            if (prevLogger != null)
202                config.destroySubcontextWithChilds(CN_AUDIT_LOGGER + deviceDN);
203            return;
204        }
205        if (prevLogger == null) {
206            store(deviceDN, logger);
207            return;
208        }
209        config.modifyAttributes(CN_AUDIT_LOGGER + deviceDN,
210                storeDiffs(prevLogger, logger, deviceDN,
211                        new ArrayList<ModificationItem>()));
212    }
213
214    private List<ModificationItem> storeDiffs(AuditLogger a, AuditLogger b,
215            String deviceDN, ArrayList<ModificationItem> mods) {
216        LdapUtils.storeDiff(mods, "dcmAuditFacility",
217                a.getFacility().ordinal(),
218                b.getFacility().ordinal(),
219                10);
220        LdapUtils.storeDiff(mods, "dcmAuditSuccessSeverity",
221                a.getSuccessSeverity().ordinal(),
222                b.getSuccessSeverity().ordinal(),
223                5);
224        LdapUtils.storeDiff(mods, "dcmAuditMinorFailureSeverity",
225                a.getMinorFailureSeverity().ordinal(),
226                b.getMinorFailureSeverity().ordinal(),
227                4);
228        LdapUtils.storeDiff(mods, "dcmAuditSeriousFailureSeverity",
229                a.getSeriousFailureSeverity().ordinal(),
230                b.getSeriousFailureSeverity().ordinal(),
231                3);
232        LdapUtils.storeDiff(mods, "dcmAuditMajorFailureSeverity",
233                a.getMajorFailureSeverity().ordinal(),
234                b.getMajorFailureSeverity().ordinal(),
235                2);
236        LdapUtils.storeDiff(mods, "dcmAuditSourceID",
237                a.getAuditSourceID(),
238                b.getAuditSourceID());
239        LdapUtils.storeDiff(mods, "dcmAuditEnterpriseSiteID",
240                a.getAuditEnterpriseSiteID(),
241                b.getAuditEnterpriseSiteID());
242        LdapUtils.storeDiff(mods, "dcmAuditSourceTypeCode",
243                a.getAuditSourceTypeCodes(),
244                b.getAuditSourceTypeCodes());
245        LdapUtils.storeDiff(mods, "dcmAuditApplicationName",
246                a.getApplicationName(),
247                b.getApplicationName());
248        LdapUtils.storeDiff(mods, "dcmAuditMessageID",
249                a.getMessageID(),
250                b.getMessageID());
251        LdapUtils.storeDiff(mods, "dcmAuditMessageEncoding",
252                StringUtils.nullify(a.getEncoding(), "UTF-8"),
253                StringUtils.nullify(b.getEncoding(), "UTF-8"));
254        LdapUtils.storeDiff(mods, "dcmAuditMessageSchemaURI",
255                a.getSchemaURI(),
256                b.getSchemaURI());
257        LdapUtils.storeDiff(mods, "dcmAuditMessageBOM",
258                a.isIncludeBOM(),
259                b.isIncludeBOM(),
260                true);
261        LdapUtils.storeDiff(mods, "dcmAuditMessageFormatXML",
262                a.isFormatXML(),
263                b.isFormatXML(),
264                false);
265        LdapUtils.storeDiff(mods, "dcmAuditTimestampInUTC",
266                a.isTimestampInUTC(),
267                b.isTimestampInUTC(),
268                false);
269        LdapUtils.storeDiff(mods, "dicomNetworkConnectionReference",
270                a.getConnections(),
271                b.getConnections(),
272                deviceDN);
273        LdapUtils.storeDiff(mods, "dcmAuditRecordRepositoryDeviceReference",
274                arrDeviceRef(a),
275                arrDeviceRef(b));
276        LdapUtils.storeDiff(mods, "dcmAuditIncludeInstanceUID", 
277                a.isIncludeInstanceUID(), 
278                b.isIncludeInstanceUID(), 
279                false);
280        LdapUtils.storeDiff(mods, "dicomInstalled",
281                a.getInstalled(),
282                b.getInstalled());
283        return mods;
284    }
285
286    private String arrDeviceRef(AuditLogger a) {
287        Device arrDevice = a.getAuditRecordRepositoryDevice();
288        return arrDevice != null
289                ? config.deviceRef(arrDevice.getDeviceName())
290                : null;
291    }
