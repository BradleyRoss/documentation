038package org.dcm4che.conf.ldap.imageio;
039
040import java.util.ArrayList;
041import java.util.List;
042import java.util.Map.Entry;
043
044import javax.naming.NameNotFoundException;
045import javax.naming.NamingEnumeration;
046import javax.naming.NamingException;
047import javax.naming.directory.Attributes;
048import javax.naming.directory.BasicAttributes;
049import javax.naming.directory.ModificationItem;
050import javax.naming.directory.SearchResult;
051
052import org.dcm4che.conf.api.ConfigurationException;
053import org.dcm4che.conf.ldap.LdapDicomConfigurationExtension;
054import org.dcm4che.conf.ldap.LdapUtils;
055import org.dcm4che.imageio.codec.ImageReaderFactory;
056import org.dcm4che.imageio.codec.ImageReaderFactory.ImageReaderParam;
057import org.dcm4che.net.Device;
058import org.dcm4che.net.imageio.ImageReaderExtension;
059
060/**
061 * @author Gunter Zeilinger <gunterze@gmail.com>
062 *
063 */
064public class LdapImageReaderConfiguration extends LdapDicomConfigurationExtension {
065
066    private static final String CN_IMAGE_READER_FACTORY = "cn=Image Reader Factory,";
067
068    @Override
069    protected void storeChilds(String deviceDN, Device device) throws NamingException {
070        ImageReaderExtension ext =
071                device.getDeviceExtension(ImageReaderExtension.class);
072        if (ext != null)
073            store(deviceDN, ext.getImageReaderFactory());
074    }
075
076    private String dnOf(String tsuid, String imageReadersDN) {
077        return LdapUtils.dnOf("dicomTransferSyntax" ,tsuid, imageReadersDN);
078    }
079
080    private void store(String deviceDN, ImageReaderFactory factory) throws NamingException {
081        String imageReadersDN = CN_IMAGE_READER_FACTORY + deviceDN;
082        config.createSubcontext(imageReadersDN,
083                LdapUtils.attrs("dcmImageReaderFactory", "cn", "Image Reader Factory"));
084        for (Entry<String, ImageReaderParam> entry : factory.getEntries()) {
085            String tsuid = entry.getKey();
086            config.createSubcontext(dnOf(tsuid, imageReadersDN),
087                    storeTo(tsuid, entry.getValue(), new BasicAttributes(true)));
088        }
089    }
090
091    private Attributes storeTo(String tsuid, ImageReaderParam param, Attributes attrs) {
092        attrs.put("objectclass", "dcmImageReader");
093        attrs.put("dicomTransferSyntax", tsuid);
094        attrs.put("dcmIIOFormatName", param.formatName);
095        LdapUtils.storeNotNull(attrs, "dcmJavaClassName", param.className);
096        LdapUtils.storeNotNull(attrs, "dcmPatchJPEGLS", param.patchJPEGLS);
097        return attrs;
098    }
099
100    @Override
101    protected void loadChilds(Device device, String deviceDN)
102            throws NamingException, ConfigurationException {
103        String imageReadersDN = CN_IMAGE_READER_FACTORY + deviceDN;
104        try {
105            config.getAttributes(imageReadersDN);
106        } catch (NameNotFoundException e) {
107            return;
108        }
109
110        ImageReaderFactory factory = new ImageReaderFactory();
111        NamingEnumeration<SearchResult> ne =
112                config.search(imageReadersDN, "(objectclass=dcmImageReader)");
113        try {
114            while (ne.hasMore()) {
115                SearchResult sr = ne.next();
116                Attributes attrs = sr.getAttributes();
117                factory.put(
118                        LdapUtils.stringValue(attrs.get("dicomTransferSyntax"), null),
119                        new ImageReaderParam(
120                                LdapUtils.stringValue(
121                                        attrs.get("dcmIIOFormatName"), null),
122                                LdapUtils.stringValue(
123                                        attrs.get("dcmJavaClassName"), null),
124                                LdapUtils.stringValue(
125                                        attrs.get("dcmPatchJPEGLS"), null)));
126            }
127        } finally {
128           LdapUtils.safeClose(ne);
129        }
130        device.addDeviceExtension(new ImageReaderExtension(factory));
131    }
132
133    @Override
134    protected void mergeChilds(Device prev, Device device, String deviceDN)
135            throws NamingException {
136        ImageReaderExtension prevExt =
137                prev.getDeviceExtension(ImageReaderExtension.class);
138        ImageReaderExtension ext =
139                device.getDeviceExtension(ImageReaderExtension.class);
140        if (ext == null) {
141            if (prevExt != null)
142                config.destroySubcontextWithChilds(CN_IMAGE_READER_FACTORY + deviceDN);
143            return;
144        }
145        if (prevExt == null) {
146            store(deviceDN, ext.getImageReaderFactory());
147            return;
148        }
149        String imageReadersDN = CN_IMAGE_READER_FACTORY + deviceDN;
150        ImageReaderFactory factory = ext.getImageReaderFactory();
151        ImageReaderFactory prevFactory = prevExt.getImageReaderFactory();
152        for (Entry<String, ImageReaderParam> entry : prevFactory.getEntries()) {
153            String tsuid = entry.getKey();
154            if (factory.get(tsuid) == null)
155                config.destroySubcontext(dnOf(tsuid, imageReadersDN));
156        }
157        for (Entry<String, ImageReaderParam> entry : factory.getEntries()) {
158            String tsuid = entry.getKey();
159            String dn = dnOf(tsuid, imageReadersDN);
160            ImageReaderParam prevParam = prevFactory.get(tsuid);
161            if (prevParam == null)
162                config.createSubcontext(dn,
163                        storeTo(tsuid, entry.getValue(), new BasicAttributes(true)));
164            else
165                config.modifyAttributes(dn,
166                        storeDiffs(prevParam, entry.getValue(), new ArrayList<ModificationItem>()));
167        }
168    }
169
170    private List<ModificationItem> storeDiffs(ImageReaderParam prevParam,
171            ImageReaderParam param, List<ModificationItem> mods) {
172        LdapUtils.storeDiff(mods, "dcmIIOFormatName",
173                prevParam.formatName, param.formatName);
174        LdapUtils.storeDiff(mods, "dcmJavaClassName",
175                prevParam.className, param.className);
176        LdapUtils.storeDiff(mods, "dcmPatchJPEGLS",
177                prevParam.patchJPEGLS, param.patchJPEGLS);
178       return mods;
179    }
