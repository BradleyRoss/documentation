125    @SuppressWarnings("unchecked")
126    public LdapDicomConfiguration(Hashtable<?,?> env)
127            throws ConfigurationException {
128        try {
129            // split baseDN from LDAP URL
130            env = (Hashtable<?,?>) env.clone();
131            String s = (String) env.get(Context.PROVIDER_URL);
132            int end = s.lastIndexOf('/');
133            ((Hashtable<Object,Object>) env)
134                .put(Context.PROVIDER_URL, s.substring(0, end));
135            this.baseDN = s.substring(end+1);
136            this.ctx = new InitialDirContext(env);
137        } catch (Exception e) {
138            throw new ConfigurationException(e);
139        }
140    }

263    public synchronized Device findDevice(String filter, String childName)
264            throws ConfigurationException {
265        if (!configurationExists())
266            throw new ConfigurationNotFoundException();
267
268        SearchControls ctls = new SearchControls();
269        ctls.setSearchScope(SearchControls.SUBTREE_SCOPE);
270        ctls.setCountLimit(1);
271        ctls.setReturningAttributes(StringUtils.EMPTY_STRING);
272        ctls.setReturningObjFlag(false);
273        NamingEnumeration<SearchResult> ne = null;
274        String childDN;
275        try {
276            ne = ctx.search(devicesDN, filter, ctls);
277            if (!ne.hasMore())
278                throw new ConfigurationNotFoundException(childName);
279
280            childDN = ne.next().getNameInNamespace();
281        } catch (NamingException e) {
282            throw new ConfigurationException(e);
283        } finally {
284           LdapUtils.safeClose(ne);
285        }
286        String deviceDN = childDN.substring(childDN.indexOf(',') + 1);
287        return loadDevice(deviceDN);
288    }

316                "dicomInstalled");
317            while (ne.hasMore()) {

322        } catch (NamingException e) {

324        } finally {
325           LdapUtils.safeClose(ne);
326        }
327        return results.toArray(new DeviceInfo[results.size()]);

330    private String toFilter(DeviceInfo keys) {
331        if (keys == null)
332            return "(objectclass=dicomDevice)";
333
334        StringBuilder sb = new StringBuilder();
335        sb.append("(&(objectclass=dicomDevice)");
336        appendFilter("dicomDeviceName", keys.getDeviceName(), sb);
337        appendFilter("dicomDescription", keys.getDescription(), sb);
338        appendFilter("dicomManufacturer", keys.getManufacturer(), sb);
339        appendFilter("dicomManufacturerModelName", keys.getManufacturerModelName(), sb);
340        appendFilter("dicomSoftwareVersion", keys.getSoftwareVersions(), sb);
341        appendFilter("dicomStationName", keys.getStationName(), sb);
342        appendFilter("dicomInstitutionName", keys.getInstitutionNames(), sb);
343        appendFilter("dicomInstitutionDepartmentName", keys.getInstitutionalDepartmentNames(), sb);
344        appendFilter("dicomPrimaryDeviceType", keys.getPrimaryDeviceTypes(), sb);
345        appendFilter("dicomInstalled", keys.getInstalled(), sb);
346        sb.append(")");
347        return sb.toString();
348    }

350    private void appendFilter(String attrid, Boolean value, StringBuilder sb) {
351        if (value != null)
352            appendFilter(attrid, LdapUtils.toString(value), sb);
353    }
354
355    private void appendFilter(String attrid, String value, StringBuilder sb) {
356        if (value == null)
357            return;
358
359        sb.append('(').append(attrid).append('=').append(value).append(')');
360    }
361
362    private void appendFilter(String attrid, String[] values, StringBuilder sb) {
363        if (values.length == 0)
364            return;
365
366        if (values.length == 1) {
367            appendFilter(attrid, values[0], sb);
368            return;
369        }
370
371        sb.append("(|");
372        for (String value : values)
373            appendFilter(attrid, value, sb);
374        sb.append(")");
375    }

436    @Override
437    public synchronized void persist(Device device) throws ConfigurationException {
438        ensureConfigurationExists();
439        String deviceName = device.getDeviceName();
440        String deviceDN = deviceRef(deviceName);
441        boolean rollback = false;
442        try {
443            createSubcontext(deviceDN, storeTo(device, new BasicAttributes(true)));
444            rollback = true;
445            storeChilds(deviceDN, device);
446            updateCertificates(device);
447            rollback = false;
448        } catch (NameAlreadyBoundException e) {
449            throw new ConfigurationAlreadyExistsException(deviceName);
450        } catch (NamingException e) {
451            throw new ConfigurationException(e);
452        } catch (CertificateException e) {
453            throw new ConfigurationException(e);
454        } finally {
455            if (rollback)
456                try {
457                    destroySubcontextWithChilds(deviceDN);
458                } catch (NamingException e) {
459                    LOG.warn("Rollback failed:", e);
460                }
461        }
462    }

464    private void updateCertificates(Device device)
465            throws CertificateException, NamingException {
466        for (String dn : device.getAuthorizedNodeCertificateRefs())
467            updateCertificates(dn, loadCertificates(dn),
468                    device.getAuthorizedNodeCertificates(dn));
469        for (String dn : device.getThisNodeCertificateRefs())
470            updateCertificates(dn, loadCertificates(dn),
471                    device.getThisNodeCertificates(dn));
472    }
473
474    private void updateCertificates(String dn,
475            X509Certificate[] prev, X509Certificate[] certs)
476            throws CertificateEncodingException, NamingException {
477        if (!LdapUtils.equals(prev, certs))
478            storeCertificates(dn, certs);
479    }

481    private void storeChilds(String deviceDN, Device device) throws NamingException {
482        for (Connection conn : device.listConnections())
483            createSubcontext(LdapUtils.dnOf(conn, deviceDN), storeTo(conn, new BasicAttributes(true)));
484        for (LdapDicomConfigurationExtension ext : extensions)
485            ext.storeChilds(deviceDN, device);
486        for (ApplicationEntity ae : device.getApplicationEntities()) {
487            String aeDN = aetDN(ae.getAETitle(), deviceDN);
488            createSubcontext(aeDN, storeTo(ae, deviceDN, new BasicAttributes(true)));
489            storeChilds(aeDN, ae);
490            for (LdapDicomConfigurationExtension ext : extensions) {
491                ext.storeChilds(aeDN, ae);
492            }
493        }
494    }
495
496    private void storeChilds(String aeDN, ApplicationEntity ae)
497            throws NamingException {
498        for (TransferCapability tc : ae.getTransferCapabilities())
499            createSubcontext(dnOf(tc, aeDN), storeTo(tc, new BasicAttributes(true)));
500    }

502    @Override
503    public synchronized void merge(Device device) throws ConfigurationException {
504        if (!configurationExists())
505            throw new ConfigurationNotFoundException();
506
507        String deviceDN = deviceRef(device.getDeviceName());
508        Device prev = loadDevice(deviceDN);
509        try {
510            modifyAttributes(deviceDN, storeDiffs(prev, device, new ArrayList<ModificationItem>()));
511            mergeChilds(prev, device, deviceDN);
512            updateCertificates(prev, device);
513        } catch (NameNotFoundException e) {
514            throw new ConfigurationNotFoundException(e);
515        } catch (NamingException e) {
516            throw new ConfigurationException(e);
517        } catch (CertificateException e) {
518            throw new ConfigurationException(e);
519        }
520    }
521
522    private void updateCertificates(Device prev, Device device)
523            throws CertificateException, NamingException {
524        for (String dn : device.getAuthorizedNodeCertificateRefs()) {
525            X509Certificate[] prevCerts = prev.getAuthorizedNodeCertificates(dn);
526            updateCertificates(dn,
527                    prevCerts != null ? prevCerts : loadCertificates(dn),
528                    device.getAuthorizedNodeCertificates(dn));
529        }
530        for (String dn : device.getThisNodeCertificateRefs()) {
531            X509Certificate[] prevCerts = prev.getThisNodeCertificates(dn);
532            updateCertificates(dn,
533                    prevCerts != null ? prevCerts : loadCertificates(dn),
534                    device.getThisNodeCertificates(dn));
535        }
536    }

538    private void mergeChilds(Device prev, Device device, String deviceDN)

610                    LdapUtils.attrs(configurationRoot, "cn", configurationCN));

620        }

623   private boolean findConfiguration() throws ConfigurationException {
624        NamingEnumeration<SearchResult> ne = null;
625        try {
626            SearchControls ctls = new SearchControls();
627            ctls.setSearchScope(SearchControls.SUBTREE_SCOPE);
628            ctls.setCountLimit(1);
629            ctls.setReturningAttributes(StringUtils.EMPTY_STRING);
630            ctls.setReturningObjFlag(false);
631            ne = ctx.search(
632                    baseDN,
633                    "(&(objectclass=" + configurationRoot
634                            + ")(cn=" + configurationCN + "))",
635                    ctls);
636            if (!ne.hasMore())
637                return false;
638    
639            setConfigurationDN(ne.next().getName() + "," + baseDN);
640            return true;
641        } catch (NamingException e) {
642            throw new ConfigurationException(e);
643        } finally {
644            LdapUtils.safeClose(ne);
645        }
646    }

890        Attribute attr = attrs.get(userCertificate);

892            return EMPTY_X509_CERTIFICATES;

894        X509Certificate[] certs = new X509Certificate[attr.size()];

897                    new ByteArrayInputStream((byte[]) attr.get(i)));
898
899        return certs;

1724            String parentDN) throws NamingException {
1725        for (AttributeCoercion prev : prevs) {
1726            String cn = prev.getCommonName();
1727            if (acs.findByCommonName(cn) == null)
1728                destroySubcontext(LdapUtils.dnOf("cn", cn, parentDN));
1729        }
1730        for (AttributeCoercion ac : acs) {
1731            String cn = ac.getCommonName();
1732            String dn = LdapUtils.dnOf("cn", cn, parentDN);
1733            AttributeCoercion prev = prevs.findByCommonName(cn);
1734            if (prev == null)
1735                createSubcontext(dn, storeTo(ac, new BasicAttributes(true)));
1736            else
1737                modifyAttributes(dn, storeDiffs(prev, ac, 
1738                        new ArrayList<ModificationItem>()));
1739        }
