197    public synchronized Device findDevice(String nodeName, String aet)
198            throws ConfigurationException {
199         if (!PreferencesUtils.nodeExists(rootPrefs, DICOM_DEVICES_ROOT))
200            throw new ConfigurationNotFoundException();
201        
202        try {
203            Preferences devicePrefs = rootPrefs.node(DICOM_DEVICES_ROOT);
204            for (String deviceName : devicePrefs.childrenNames()) {
205                Preferences deviceNode = devicePrefs.node(deviceName);
206                for (String aet2 : deviceNode.node(nodeName).childrenNames())
207                    if (aet.equals(aet2))
208                        return loadDevice(deviceNode);
209            }
210        } catch (BackingStoreException e) {
211            throw new ConfigurationException(e);
212        }
213        throw new ConfigurationNotFoundException(aet);
214    }

223    public DeviceInfo[] listDeviceInfos(DeviceInfo keys)
224            throws ConfigurationException {
225        if (!PreferencesUtils.nodeExists(rootPrefs, DICOM_DEVICES_ROOT))
226            return new DeviceInfo[0];
227
228        ArrayList<DeviceInfo> results = new ArrayList<DeviceInfo>();
229        try {
230            Preferences devicePrefs = rootPrefs.node(DICOM_DEVICES_ROOT);
231            for (String deviceName : devicePrefs.childrenNames()) {
232                Preferences deviceNode = devicePrefs.node(deviceName);
233                DeviceInfo deviceInfo = new DeviceInfo();
234                deviceInfo.setDeviceName(deviceName);
235                loadFrom(deviceInfo, deviceNode);
236                if (match(deviceInfo, keys))
237                    results.add(deviceInfo);
238            }
239        } catch (BackingStoreException e) {
240            throw new ConfigurationException(e);
241        }
242        return results.toArray(new DeviceInfo[results.size()]) ;
243    }

245    private boolean match(DeviceInfo deviceInfo, DeviceInfo keys) {
246        if (keys == null)
247            return true;
248
249        return StringUtils.matches(
250                deviceInfo.getDeviceName(),
251                keys.getDeviceName(),
252                true, true)
253            && StringUtils.matches(
254                deviceInfo.getDescription(),
255                keys.getDescription(),
256                true, true)
257            && StringUtils.matches(
258                deviceInfo.getManufacturer(),
259                keys.getManufacturer(),
260                true, true)
261            && StringUtils.matches(
262                deviceInfo.getManufacturerModelName(),
263                keys.getManufacturerModelName(),
264                true, true)
265            && matches(
266                deviceInfo.getSoftwareVersions(),
267                keys.getSoftwareVersions())
268            && StringUtils.matches(
269                deviceInfo.getStationName(),
270                keys.getStationName(),
271                true, true)
272            && matches(
273                deviceInfo.getInstitutionNames(),
274                keys.getInstitutionNames())
275            && matches(
276                deviceInfo.getInstitutionalDepartmentNames(),
277                keys.getInstitutionalDepartmentNames())
278            && matches(
279                deviceInfo.getPrimaryDeviceTypes(),
280                keys.getPrimaryDeviceTypes())
281            && (keys.getInstalled() == null
282             || keys.getInstalled().equals(deviceInfo.getInstalled()));
283    }

285    private boolean matches(String[] values, String[] keys) {
286        if (keys.length == 0)
287            return true;
288
289        for (String key : keys)
290            for (String value : values)
291                if (StringUtils.matches(value, key, true, true))
292                    return true;
293
294        return false;
295    }

346    @Override
347    public synchronized void persist(Device device) throws ConfigurationException {
348        String deviceName = device.getDeviceName();
349        String pathName = deviceRef(deviceName);
350        if (PreferencesUtils.nodeExists(rootPrefs, pathName))
351            throw new ConfigurationAlreadyExistsException(pathName);
352
353        Preferences deviceNode = rootPrefs.node(pathName);
354        storeTo(device, deviceNode);
355        storeChilds(device, deviceNode);
356        try {
357            updateCertificates(device);
358            deviceNode.flush();
359            deviceNode = null;
360        } catch (BackingStoreException e) {
361            throw new ConfigurationException(e);
362        } catch (CertificateException e) {
363            throw new ConfigurationException(e);
364        } finally {
365            if (deviceNode != null)
366                try {
367                    deviceNode.removeNode();
368                    deviceNode.flush();
369                } catch (BackingStoreException e) {
370                    LOG.warn("Rollback failed:", e);
371                }
372        }
373    }

375    private void updateCertificates(Device device)
376            throws CertificateException, BackingStoreException {
377        for (String dn : device.getAuthorizedNodeCertificateRefs())
378            updateCertificates(dn, loadCertificates(dn),
379                    device.getAuthorizedNodeCertificates(dn));
380        for (String dn : device.getThisNodeCertificateRefs())
381            updateCertificates(dn, loadCertificates(dn),
382                    device.getThisNodeCertificates(dn));
383    }

385    private void updateCertificates(String ref, 
386            X509Certificate[] prev, X509Certificate[] certs)
387            throws CertificateEncodingException, BackingStoreException {
388        if (!Arrays.equals(prev, certs))
389            storeCertificates(ref, certs);
390    }

392    private void storeChilds(Device device, Preferences deviceNode) {
393        Preferences connsNode = deviceNode.node("dcmNetworkConnection");
394        int connIndex = 1;
395        List<Connection> devConns = device.listConnections();
396        for (Connection conn : devConns)
397            storeTo(conn, connsNode.node("" + connIndex++));
398        Preferences aesNode = deviceNode.node("dcmNetworkAE");
399        for (ApplicationEntity ae : device.getApplicationEntities()) {
400            Preferences aeNode = aesNode.node(ae.getAETitle());
401            storeTo(ae, aeNode, devConns);
402            storeChilds(ae, aeNode);
403        }
404
405        for (PreferencesDicomConfigurationExtension ext : extensions)
406            ext.storeChilds(device, deviceNode);
407    }

446    @Override
447    public synchronized void merge(Device device) throws ConfigurationException {
448        String pathName = deviceRef(device.getDeviceName());
449        if (!PreferencesUtils.nodeExists(rootPrefs, pathName))
450            throw new ConfigurationNotFoundException();
451        
452        Preferences devicePrefs = rootPrefs.node(pathName);
453        Device prev = loadDevice(devicePrefs);
454        try {
455            storeDiffs(devicePrefs, prev, device);
456            mergeChilds(prev, device, devicePrefs);
457            updateCertificates(prev, device);
458            devicePrefs.flush();
459        } catch (BackingStoreException e) {
460            throw new ConfigurationException(e);
461        } catch (CertificateException e) {
462            throw new ConfigurationException(e);
463        }
464    }

466    private void updateCertificates(Device prev, Device device)
467            throws CertificateException, BackingStoreException {
468        for (String dn : device.getAuthorizedNodeCertificateRefs()) {
469            X509Certificate[] prevCerts = prev.getAuthorizedNodeCertificates(dn);
470            updateCertificates(dn,
471                    prevCerts != null ? prevCerts : loadCertificates(dn),
472                    device.getAuthorizedNodeCertificates(dn));
473        }
474        for (String dn : device.getThisNodeCertificateRefs()) {
475            X509Certificate[] prevCerts = prev.getThisNodeCertificates(dn);
476            updateCertificates(dn,
477                    prevCerts != null ? prevCerts : loadCertificates(dn),
478                    device.getThisNodeCertificates(dn));
479        }
480    }

965    private void mergeConnections(Device prevDev, Device device, Preferences deviceNode)
966            throws BackingStoreException {
967        Preferences connsNode = deviceNode.node("dcmNetworkConnection");
968        List<Connection> prevs = prevDev.listConnections();
969        List<Connection> conns = device.listConnections();
970        int prevsSize = prevs.size();
971        int connsSize = conns.size();
972        int i = 0;
973        for (int n = Math.min(prevsSize, connsSize); i < n; ++i)
974            storeDiffs(connsNode.node("" + (i+1)), prevs.get(i), conns.get(i));
975        for (; i < prevsSize; ++i)
976            connsNode.node("" + (i+1)).removeNode();
977        for (; i < connsSize; ++i)
978            storeTo(conns.get(i), connsNode.node("" + (i+1)));
979    }

1441    private X509Certificate[] loadCertificates(String certRef)
1442            throws CertificateException, BackingStoreException {
1443        if (!rootPrefs.nodeExists(certRef))
1444            return EMPTY_X509_CERTIFICATES;
1445
1446        Preferences prefs = rootPrefs.node(certRef);
1447        int n = prefs.getInt(USER_CERTIFICATE + ".#", 0);
1448        if (n == 0)
1449            return EMPTY_X509_CERTIFICATES;
1450        
1451        CertificateFactory cf = CertificateFactory.getInstance("X509");
1452        X509Certificate[] certs = new X509Certificate[n];
1453        for (int i = 0; i < n; i++)
1454            certs[i] = (X509Certificate) cf.generateCertificate(
1455                    new ByteArrayInputStream(
1456                            prefs.getByteArray(USER_CERTIFICATE + '.' + (i+1), null)));
1457        return certs;
1458    }
