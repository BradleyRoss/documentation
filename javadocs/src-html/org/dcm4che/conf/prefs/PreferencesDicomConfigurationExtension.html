039package org.dcm4che.conf.prefs;
040
041import java.security.cert.CertificateException;
042import java.util.prefs.BackingStoreException;
043import java.util.prefs.Preferences;
044
045import org.dcm4che.conf.api.ConfigurationException;
046import org.dcm4che.net.ApplicationEntity;
047import org.dcm4che.net.Device;
048
049/**
050 * @author Gunter Zeilinger <gunterze@gmail.com>
051 *
052 */
053public class PreferencesDicomConfigurationExtension {
054
055    protected PreferencesDicomConfiguration config;
056
057    public PreferencesDicomConfiguration getDicomConfiguration() {
058        return config;
059    }
060
061    public void setDicomConfiguration(PreferencesDicomConfiguration config) {
062        if (config != null && this.config != null)
063            throw new IllegalStateException("already owned by other Dicom Configuration");
064        this.config = config;
065    }
066
067    protected void storeTo(Device device, Preferences prefs) {}
068
069    protected void storeChilds(Device device, Preferences prefs) {}
070
071    protected void loadFrom(Device device, Preferences prefs)
072            throws CertificateException, BackingStoreException {}
073
074    protected void loadChilds(Device device, Preferences prefs)
075            throws BackingStoreException, ConfigurationException {}
076
077    protected void storeDiffs(Device a, Device b, Preferences prefs) {}
078
079    protected void mergeChilds(Device a, Device b, Preferences prefs)
080            throws BackingStoreException {}
081
082    protected void storeTo(ApplicationEntity ae, Preferences prefs) {}
083
084    protected void storeChilds(ApplicationEntity ae, Preferences aeNode) {}
085
086    protected void loadFrom(ApplicationEntity ae, Preferences prefs) {}
087
088    protected void loadChilds(ApplicationEntity ae, Preferences aeNode)
089            throws BackingStoreException {}
090
091    protected void storeDiffs(ApplicationEntity a, ApplicationEntity b,
092            Preferences prefs) {}
093
094    protected void mergeChilds(ApplicationEntity prev, ApplicationEntity ae,
095            Preferences aePrefs) throws BackingStoreException {}
096}
