107    private enum Codec {
108        ISO_646("US-ASCII", 0x2842, 0),
109        ISO_8859_1("ISO-8859-1", 0x2842, 0x2d41),
110        ISO_8859_2("ISO-8859-2", 0x2842, 0x2d42),
111        ISO_8859_3("ISO-8859-3", 0x2842, 0x2d43),

113        ISO_8859_5("ISO-8859-5", 0x2842, 0x2d4c),

136        public static Codec forCode(String code) {
137            if (code == null)
138                return ISO_646;
139
140            switch(last2digits(code)) {
141            case 0:
142                if (code.equals("ISO_IR 100") || code.equals("ISO 2022 IR 100"))
143                    return Codec.ISO_8859_1;
144                break;
145            case 1:
146                if (code.equals("ISO_IR 101") || code.equals("ISO 2022 IR 101"))
147                    return Codec.ISO_8859_2;
148                break;
149            case 6:
150                if (code.equals("ISO 2022 IR 6"))
151                    return Codec.ISO_646;
152                break;
153            case 9:
154                if (code.equals("ISO_IR 109") || code.equals("ISO 2022 IR 109"))
155                    return Codec.ISO_8859_3;
156                break;
157            case 10:
158                if (code.equals("ISO_IR 110") || code.equals("ISO 2022 IR 110"))
159                    return Codec.ISO_8859_4;
160                break;
161            case 13:
162                if (code.equals("ISO_IR 13") || code.equals("ISO 2022 IR 13"))
163                    return Codec.JIS_X_201;
164                break;
165            case 26:
166                if (code.equals("ISO_IR 126") || code.equals("ISO 2022 IR 126"))
167                    return Codec.ISO_8859_7;
168                break;
169            case 27:
170                if (code.equals("ISO_IR 127") || code.equals("ISO 2022 IR 127"))
171                    return Codec.ISO_8859_6;
172                break;
173            case 30:
174                if (code.equals("GB18030"))
175                    return Codec.GB18030;
176                break;
177            case 38:
178                if (code.equals("ISO_IR 138") || code.equals("ISO 2022 IR 138"))
179                    return Codec.ISO_8859_8;
180                break;
181            case 44:
182                if (code.equals("ISO_IR 144") || code.equals("ISO 2022 IR 144"))
183                    return Codec.ISO_8859_5;
184                break;
185            case 48:
186                if (code.equals("ISO_IR 148") || code.equals("ISO 2022 IR 148"))
187                    return Codec.ISO_8859_9;
188                break;
189            case 49:
190                if (code.equals("ISO 2022 IR 149"))
191                    return Codec.KS_X_1001;
192                break;
193            case 59:
194                if (code.equals("ISO 2022 IR 159"))
195                    return Codec.JIS_X_212;
196                break;
197            case 66:
198                if (code.equals("ISO_IR 166") || code.equals("ISO 2022 IR 166"))
199                    return Codec.TIS_620;
200                break;
201            case 87:
202                if (code.equals("ISO 2022 IR 87"))
203                    return Codec.JIS_X_208;
204                break;
205            case 92:
206                if (code.equals("ISO_IR 192"))
207                    return Codec.UTF_8;
208                break;
209            }
210            return ISO_646;
211        }

213        private static int last2digits(String code) {
214            int len = code.length();
215            if (len < 2)
216                return -1;
217            char ch1 = code.charAt(len-1);
218            char ch2 = code.charAt(len-2);
219            return (ch2 & 15) * 10 + (ch1 & 15);
220        }

252    private static final class Encoder {
253        final Codec codec;
254        final CharsetEncoder encoder;
255 
256        public Encoder(Codec codec) {
257            this.codec = codec;
258            this.encoder = Charset.forName(codec.charsetName).newEncoder();
259        }
260
261        public boolean encode(CharBuffer cb, ByteBuffer bb, boolean escSeq,
262                CodingErrorAction errorAction) {
263            encoder.onMalformedInput(errorAction)
264                    .onUnmappableCharacter(errorAction)
265                    .reset();
266            int cbmark = cb.position();
267            int bbmark = bb.position();
268            try {
269                if (escSeq)
270                    escSeq(bb, codec.getEscSeq1());
271                CoderResult cr = encoder.encode(cb, bb, true);
272                if (!cr.isUnderflow())
273                    cr.throwException();
274                cr = encoder.flush(bb);
275                if (!cr.isUnderflow())
276                    cr.throwException();
277            } catch (CharacterCodingException x) {
278                cb.position(cbmark);
279                bb.position(bbmark);
280                return false;
281            }
282            return true;
283        }
284
285        private static void escSeq(ByteBuffer bb, int seq) {
286            bb.put((byte) 0x1b);
287            int b1 = seq >> 16;
288            if (b1 != 0)
289                bb.put((byte) b1);
290            bb.put((byte) (seq >> 8));
291            bb.put((byte) seq);
292        }
293    }

295    private static final class ISO2022 extends SpecificCharacterSet {
296
297        private ISO2022(Codec... charsetInfos) {
298            super(charsetInfos);
299        }
300
301        @Override
302        public byte[] encode(String val, String delimiters) {
303            int strlen = val.length();
304            CharBuffer cb = CharBuffer.wrap(val.toCharArray());
305            Encoder enc1 = encoder(cachedEncoder1, codecs[0]);
306            byte[] buf = new byte[strlen];
307            ByteBuffer bb = ByteBuffer.wrap(buf);
308            // try to encode whole string value with character set specified
309            // by value1 of (0008,0005) Specific Character Set
310            if (!enc1.encode(cb, bb, false, CodingErrorAction.REPORT)) {
311                // split whole string value according VR specific delimiters
312                // and try to encode each component separately
313                Encoder[] encs = new Encoder[codecs.length];
314                encs[0] = enc1;
315                encs[1] = encoder(cachedEncoder2, codecs[1]);
316                StringTokenizer comps =
317                        new StringTokenizer(val, delimiters, true);
318                buf = new byte[2 * strlen + 4 * (comps.countTokens() + 1)];
319                bb = ByteBuffer.wrap(buf);
320                int cur = 0;
321                while (comps.hasMoreTokens()) {
322                    String comp = comps.nextToken();
323                    if (comp.length() == 1 // if delimiter
324                            && delimiters.indexOf(comp.charAt(0)) >= 0) {
325                        // switch to initial character set, if current active
326                        // character set does not contain ASCII
327                        if (!codecs[cur].containsASCII())
328                            Encoder.escSeq(bb, codecs[0].getEscSeq0());
329                        bb.put((byte) comp.charAt(0));
330                        cur = 0;
331                        continue;
332                    }
333                    cb = CharBuffer.wrap(comp.toCharArray());
334                    // try to encode component with current active character set
335                    if (encs[cur].encode(cb, bb, false,
336                            CodingErrorAction.REPORT))
337                        continue;
338                    int next = cur;
339                    // try to encode component with other character sets
340                    // specified by values of (0008,0005) Specific Character Set
341                    do {
342                        next = (next + 1) % encs.length;
343                        if (next == cur) {
344                            // component could not be encoded with any of the
345                            // specified character sets, encode it with the
346                            // initial character set, using the default
347                            // replacement of the character set decoder
348                            // for characters which cannot be encoded
349                            if (!codecs[cur].containsASCII())
350                                Encoder.escSeq(bb, codecs[0].getEscSeq0());
351                            encs[0].encode(cb, bb, false,
352                                    CodingErrorAction.REPLACE);
353                            next = 0;
354                            break;
355                        }
356                        if (encs[next] == null)
357                            encs[next] = new Encoder(codecs[next]);
358                    } while (!encs[next].encode(cb, bb, true, 
359                            CodingErrorAction.REPORT));
360                    cur = next;
361                }
362                if (!codecs[cur].containsASCII())
363                    Encoder.escSeq(bb, codecs[0].getEscSeq0());
364            }
365            return Arrays.copyOf(buf, bb.position());
366        }
367
368        @Override
369        public String decode(byte[] b) {
370            Codec codec = codecs[0];
371            int off = 0;
372            int cur = 0;
373            int step = 1;
374            StringBuffer sb = new StringBuffer(b.length);
375            while (cur < b.length) {
376                if (b[cur] == 0x1b) { // ESC
377                    if (off < cur) {
378                        sb.append(codec.decode(b, off, cur - off));
379                    }
380                    cur += 3;
381                    switch (((b[cur - 2] & 255) << 8) + (b[cur - 1] & 255)) {
382                    case 0x2428:
383                        if (b[cur++] == 0x44) {
384                            codec = Codec.JIS_X_212;
385                            step = 2;
386                        } else { // decode invalid ESC sequence as chars
387                            sb.append(codec.decode(b, cur - 4, 4));
388                        }
389                        break;
390                    case 0x2429:
391                        if (b[cur++] == 0x43) {
392                            codec = Codec.KS_X_1001;
393                            step = -1;
394                        } else { // decode invalid ESC sequence as chars
395                            sb.append(codec.decode(b, cur - 4, 4));
396                        }
397                        break;
398                    case 0x2442:
399                        codec = Codec.JIS_X_208;
400                        step = 2;
401                        break;
402                    case 0x2842:
403                        codec = Codec.ISO_646;
404                        step = 1;
405                        break;
406                    case 0x284a:
407                    case 0x2949:
408                        codec = Codec.JIS_X_201;
409                        step = 1;
410                        break;
411                    case 0x2d41:
412                        codec = Codec.ISO_8859_1;
413                        step = 1;
414                        break;
415                    case 0x2d42:
416                        codec = Codec.ISO_8859_2;
417                        step = 1;
418                        break;
419                    case 0x2d43:
420                        codec = Codec.ISO_8859_3;
421                        step = 1;
422                        break;
423                    case 0x2d44:
424                        codec = Codec.ISO_8859_4;
425                        step = 1;
426                        break;
427                    case 0x2d46:
428                        codec = Codec.ISO_8859_7;
429                        step = 1;
430                        break;
431                    case 0x2d47:
432                        codec = Codec.ISO_8859_6;
433                        step = 1;
434                        break;
435                    case 0x2d48:
436                        codec = Codec.ISO_8859_8;
437                        step = 1;
438                        break;
439                    case 0x2d4c:
440                        codec = Codec.ISO_8859_5;
441                        step = 1;
442                        break;
443                    case 0x2d4d:
444                        codec = Codec.ISO_8859_9;
445                        step = 1;
446                        break;
447                    case 0x2d54:
448                        codec = Codec.TIS_620;
449                        step = 1;
450                        break;
451                    default: // decode invalid ESC sequence as chars
452                        sb.append(codec.decode(b, cur - 3, 3));
453                    }
454                    off = cur;
455                } else {
456                    cur += step > 0 ? step : b[cur] < 0 ? 2 : 1;
457                }
458            }
459            if (off < cur) {
460                sb.append(codec.decode(b, off, cur - off));
461            }
462            return sb.toString();
463        }
464    }

466    public static SpecificCharacterSet valueOf(String[] codes) {
467        if (codes == null || codes.length == 0)
468            return DEFAULT;
469
470        Codec[] infos = new Codec[codes.length];
471        for (int i = 0; i < codes.length; i++)
472            infos[i] = Codec.forCode(codes[i]);
473        return codes.length > 1 ? new ISO2022(infos)
474                : new SpecificCharacterSet(infos);
475    }

477    private static Encoder encoder(ThreadLocal<SoftReference<Encoder>> tl,
478            Codec codec) {
479        SoftReference<Encoder> sr;
480        Encoder enc;
481        if ((sr = tl.get()) == null || (enc = sr.get()) == null
482                || enc.codec != codec)
483            tl.set(new SoftReference<Encoder>(enc = new Encoder(codec)));
484        return enc;
485    }
486
487    protected SpecificCharacterSet(Codec... codecs) {
488        this.codecs = codecs;
489    }
490
491    public byte[] encode(String val, String delimiters) {
492        return codecs[0].encode(val);
493    }
494
495    public String decode(byte[] val) {
496        return codecs[0].decode(val, 0, val.length);
497    }
