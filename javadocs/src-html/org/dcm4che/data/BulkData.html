039package org.dcm4che.data;
040
041import java.io.File;
042import java.io.IOException;
043import java.io.InputStream;
044import java.io.ObjectInputStream;
045import java.io.ObjectOutputStream;
046import java.net.URI;
047import java.net.URISyntaxException;
048import java.util.StringTokenizer;
049
050import org.dcm4che.io.DicomEncodingOptions;
051import org.dcm4che.io.DicomOutputStream;
052import org.dcm4che.util.ByteUtils;
053import org.dcm4che.util.StreamUtils;
054
055/**
056 * @author Gunter Zeilinger <gunterze@gmail.com>
057 */
058public class BulkData implements Value {
059
060    public static final int MAGIC_LEN = 0xfbfb;
061
062    public final String uri;
063    public final String transferSyntax;
064    public final long offset;
065    public final int length;
066
067    public BulkData(String uri, String transferSyntax, long offset,
068            int length) {
069        if (transferSyntax == null)
070            throw new NullPointerException("transferSyntax");
071        this.uri = uri;
072        this.transferSyntax = transferSyntax;
073        this.offset = offset;
074        this.length = length;
075    }
076
077    @Override
078    public boolean isEmpty() {
079        return length == 0;
080    }
081
082    @Override
083    public String toString() {
084        return "BulkData[uri=" +  uri 
085                + ", tsuid=" + transferSyntax
086                + ", offset=" + offset
087                + ", length=" + length + "]";
088    }
089
090    public String toURI() {
091        return uri + "?transferSyntax=" + transferSyntax
092                   + ",offset=" + offset
093                   + ",length=" + length;
094    }
095
096    public static BulkData fromURI(String s) {
097        String transferSyntax = UID.ImplicitVRLittleEndian;
098        int offset = 0;
099        int length = -1;
100        StringTokenizer stk = new StringTokenizer(s, "?=, '\"");
101        String uri = stk.nextToken();
102        while (stk.hasMoreTokens()) {
103            String tk = stk.nextToken();
104            if (tk.equals("transferSyntax"))
105                transferSyntax = stk.nextToken();
106            else if (tk.equals("offset"))
107                offset = Integer.parseInt(stk.nextToken());
108            else if (tk.equals("length"))
109                length = Integer.parseInt(stk.nextToken());
110        }
111        return new BulkData(uri, transferSyntax, offset, length);
112    }
113
114    public File getFile() {
115        try {
116            return new File(new URI(uri));
117        } catch (URISyntaxException e) {
118            throw new IllegalStateException("uri: " + uri);
119        } catch (IllegalArgumentException e) {
120            throw new IllegalStateException("uri: " + uri);
121        }
122    }
123
124    public InputStream openStream() throws IOException {
125        try {
126            return new URI(uri).toURL().openStream();
127        } catch (URISyntaxException e) {
128            throw new AssertionError(e);
129        }
130    }
131
132    @Override
133    public int calcLength(DicomEncodingOptions encOpts, boolean explicitVR, VR vr) {
134        return getEncodedLength(encOpts, explicitVR, vr);
135    }
136
137    @Override
138    public int getEncodedLength(DicomEncodingOptions encOpts, boolean explicitVR, VR vr) {
139        return (length + 1) & ~1;
140    }
141
142    @Override
143    public byte[] toBytes(VR vr, boolean bigEndian) throws IOException {
144        if (length == 0)
145            return ByteUtils.EMPTY_BYTES;
146
147        InputStream in = openStream();
148        try {
149            StreamUtils.skipFully(in, offset);
150            byte[] b = new byte[length];
151            StreamUtils.readFully(in, b, 0, b.length);
152            if (transferSyntax.equals(UID.ExplicitVRBigEndian) 
153                    ? !bigEndian
154                    : bigEndian) {
155                vr.toggleEndian(b, false);
156            }
157            return b;
158        } finally {
159            in.close();
160        }
161
162    }
163
164    @Override
165    public void writeTo(DicomOutputStream out, VR vr) throws IOException {
166        InputStream in = openStream();
167        try {
168            StreamUtils.skipFully(in, offset);
169            if (transferSyntax.equals(UID.ExplicitVRBigEndian)
170                    ? !out.isBigEndian()
171                    : out.isBigEndian())
172                StreamUtils.copy(in, out, length, vr.numEndianBytes());
173            else
174                StreamUtils.copy(in, out, length);
175            if ((length & 1) != 0)
176                out.write(vr.paddingByte());
177        } finally {
178            in.close();
179        }
180    }
181
182    public void serializeTo(ObjectOutputStream oos) throws IOException {
183        oos.writeInt(length);
184        oos.writeLong(offset);
185        oos.writeUTF(uri);
186        oos.writeUTF(transferSyntax);
187    }
188
189    public static Value deserializeFrom(ObjectInputStream ois)
190            throws IOException {
191        int len = ois.readInt();
192        long off = ois.readLong();
193        String uri = ois.readUTF();
194        String tsuid = ois.readUTF();
195        return new BulkData(uri, tsuid, off, len);
196    }
197}
