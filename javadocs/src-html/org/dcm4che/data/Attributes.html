213    public void trimToSize(boolean recursive) {
214        int oldCapacity = tags.length;
215        if (size < oldCapacity) {
216            tags = Arrays.copyOf(tags, size);
217            vrs = Arrays.copyOf(vrs, size);
218            values = Arrays.copyOf(values, size);
219        }
220        if (recursive)
221            for (Object value : values) {
222                if (value instanceof Sequence) {
223                    ((Sequence) value).trimToSize(recursive);
224                } else if (value instanceof Fragments)
225                    ((Fragments) value).trimToSize();
226            }

229    public void internalizeStringValues(boolean decode) {
230        SpecificCharacterSet cs = getSpecificCharacterSet();
231        for (int i = 0; i < values.length; i++) {
232            VR vr = vrs[i];
233            Object value = values[i];
234            if (vr.isStringType()) {
235                if (value instanceof byte[]) {
236                    if (!decode)
237                        continue;
238                    value = vr.toStrings((byte[]) value, bigEndian, cs);
239                }
240                if (value instanceof String)
241                    values[i] = ((String) value).intern();
242                else if (value instanceof String[]) {
243                    String[] ss = (String[]) value;
244                    for (int j = 0; j < ss.length; j++)
245                        ss[j] = ss[j].intern();
246                }
247            } else if (value instanceof Sequence)
248                for (Attributes item : (Sequence) value)
249                    item.internalizeStringValues(decode);
250        }
251    }

253    public void decodeStringValues() {
254        Object value;
255        VR vr;
256        SpecificCharacterSet cs = getSpecificCharacterSet();
257        for (int i = 0; i < values.length; i++) {
258            value = values[i];
259            if (value instanceof Sequence) {
260                for (Attributes item : (Sequence) value)
261                    item.decodeStringValues();
262            } else if ((vr = vrs[i]).isStringType())
263                if (value instanceof byte[])
264                    values[i] =
265                        vr.toStrings((byte[]) value, bigEndian, cs);
266        }
267    }

339    private int creatorTagOf(String privateCreator, int tag, boolean reserve) {
340        if (!TagUtils.isPrivateGroup(tag))
341            throw new IllegalArgumentException(TagUtils.toString(tag)
342                    + " is not a private Data Element");
343
344        int group = tag & 0xffff0000;
345        int creatorTag = group | 0x10;
346        int index = indexOf(creatorTag);
347        if (index < 0)
348            index = -index-1;
349        while (index < size && (tags[index] & 0xffffff00) == group) {
350            creatorTag = tags[index];
351            if (vrs[index] == VR.LO) {
352                Object creatorID = decodeStringValue(index);
353                if (privateCreator.equals(creatorID))
354                    return creatorTag;
355            }
356            index++;
357            creatorTag++;
358        }
359        if (!reserve)
360            return -1;
361
362        if ((creatorTag & 0xff00) != 0)
363            throw new IllegalStateException("No free block for Private Element "
364                    + TagUtils.toString(tag));
365        setString(creatorTag, VR.LO, privateCreator);
366        return creatorTag;
367    }

369    private Object decodeStringValue(int index) {
370        Object value = values[index];
371        if (value instanceof byte[]) {
372            value = vrs[index].toStrings((byte[]) value, bigEndian,
373                        getSpecificCharacterSet());
374            if (value instanceof String && ((String) value).isEmpty())
375                value = Value.NULL;
376            values[index] = value;
377        }
378        return value;
379    }

381    private double[] decodeDSValue(int index) {
382        Object value = values[index];
383        if (value == Value.NULL)
384            return ByteUtils.EMPTY_DOUBLES;
385
386        if (value instanceof double[])
387            return (double[]) value;
388
389        double[] ds;
390        if (value instanceof byte[])
391            value = vrs[index].toStrings((byte[]) value, bigEndian,
392                        getSpecificCharacterSet());
393        if (value instanceof String) {
394            String s = (String) value;
395            if (s.isEmpty()) {
396                values[index] = Value.NULL;
397                return ByteUtils.EMPTY_DOUBLES;
398            }
399            ds = new double[] { StringUtils.parseDS(s) };
400        } else { // value instanceof String[]
401            String[] ss = (String[]) value;
402            ds = new double[ss.length];
403            for (int i = 0; i < ds.length; i++) {
404                String s = ss[i];
405                ds[i] = (s != null && !s.isEmpty())
406                        ? StringUtils.parseDS(s)
407                        : Double.NaN;
408            }
409        }
410        values[index] = ds;
411        return ds;
412    }

414    private int[] decodeISValue(int index) {
415        Object value = values[index];
416        if (value == Value.NULL)
417            return ByteUtils.EMPTY_INTS;
418
419        if (value instanceof int[])
420            return (int[]) value;
421
422        int[] is;
423        if (value instanceof byte[])
424            value = vrs[index].toStrings((byte[]) value, bigEndian,
425                        getSpecificCharacterSet());
426        if (value instanceof String) {
427            String s = (String) value;
428            if (s.isEmpty()) {
429                values[index] = Value.NULL;
430                return ByteUtils.EMPTY_INTS;
431            }
432            is = new int[] { StringUtils.parseIS(s) };
433        } else { // value instanceof String[]
434            String[] ss = (String[]) value;
435            is = new int[ss.length];
436            for (int i = 0; i < is.length; i++) {
437                String s = ss[i];
438                is[i] = (s != null && !s.isEmpty())
439                            ? StringUtils.parseIS(s)
440                            : Integer.MIN_VALUE;
441            }
442        }
443        values[index] = is;
444        return is;
445    }

483    public String privateCreatorOf(int tag) {
484        if (!TagUtils.isPrivateTag(tag))
485            return null;
486
487        int creatorTag = (tag & 0xffff0000) | ((tag >>> 8) & 0xff);
488        int index = indexOf(creatorTag);
489        if (index < 0 || vrs[index] != VR.LO || values[index] == Value.NULL)
490            return null;
491        
492        Object value = decodeStringValue(index);
493        if (value == Value.NULL)
494            return null;
495
496        return VR.LO.toString(value, false, 0, null);
497    }

630        int index = indexOf(privateCreator, tag);
631        if (index < 0)
632            return defVal;
633
634        Object value = values[index];
635        if (value == Value.NULL)
636            return defVal;
637
638        if (vr == null)
639            vr = vrs[index];
640        else
641            updateVR(index, vr);
642        if (vr.isStringType()) {
643            value = decodeStringValue(index);
644            if (value == Value.NULL)
645                return defVal;
646        }
647
648        try {
649            return vr.toString(value, bigEndian, valueIndex, defVal);
650        } catch (UnsupportedOperationException e) {
651            LOG.info("Attempt to access {} {} as string", TagUtils.toString(tag), vr);
652            return defVal;
653        }
654    }

1585    private boolean add(Attributes other, int[] include, int[] exclude,
1586            int fromIndex, int toIndex, boolean merge, boolean update,
1587            Attributes modified) {
1588        boolean toggleEndian = bigEndian != other.bigEndian;
1589        boolean modifiedToggleEndian = modified != null
1590                && bigEndian != modified.bigEndian;
1591        final int[] tags = other.tags;
1592        final VR[] srcVRs = other.vrs;
1593        final Object[] srcValues = other.values;
1594        final int otherSize = other.size;
1595        int numAdd = 0;
1596        String privateCreator = null;
1597        int creatorTag = 0;
1598        for (int i = 0; i < otherSize; i++) {
1599            int tag = tags[i];
1600            VR vr = srcVRs[i];
1601            Object value = srcValues[i];
1602            if (TagUtils.isPrivateCreator(tag)) {
1603                if (contains(tag))
1604                    continue; // do not overwrite private creator IDs
1605
1606                if (vr == VR.LO) {
1607                    value = other.decodeStringValue(i);
1608                    if ((value instanceof String)
1609                            && creatorTagOf((String) value, tag, false) != -1)
1610                        continue; // do not add duplicate private creator ID
1611                }
1612            }
1613            if (include != null && Arrays.binarySearch(include, fromIndex, toIndex, tag) < 0)
1614                continue;
1615            if (exclude != null && Arrays.binarySearch(exclude, fromIndex, toIndex, tag) >= 0)
1616                continue;
1617            if (TagUtils.isPrivateTag(tag)) {
1618                int tmp = TagUtils.creatorTagOf(tag);
1619                if (creatorTag != tmp) {
1620                    creatorTag = tmp;
1621                    privateCreator = other.privateCreatorOf(tag);
1622                }
1623            } else {
1624                creatorTag = 0;
1625                privateCreator = null;
1626            }
1627            if (merge || update) {
1628                int j = indexOf(tag);
1629                if (j >= 0) {
1630                    if (update && equalValues(other, j, i)) {
1631                        continue;
1632                    }
1633                    Object origValue = vrs[j].isStringType()
1634                            ? decodeStringValue(j)
1635                            : values[j];
1636                    if (!isEmpty(origValue)) {
1637                        if (merge) {
1638                            continue;
1639                        }
1640                        if (modified != null) {
1641                            if (origValue instanceof Sequence) {
1642                                modified.set(privateCreator, tag, (Sequence) origValue);
1643                            } else if (origValue instanceof Fragments) {
1644                                modified.set(privateCreator, tag, (Fragments) origValue);
1645                            } else {
1646                                modified.set(privateCreator, tag, vr,
1647                                        toggleEndian(vr, origValue, modifiedToggleEndian));
1648                            }
1649                        }
1650                    }
1651                }
1652            }
1653            if (value instanceof Sequence) {
1654                set(privateCreator, tag, (Sequence) value);
1655            } else if (value instanceof Fragments) {
1656                set(privateCreator, tag, (Fragments) value);
1657            } else {
1658                set(privateCreator, tag, vr,
1659                        toggleEndian(vr, value, toggleEndian));
1660            }
1661            numAdd++;
1662        }
1663        return numAdd != 0;
1664    }
