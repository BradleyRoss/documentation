129        public String toString() {
130            StringBuilder sb = new StringBuilder();
131            sb.append("<DataElement keyword=\"")
132              .append(ElementDictionary.getStandardElementDictionary().keywordOf(tag))
133              .append("\" tag=\"").append(TagUtils.toHexString(tag))
134              .append("\" vr=\"").append(vr)
135              .append("\" type=\"").append(type);
136            if (condition != null)
137                sb.append('C');
138            sb.append(vr == VR.SQ ? "\" items=\"" : "\" vm=\"");
139            if (minVM == -1)
140                sb.append('n');
141            else
142                sb.append(minVM);
143            if (maxVM != minVM) {
144                sb.append('-');
145                if (maxVM == -1)
146                    sb.append('n');
147                else
148                    sb.append(maxVM);
149            }
150            if (valueNumber != 0)
151                sb.append("\" valueNumber=\"").append(valueNumber);
152            if (values instanceof String[] || values instanceof int[]) {
153                sb.append("\">").append(StringUtils.LINE_SEPARATOR);
154                if (values instanceof String[])
155                    for (String val : (String[]) values)
156                        sb.append("  <Value>").append(val).append("/Value")
157                          .append(StringUtils.LINE_SEPARATOR);
158                else 
159                    for (int val : (int[]) values)
160                        sb.append("  <Value>").append(val).append("/Value")
161                          .append(StringUtils.LINE_SEPARATOR);
162                sb.append("</DataElement>");
163            } else {
164                sb.append("\"/>");
165            }
166            return sb.toString();
167        }

171    public abstract static class Condition {
172         protected String id;
173         protected boolean not;
174
175         public Condition id(String id) {
176             this.id = id;
177             return this;
178         }
179
180         public final String id() {
181             return id;
182         }
183
184         public final Condition not() {
185             this.not = !not;
186             return this;
187         }
188
189         public abstract boolean match(Attributes attrs);
190
191         public void addChild(Condition child) {
192             throw new UnsupportedOperationException();
193         }
194
195         public Condition trim() {
196             return this;
197         }
198
199         public boolean isEmpty() {
200             return false;
201         }
202    }
203
204    abstract static class CompositeCondition extends Condition  {
205        protected final ArrayList<Condition> childs = new ArrayList<Condition>();
206
207        public abstract boolean match(Attributes attrs);
208
209        @Override
210        public void addChild(Condition child) {
211            childs.add(child);
212        }
213
214        @Override
215        public Condition trim() {
216            int size = childs.size();
217            if (size == 1) {
218                Condition child = childs.get(0).id(id);
219                return not ? child.not() : child;
220            }
221            childs.trimToSize();
222            return this;
223        }
224
225        @Override
226        public boolean isEmpty() {
227            return childs.isEmpty();
228        }
229    }
230
231    public static class And extends CompositeCondition {
232
233        public boolean match(Attributes attrs) {
234            for (Condition child : childs) {
235                if (!child.match(attrs))
236                    return not;
237            }
238            return !not;
239        }
240   }
241
242    public static class Or extends CompositeCondition {
243
244        public boolean match(Attributes attrs) {
245            for (Condition child : childs) {
246                if (child.match(attrs))
247                    return !not;
248            }
249            return not;
250        }
251   }

253    public static class Present extends Condition {
254        protected final int tag;
255        protected final int[] itemPath;
256
257        public Present(int tag, int... itemPath) {
258            this.tag = tag;
259            this.itemPath = itemPath;
260        }
261
262        public boolean match(Attributes attrs) {
263            return not ? !item(attrs).containsValue(tag)
264                        : item(attrs).containsValue(tag);
265        }
266
267        protected Attributes item(Attributes attrs) {
268            for (int sqtag : itemPath) {
269                if (sqtag == -1)
270                attrs = (sqtag == -1)
271                        ? attrs.getParent()
272                        : attrs.getNestedDataset(sqtag);
273            }
274            return attrs;
275        }
276    }

278    public static class MemberOf extends Present {
279        private final VR vr;
280        private final int valueIndex;
281        private final boolean matchNotPresent;
282        private Object values;
283
284        public MemberOf(int tag, VR vr, int valueIndex,
285                boolean matchNotPresent, int... itemPath) {
286            super(tag, itemPath);
287            this.vr = vr;
288            this.valueIndex = valueIndex;
289            this.matchNotPresent = matchNotPresent;
290        }
291
292        public VR vr() {
293            return vr;
294        }
295
296        public MemberOf setValues(String[] values) {
297            this.values = values;
298            return this;
299        }
300
301        public MemberOf setValues(int[] values) {
302            this.values = values;
303            return this;
304        }
305
306        public boolean match(Attributes attrs) {
307            if (values == null)
308                throw new IllegalStateException("values not initialized");
309            Attributes item = item(attrs);
310            if (item == null)
311                return matchNotPresent;
312
313            if (values instanceof int[])
314                return not ? !match(item, ((int[]) values))
315                           : match(item, ((int[]) values));
316            else
317                return not ? !match(item, ((String[]) values))
318                           : match(item, ((String[]) values));
319        }
320
321        private boolean match(Attributes item, String[] ss) {
322            String val = item.getString(tag, valueIndex);
323            if (val == null)
324                return not ? !matchNotPresent : matchNotPresent;
325            for (String s : ss) {
326                if (s.equals(val))
327                    return !not;
328            }
329            return not;
330        }
331
332        private boolean match(Attributes item, int[] is) {
333            int val = item.getInt(tag, valueIndex, Integer.MIN_VALUE);
334            if (val == Integer.MIN_VALUE)
335                return matchNotPresent;
336            for (int i : is) {
337                if (i == val)
338                    return true;
339            }
340            return false;
341        }

345    public void parse(String uri) throws IOException {
346        try {
347            SAXParserFactory f = SAXParserFactory.newInstance();
348            SAXParser parser = f.newSAXParser();
349            parser.parse(uri, new SAXHandler(this));
350        } catch (SAXException e) {
351            throw new IOException("Failed to parse " + uri, e);
352        } catch (ParserConfigurationException e) {
353            throw new RuntimeException(e);
354        }
355    }

357    private static class SAXHandler extends DefaultHandler {
358
359        private DataElement el;
360        private StringBuilder sb = new StringBuilder();
361        private boolean processCharacters;
362        private String idref;
363        private List<String> values = new ArrayList<String>();
364        private LinkedList<IOD> iodStack = new LinkedList<IOD>();
365        private LinkedList<Condition> conditionStack = new LinkedList<Condition>();
366        private Map<String, IOD> id2iod = new HashMap<String, IOD>();
367        private Map<String, Condition> id2cond = new HashMap<String, Condition>();
368
369        public SAXHandler(IOD iod) {
370            iodStack.add(iod);
371        }
372
373        @Override
374        public void startElement(String uri, String localName, String qName,
375                org.xml.sax.Attributes atts) throws SAXException {
376            switch (qName.charAt(0)) {
377            case 'A':
378                if (qName.equals("And"))
379                    startCondition(qName, new And());
380                break;
381            case 'D':
382                if (qName.equals("DataElement"))
383                    startDataElement(
384                            atts.getValue("tag"),
385                            atts.getValue("vr"),
386                            atts.getValue("type"),
387                            atts.getValue("vm"),
388                            atts.getValue("items"),
389                            atts.getValue("valueNumber"));
390                break;
391            case 'I':
392                if (qName.equals("If"))
393                    startIf(atts.getValue("id"), atts.getValue("idref"));
394                else if (qName.equals("Item"))
395                    startItem(atts.getValue("id"), atts.getValue("idref"));
396                break;
397            case 'M':
398                if (qName.equals("MemberOf"))
399                    startCondition(qName, memberOf(atts));
400                break;
401            case 'N':
402                if (qName.equals("NotAnd"))
403                    startCondition(qName, new And().not());
404                else if (qName.equals("NotMemberOf"))
405                    startCondition(qName, memberOf(atts).not());
406                else if (qName.equals("NotOr"))
407                    startCondition(qName, new Or().not());
408                else if (qName.equals("NotPresent"))
409                    startCondition(qName, present(atts).not());
410                break;
411            case 'O':
412                if (qName.equals("Or"))
413                    startCondition(qName, new Or());
414                break;
415            case 'P':
416                if (qName.equals("Present"))
417                    startCondition(qName, present(atts));
418                break;
419            case 'V':
420                if (qName.equals("Value"))
421                    startValue();
422                break;
423            }
424        }
425
426        private Present present(org.xml.sax.Attributes atts)
427                throws SAXException {
428            int[] tagPath = tagPathOf(atts.getValue("tag"));
429            int lastIndex = tagPath.length-1;
430            return new Present(tagPath[lastIndex],
431                    lastIndex > 0 ? Arrays.copyOf(tagPath, lastIndex)
432                            : ByteUtils.EMPTY_INTS);
433        }
434
435        private MemberOf memberOf(org.xml.sax.Attributes atts)
436                throws SAXException {
437            int[] tagPath = tagPathOf(atts.getValue("tag"));
438            int lastIndex = tagPath.length-1;
439            return new MemberOf(
440                    tagPath[lastIndex],
441                    vrOf(atts.getValue("vr")),
442                    valueNumberOf(atts.getValue("valueNumber"), 1) - 1,
443                    matchNotPresentOf(atts.getValue("matchNotPresent")),
444                    lastIndex > 0 ? Arrays.copyOf(tagPath, lastIndex)
445                                  : ByteUtils.EMPTY_INTS);
446        }
447
448        @Override
449        public void endElement(String uri, String localName, String qName)
450                throws SAXException {
451            switch (qName.charAt(0)) {
452            case 'A':
453                if (qName.equals("And"))
454                    endCondition(qName);
455                break;
456            case 'D':
457                if (qName.equals("DataElement"))
458                    endDataElement();
459                break;
460            case 'I':
461                if (qName.equals("If"))
462                    endCondition(qName);
463                else if (qName.equals("Item"))
464                    endItem();
465                break;
466            case 'M':
467                if (qName.equals("MemberOf"))
468                    endCondition(qName);
469                break;
470            case 'N':
471                if (qName.equals("NotAnd"))
472                    endCondition(qName);
473                else if (qName.equals("NotMemberOf"))
474                    endCondition(qName);
475                else if (qName.equals("NotOr"))
476                    endCondition(qName);
477                else if (qName.equals("NotPresent"))
478                    endCondition(qName);
479                break;
480            case 'O':
481                if (qName.equals("Or"))
482                    endCondition(qName);
483                break;
484            case 'P':
485                if (qName.equals("Present"))
486                    endCondition(qName);
487                break;
488            case 'V':
489                if (qName.equals("Value"))
490                    endValue();
491                break;
492            }
493            processCharacters = false;
494            idref = null;
495        }
496
497        @Override
498        public void characters(char[] ch, int start, int length)
499                throws SAXException {
500            if (processCharacters)
501                sb.append(ch, start, length);
502        }

504        private void startDataElement(String tagStr, String vrStr,
505                String typeStr, String vmStr, String items,
506                String valueNumberStr) throws SAXException {
507            if (idref != null)
508                throw new SAXException("<Item> with idref must be empty");
509
510            IOD iod = iodStack.getLast();
511            int tag = tagOf(tagStr);
512            VR vr = vrOf(vrStr);
513            DataElementType type = typeOf(typeStr);
514            
515            int minVM = -1;
516            int maxVM = -1;
517            String vm = vr == VR.SQ ? items : vmStr;
518            if (vm != null) {
519                try {
520                    String[] ss = StringUtils.split(vm, '-');
521                    if (ss[0].charAt(0) != 'n') {
522                        minVM = Integer.parseInt(ss[0]);
523                        if (ss.length > 1) {
524                            if (ss[1].charAt(0) != 'n')
525                                maxVM = Integer.parseInt(ss[1]);
526                        } else {
527                            maxVM = minVM;
528                        }
529                    }
530                } catch (IllegalArgumentException e) {
531                    throw new SAXException(
532                            (vr == VR.SQ ? "invalid items=\"" 
533                                         : "invalid vm=\"")
534                            + vm + '"');
535                }
536            }
537            el = new DataElement(tag, vr, type, minVM, maxVM,
538                    valueNumberOf(valueNumberStr, 0));
539            iod.add(el);
540        }

542        private DataElementType typeOf(String s) throws SAXException {
543            try {
544                return DataElementType.valueOf("TYPE_" + s);
545            } catch (NullPointerException e) {
546                throw new SAXException("missing type attribute");
547            } catch (IllegalArgumentException e) {
548                throw new SAXException("unrecognized type=\"" + s + '"');
549            }
550        }
551
552        private VR vrOf(String s) throws SAXException {
553            try {
554                return VR.valueOf(s);
555            } catch (NullPointerException e) {
556                throw new SAXException("missing vr attribute");
557            } catch (IllegalArgumentException e) {
558                throw new SAXException("unrecognized vr=\"" + s + '"');
559            }
560        }
561
562        private int tagOf(String s) throws SAXException {
563            try {
564               return (int) Long.parseLong(s, 16);
565            } catch (NullPointerException e) {
566                throw new SAXException("missing tag attribute");
567            } catch (IllegalArgumentException e) {
568                throw new SAXException("invalid tag=\"" + s + '"');
569            }
570        }
571
572        private int[] tagPathOf(String s) throws SAXException {
573            String[] ss = StringUtils.split(s, '/');
574            if (ss.length == 0)
575                throw new SAXException("missing tag attribute");
576            
577            try {
578                int[] tagPath = new int[ss.length];
579                for (int i = 0; i < tagPath.length; i++)
580                    tagPath[i] = ss[i].equals("..") 
581                                ? -1
582                                : (int) Long.parseLong(s, 16);
583                return tagPath;
584            } catch (IllegalArgumentException e) {
585                throw new SAXException("invalid tag=\"" + s + '"');
586            }
587        }
588
589
590        private int valueNumberOf(String s, int def) throws SAXException {
591            try {
592               return s != null ? Integer.parseInt(s) : def;
593            } catch (IllegalArgumentException e) {
594                throw new SAXException("invalid valueNumber=\"" + s + '"');
595            }
596        }
597
598        private boolean matchNotPresentOf(String s) {
599            return s != null && s.equalsIgnoreCase("true");
600        }
601
602
603        private void endDataElement() {
604            if (values.isEmpty())
605                return;
606
607            if (el.vr.isIntType())
608                el.setValues(parseInts(values));
609            else
610                el.setValues(values.toArray(new String[values.size()]));
611            values.clear();
612        }
613
614        private int[] parseInts(List<String> list) {
615            int[] is = new int[list.size()];
616            for (int i = 0; i < is.length; i++)
617                is[i] = Integer.parseInt(list.get(i));
618            return is;
619        }

621        private void startValue() {
622            sb.setLength(0);
623            processCharacters = true;
624        }
625
626        private void endValue() {
627            values.add(sb.toString());
628        }

630        private void startItem(String id, String idref) throws SAXException {
631            IOD iod;
632            if (idref != null) {
633                iod = id2iod.get(idref);
634                if (iod == null)
635                    throw new SAXException(
636                            "could not resolve <Item idref:\"" + idref + "\"/>");
637            } else { 
638                iod = new IOD();
639            }
640            el.setItemIOD(iod);
641            iodStack.add(iod);
642            if (id != null)
643                id2iod.put(id, iod);
644
645            this.idref = idref;
646        }
647
648        private void endItem() {
649            iodStack.removeLast().trimToSize();
650        }
651
652        private void startIf(String id, String idref) throws SAXException {
653            if (el == null || !conditionStack.isEmpty())
654                throw new SAXException("unexpected <If>");
655
656            Condition cond;
657            if (idref != null) {
658                cond = id2cond.get(idref);
659                if (cond == null)
660                    throw new SAXException(
661                            "could not resolve <If idref:\"" + idref + "\"/>");
662            } else { 
663                cond = new And().id(id);
664            }
665            conditionStack.add(cond);
666            if (id != null)
667                id2cond.put(id, cond);
668            this.idref = idref;
669        }
670
671       private void startCondition(String name, Condition cond)
672               throws SAXException {
673            try {
674                conditionStack.add(cond);
675            } catch (Exception e) {
676                throw new SAXException("unexpected <" + name + '>');
677            }
678        }
679
680        private void endCondition(String name) throws SAXException {
681            Condition cond = conditionStack.removeLast();
682            if (cond.isEmpty())
683                throw new SAXException('<' + name + "> must not be empty");
684
685            if (!values.isEmpty()) {
686                if (!(cond instanceof MemberOf))
687                    throw new SAXException('<' + name + "> cannot contains <Value>");
688                
689                MemberOf memberOf = (MemberOf) cond;
690                if (memberOf.vr.isIntType())
691                    memberOf.setValues(parseInts(values));
692                else
693                    memberOf.setValues(values.toArray(new String[values.size()]));
694                values.clear();
695            }
696
697            if (conditionStack.isEmpty())
698                el.setCondition(cond.trim());
699            else
700                conditionStack.getLast().addChild(cond.trim());
701        }

704    public static IOD load(String uri) throws IOException {
705        if (uri.startsWith("resource:")) {
706            try {
707                uri = Thread.currentThread().getContextClassLoader()
708                        .getResource(uri.substring(9)).toString();
709            } catch (NullPointerException npe) {
710                throw new FileNotFoundException(uri);
711            }
712        } else if (uri.indexOf(':') < 2) {
713            uri = new File(uri).toURI().toString();
714        }
715        IOD iod = new IOD();
716        iod.parse(uri);
717        iod.trimToSize();
718        return iod;
719    }
