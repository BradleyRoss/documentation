039package org.dcm4che.sample.servlet;
040
041import java.lang.management.ManagementFactory;
042
043import javax.management.ObjectInstance;
044import javax.management.ObjectName;
045import javax.servlet.ServletConfig;
046import javax.servlet.ServletException;
047import javax.servlet.http.HttpServlet;
048
049import org.dcm4che.conf.api.DicomConfiguration;
050
051/**
052 * @author Gunter Zeilinger <gunterze@gmail.com>
053 *
054 */
055@SuppressWarnings("serial")
056public class EchoSCPServlet extends HttpServlet {
057
058    private ObjectInstance mbean;
059
060    private EchoSCP echoSCP;
061
062    private DicomConfiguration dicomConfig;
063
064    @Override
065    public void init(ServletConfig config) throws ServletException {
066        super.init(config);
067        try {
068            dicomConfig = (DicomConfiguration) Class.forName(
069                    config.getInitParameter("dicomConfigurationClass"), false,
070                    Thread.currentThread().getContextClassLoader()).newInstance();
071            echoSCP = new EchoSCP(dicomConfig,
072                    config.getInitParameter("deviceName"));
073            echoSCP.start();
074            mbean = ManagementFactory.getPlatformMBeanServer()
075                    .registerMBean(echoSCP, 
076                            new ObjectName(config.getInitParameter("jmxName")));
077        } catch (Exception e) {
078            destroy();
079            throw new ServletException(e);
080        }
081    }
082
083    @Override
084    public void destroy() {
085        if (mbean != null)
086            try {
087                ManagementFactory.getPlatformMBeanServer()
088                    .unregisterMBean(mbean.getObjectName());
089            } catch (Exception e) {
090                e.printStackTrace();
091            }
092        if (echoSCP != null)
093            echoSCP.stop();
094        if (dicomConfig != null)
095            dicomConfig.close();
096    }
097
098}
