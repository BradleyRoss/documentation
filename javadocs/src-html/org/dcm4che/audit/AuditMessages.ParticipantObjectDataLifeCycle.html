587    public static EventIdentification createEventIdentification(
588            EventID eventID, String action, Calendar eventDateTime,
589            String outcome, String outcomeDescription, org.dcm4che.audit.EventTypeCode... types) {
590        EventIdentification ei = new EventIdentification();
591        ei.setEventID(eventID);
592        ei.setEventDateTime(
593                eventDateTime != null ? eventDateTime : Calendar.getInstance());
594        ei.setEventActionCode(action);
595        ei.setEventOutcomeIndicator(outcome);
596        ei.setEventOutcomeDescription(outcomeDescription);
597        for (org.dcm4che.audit.EventTypeCode type : types)
598            ei.getEventTypeCode().add(type);
599        return ei;
600    }

723    public static void toXML(AuditMessage message, OutputStream os,
724            boolean format, String encoding, String schemaURI)
725            throws IOException {
726        try {
727            Marshaller m = jc().createMarshaller();
728            if (format)
729                m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
730            if (schemaURI != null)
731                m.setProperty(Marshaller.JAXB_NO_NAMESPACE_SCHEMA_LOCATION,
732                        schemaURI);
733            if (encoding != null)
734                m.setProperty(Marshaller.JAXB_ENCODING, encoding);
735            m.marshal(of.createAuditMessage(message), os );
736        } catch( JAXBException jbe ){
737            if (jbe.getLinkedException() instanceof IOException)
738                throw (IOException) jbe.getLinkedException();
739            throw new IllegalStateException(jbe);
740        }
741    }
