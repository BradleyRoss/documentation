091    public Compressor(Attributes dataset, String tsuid) {
092        super(dataset, tsuid);
093
094        Object pixeldata = dataset.getValue(Tag.PixelData);
095        if (pixeldata == null)
096            return;
097
098        if (pixeldata instanceof BulkDataLocator) {
099            this.pixeldata = (BulkDataLocator) pixeldata;
100            if (pmi.isSubSambled())
101                throw new UnsupportedOperationException(
102                        "Unsupported Photometric Interpretation: " + pmi);
103            if (this.pixeldata.length < length)
104                throw new IllegalArgumentException(
105                        "Pixel data too short: " + this.pixeldata.length
106                        + " instead " + length + " bytes");
107        }
108        embeddedOverlays = Overlays.getEmbeddedOverlayGroupOffsets(dataset);
109    }
110
111    public boolean compress(String tsuid, Property... params)
112            throws IOException {
113
114        if (tsuid == null)
115            throw new NullPointerException("desttsuid");
116
117        if (frames == 0)
118            return false;
119
120        ImageWriterFactory.ImageWriterParam param =
121                ImageWriterFactory.getImageWriterParam(tsuid);
122        if (param == null)
123            throw new UnsupportedOperationException(
124                    "Unsupported Transfer Syntax: " + tsuid);
125
126        this.compressor = ImageWriterFactory.getImageWriter(param);
127        this.patchJPEGLS = param.patchJPEGLS;
128
129        this.compressParam = compressor.getDefaultWriteParam();
130        Property[] imageWriteParams = param.getImageWriteParams();
131        if (imageWriteParams.length != 0 || params.length != 0)
132            compressParam.setCompressionMode(ImageWriteParam.MODE_EXPLICIT);
133        for (Property property : imageWriteParams)
134            property.setAt(compressParam);
135        for (Property property : params)
136            property.setAt(compressParam);
137
138        TransferSyntaxType tstype = TransferSyntaxType.forUID(tsuid);
139        adjustDestination(Math.min(bitsStored, tstype.getMaxBitsStored()),
140                signed && tstype.canEncodeSigned());
141        adjustAttributes(tsuid, tstype.getPlanarConfiguration());
142        Fragments compressedPixeldata = 
143                dataset.newFragments(Tag.PixelData, VR.OB, frames + 1);
144        compressedPixeldata.add(Value.NULL);
145        for (int i = 0; i < frames; i++) {
146            CompressedFrame frame = new CompressedFrame(i);
147            if (embeddedOverlays.length != 0)
148                frame.compress();
149            compressedPixeldata.add(frame);
150        }
151        for (int gg0000 : embeddedOverlays) {
152            dataset.setInt(Tag.OverlayBitsAllocated | gg0000, VR.US, 1);
153            dataset.setInt(Tag.OverlayBitPosition | gg0000, VR.US, 0);
154        }
155        return true;
156    }
157
158    private void adjustAttributes(String tsuid, int planarConfiguration) {
159        if (samples > 1) {
160            dataset.setString(Tag.PhotometricInterpretation, VR.CS, 
161                    (decompressor != null ? pmi.decompress() : pmi)
162                            .compress(tsuid).toString());
163 
164            dataset.setInt(Tag.PlanarConfiguration, VR.US, planarConfiguration);
165        }
166    }

168    public void close() {
169        if (iis != null)
170            try { iis.close(); } catch (IOException ignore) {}
171        dispose();
172    }

176        super.dispose();

179            compressor.dispose();
180
181        compressor = null;
182    }

184    private class CompressedFrame implements Value {
185
186        private int frameIndex;
187        private CacheOutputStream cacheout = new CacheOutputStream();
188        private MemoryCacheImageOutputStream cache;
189
190        public CompressedFrame(int frameIndex) throws IOException {
191            this.frameIndex = frameIndex;
192        }
193
194        @Override
195        public boolean isEmpty() {
196            return false;
197        }
198
199        @Override
200        public byte[] toBytes(VR vr, boolean bigEndian) throws IOException {
201            ByteArrayOutputStream out = new ByteArrayOutputStream();
202            writeTo(out);
203            return out.toByteArray();
204        }
205
206        @Override
207        public void writeTo(DicomOutputStream out, VR vr) throws IOException {
208            writeTo(out);
209        }
210
211        @Override
212        public int calcLength(DicomEncodingOptions encOpts, boolean explicitVR, VR vr) {
213            return getEncodedLength(encOpts, explicitVR, vr);
214        }
215
216        @Override
217        public int getEncodedLength(DicomEncodingOptions encOpts, boolean explicitVR, VR vr) {
218            try {
219                compress();
220            } catch (IOException e) {
221                return -1;
222            }
223            return (((int) cache.length()) + 1) & ~1;
224        }
225        
226        private void writeTo(OutputStream out) throws IOException {
227            compress();
228            try {
229                cacheout.set(out);
230                cache.flush();
231                if ((cache.length() & 1) != 0)
232                    out.write(0);
233            } finally {
234                try { cache.close(); } catch (IOException ignore) {}
235                cache = null;
236                LOG.debug("Flushed frame #{} from memory", frameIndex + 1);
237            }
238        }
239
240        private void compress() throws IOException {
241            if (cache != null)
242                return;
243
244            if (ex != null)
245                throw ex;
246
247            try {
248                BufferedImage bi = Compressor.this.readFrame(frameIndex);
249                Compressor.this.extractEmbeddedOverlays(frameIndex, bi);
250                if (bitsStored < bitsAllocated)
251                    Compressor.this.nullifyUnusedBits(bitsStored, bi);
252                cache = new MemoryCacheImageOutputStream(cacheout);
253                compressor.setOutput(patchJPEGLS != null
254                        ? new PatchJPEGLSImageOutputStream(cache, patchJPEGLS)
255                        : cache);
256                long start = System.currentTimeMillis();
257                compressor.write(null, new IIOImage(bi, null, null), compressParam);
258                long end = System.currentTimeMillis();
259                if (LOG.isDebugEnabled())
260                    LOG.debug("Compressed frame #{} {}:1 in {} ms", 
261                            new Object[] {frameIndex + 1,
262                            (float) sizeOf(bi) / cache.length(),
263                            end - start });
264            } catch (IOException ex) {
265                cache = null;
266                Compressor.this.ex = ex;
267                throw ex;
268            }
269        }
270
271    }

284    public BufferedImage readFrame(int frameIndex) throws IOException {
285        if (iis == null)
286            iis = new FileImageInputStream(file);
287
288        if (decompressor != null)
289            return decompressFrame(iis, frameIndex);
290
291        iis.setByteOrder(UID.ExplicitVRBigEndian.equals(pixeldata.transferSyntax) 
292                ? ByteOrder.BIG_ENDIAN
293                : ByteOrder.LITTLE_ENDIAN);
294        iis.seek(pixeldata.offset + frameLength * frameIndex);
295        DataBuffer db = destination.getRaster().getDataBuffer();
296        switch (db.getDataType()) {
297        case DataBuffer.TYPE_BYTE:
298            for (byte[] bs : ((DataBufferByte) db).getBankData())
299                iis.readFully(bs);
300            break;
301        case DataBuffer.TYPE_USHORT:
302            readFully(((DataBufferUShort) db).getData());
303            break;
304        case DataBuffer.TYPE_SHORT:
305            readFully(((DataBufferShort) db).getData());
306            break;
307        default:
308            throw new UnsupportedOperationException(
309                    "Unsupported Datatype: " + db.getDataType());
310        }
311        return destination;
312    }
313
314    private void nullifyUnusedBits(int bitsStored, BufferedImage bi) {
315        DataBuffer db = bi.getRaster().getDataBuffer();
316        switch (db.getDataType()) {
317        case DataBuffer.TYPE_USHORT:
318            nullifyUnusedBits(bitsStored, ((DataBufferUShort) db).getData());
319            break;
320        case DataBuffer.TYPE_SHORT:
321            nullifyUnusedBits(bitsStored, ((DataBufferShort) db).getData());
322            break;
323        }
324    }
325
326    private void nullifyUnusedBits(int bitsStored, short[] data) {
327        int mask = (1<<bitsStored)-1;
328        for (int i = 0; i < data.length; i++)
329            data[i] &= mask;
330    }
331
332    private void extractEmbeddedOverlays(int frameIndex, BufferedImage bi) {
333        for (int gg0000 : embeddedOverlays) {
334            int ovlyRow = dataset.getInt(Tag.OverlayRows | gg0000, 0);
335            int ovlyColumns = dataset.getInt(Tag.OverlayColumns | gg0000, 0);
336            int ovlyBitPosition = dataset.getInt(Tag.OverlayBitPosition | gg0000, 0);
337            int mask = 1 << ovlyBitPosition;
338            int ovlyLength = ovlyRow * ovlyColumns;
339            byte[] ovlyData = dataset.getSafeBytes(Tag.OverlayData | gg0000);
340            if (ovlyData == null) {
341                ovlyData = new byte[(((ovlyLength*frames+7)>>>3)+1)&(~1)];
342                dataset.setBytes(Tag.OverlayData | gg0000, VR.OB, ovlyData);
343            }
344            Overlays.extractFromPixeldata(bi.getRaster(), mask, ovlyData,
345                    ovlyLength * frameIndex, ovlyLength);
346            LOG.debug("Extracted embedded overlay #{} from bit #{} of frame #{}",
347                    new Object[]{(gg0000 >>> 17) + 1, ovlyBitPosition, frameIndex + 1});
348        }
349    }
350
351    private void readFully(short[] data) throws IOException {
352        iis.readFully(data, 0, data.length);
353    }
