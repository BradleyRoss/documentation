101    public Compressor(Attributes dataset, String from) {
102        super(dataset, from);
103
104        Object pixeldata = dataset.getValue(Tag.PixelData);
105        if (pixeldata == null)
106            return;
107
108        if (pixeldata instanceof BulkData) {
109            this.pixeldata = (BulkData) pixeldata;
110            if (pmi.isSubSambled())
111                throw new UnsupportedOperationException(
112                        "Unsupported Photometric Interpretation: " + pmi);
113            if (this.pixeldata.length < length)
114                throw new IllegalArgumentException(
115                        "Pixel data too short: " + this.pixeldata.length
116                        + " instead " + length + " bytes");
117        }
118        embeddedOverlays = Overlays.getEmbeddedOverlayGroupOffsets(dataset);
119    }

121    public boolean compress(String tsuid, Property... params)
122            throws IOException {
123
124        if (tsuid == null)
125            throw new NullPointerException("desttsuid");
126
127        if (frames == 0)
128            return false;
129
130        ImageWriterFactory.ImageWriterParam param =
131                ImageWriterFactory.getImageWriterParam(tsuid);
132        if (param == null)
133            throw new UnsupportedOperationException(
134                    "Unsupported Transfer Syntax: " + tsuid);
135
136        this.compressor = ImageWriterFactory.getImageWriter(param);
137        LOG.debug("Compressor: {}", compressor.getClass().getName());
138        this.patchJPEGLS = param.patchJPEGLS;
139
140        this.compressParam = compressor.getDefaultWriteParam();
141        int count = 0;
142        for (Property property : cat(param.getImageWriteParams(), params)) {
143            String name = property.getName();
144            if (name.equals("maxPixelValueError"))
145                this.maxPixelValueError = ((Number) property.getValue()).intValue();
146            else if (name.equals("avgPixelValueBlockSize"))
147                this.avgPixelValueBlockSize = ((Number) property.getValue()).intValue();
148            else {
149                if (count++ == 0)
150                    compressParam.setCompressionMode(
151                            ImageWriteParam.MODE_EXPLICIT);
152                property.setAt(compressParam);
153            }
154        }
155
156        if (maxPixelValueError >= 0) {
157            ImageReaderFactory.ImageReaderParam readerParam =
158                    ImageReaderFactory.getImageReaderParam(tsuid);
159            if (readerParam == null)
160                throw new UnsupportedOperationException(
161                        "Unsupported Transfer Syntax: " + tsuid);
162
163            this.verifier = ImageReaderFactory.getImageReader(readerParam);
164            this.verifyParam = verifier.getDefaultReadParam();
165            LOG.debug("Verifier: {}", verifier.getClass().getName());
166        }
167
168        TransferSyntaxType tstype = TransferSyntaxType.forUID(tsuid);
169        if (decompressor == null || super.tstype == TransferSyntaxType.RLE)
170            bi = createBufferedImage(
171                    Math.min(bitsStored, tstype.getMaxBitsStored()),
172                    super.tstype == TransferSyntaxType.RLE || banded,
173                    signed && tstype.canEncodeSigned());
174        Fragments compressedPixeldata = 
175                dataset.newFragments(Tag.PixelData, VR.OB, frames + 1);
176        compressedPixeldata.add(Value.NULL);
177        for (int i = 0; i < frames; i++) {
178            CompressedFrame frame = new CompressedFrame(i);
179            if (embeddedOverlays.length != 0)
180                frame.compress();
181            compressedPixeldata.add(frame);
182        }
183        if (samples > 1) {
184            dataset.setString(Tag.PhotometricInterpretation, VR.CS, 
185                    (decompressor != null ? pmi.decompress() : pmi)
186                            .compress(tsuid).toString());
187            dataset.setInt(Tag.PlanarConfiguration, VR.US, 
188                    tstype.getPlanarConfiguration());
189        }
190        for (int gg0000 : embeddedOverlays) {
191            dataset.setInt(Tag.OverlayBitsAllocated | gg0000, VR.US, 1);
192            dataset.setInt(Tag.OverlayBitPosition | gg0000, VR.US, 0);
193        }
194        return true;
195    }

197    private Property[] cat(Property[] a, Property[] b) {
198        if (a.length == 0)
199            return b;
200        if (b.length == 0)
201            return a;
202        Property[] c = new Property[a.length + b.length];
203        System.arraycopy(a, 0, c, 0, a.length);
204        System.arraycopy(b, 0, c, a.length, b.length);
205        return c;
206    }

208    public void close() {
209        if (iis != null)
210            try { iis.close(); } catch (IOException ignore) {}
211        dispose();
212    }

221        if (verifier != null)
222            verifier.dispose();
223
224        compressor = null;
225        verifier = null;

228    private class CompressedFrame implements Value {
229
230        private int frameIndex;
231        private int streamLength;
232        private CacheOutputStream cacheout = new CacheOutputStream();
233        private MemoryCacheImageOutputStream cache;
234
235        public CompressedFrame(int frameIndex) throws IOException {
236            this.frameIndex = frameIndex;
237        }
238
239        @Override
240        public boolean isEmpty() {
241            return false;
242        }
243
244        @Override
245        public byte[] toBytes(VR vr, boolean bigEndian) throws IOException {
246            ByteArrayOutputStream out = new ByteArrayOutputStream();
247            writeTo(out);
248            return out.toByteArray();
249        }
250
251        @Override
252        public void writeTo(DicomOutputStream out, VR vr) throws IOException {
253            writeTo(out);
254        }
255
256        @Override
257        public int calcLength(DicomEncodingOptions encOpts, boolean explicitVR, VR vr) {
258            return getEncodedLength(encOpts, explicitVR, vr);
259        }
260
261        @Override
262        public int getEncodedLength(DicomEncodingOptions encOpts, boolean explicitVR, VR vr) {
263            try {
264                compress();
265            } catch (IOException e) {
266                return -1;
267            }
268            return (streamLength + 1) & ~1;
269        }
270        
271        private void writeTo(OutputStream out) throws IOException {
272            compress();
273            try {
274                cacheout.set(out);
275                long start = System.currentTimeMillis();
276                cache.seek(streamLength);
277                cache.flushBefore(streamLength);
278                if ((streamLength & 1) != 0)
279                    out.write(0);
280                long end = System.currentTimeMillis();
281                LOG.debug("Flushed frame #{} from memory in {} ms",
282                        frameIndex + 1, end - start);
283            } finally {
284                try { cache.close(); } catch (IOException ignore) {}
285                cache = null;
286            }
287        }
288
289        private void compress() throws IOException {
290            if (cache != null)
291                return;
292
293            if (ex != null)
294                throw ex;
295
296            try {
297                BufferedImage bi = Compressor.this.readFrame(frameIndex);
298                Compressor.this.extractEmbeddedOverlays(frameIndex, bi);
299                if (bitsStored < bitsAllocated)
300                    Compressor.this.nullifyUnusedBits(bitsStored, bi);
301                cache = new MemoryCacheImageOutputStream(cacheout) {
302
303                    @Override
304                    public void flush() throws IOException {
305                        // defer flush to writeTo()
306                        LOG.debug("Ignore invoke of MemoryCacheImageOutputStream.flush()");
307                    }
308                };
309                compressor.setOutput(patchJPEGLS != null
310                        ? new PatchJPEGLSImageOutputStream(cache, patchJPEGLS)
311                        : cache);
312                long start = System.currentTimeMillis();
313                compressor.write(null, new IIOImage(bi, null, null), compressParam);
314                long end = System.currentTimeMillis();
315                streamLength = (int) cache.getStreamPosition();
316                if (LOG.isDebugEnabled())
317                    LOG.debug("Compressed frame #{} {}:1 in {} ms", 
318                            new Object[] {frameIndex + 1,
319                            (float) sizeOf(bi) / streamLength,
320                            end - start });
321                Compressor.this.verify(cache, frameIndex);
322            } catch (IOException ex) {
323                cache = null;
324                Compressor.this.ex = ex;
325                throw ex;
326            }
327        }

342    public BufferedImage readFrame(int frameIndex) throws IOException {
343        if (iis == null)
344            iis = new FileImageInputStream(file);
345
346        if (decompressor != null)
347            return decompressFrame(iis, frameIndex);
348
349        iis.setByteOrder(UID.ExplicitVRBigEndian.equals(pixeldata.transferSyntax) 
350                ? ByteOrder.BIG_ENDIAN
351                : ByteOrder.LITTLE_ENDIAN);
352        iis.seek(pixeldata.offset + frameLength * frameIndex);
353        DataBuffer db = bi.getRaster().getDataBuffer();
354        switch (db.getDataType()) {
355        case DataBuffer.TYPE_BYTE:
356            for (byte[] bs : ((DataBufferByte) db).getBankData())
357                iis.readFully(bs);
358            break;
359        case DataBuffer.TYPE_USHORT:
360            readFully(((DataBufferUShort) db).getData());
361            break;
362        case DataBuffer.TYPE_SHORT:
363            readFully(((DataBufferShort) db).getData());
364            break;
365        default:
366            throw new UnsupportedOperationException(
367                    "Unsupported Datatype: " + db.getDataType());
368        }
369        return bi;
370    }

372    private void verify(MemoryCacheImageOutputStream cache, int index)
373            throws IOException {
374        if (verifier == null)
375            return;
376
377        cache.seek(0);
378        verifier.setInput(cache);
379        verifyParam.setDestination(bi2);
380        long start = System.currentTimeMillis();
381        bi2 = verifier.read(0, verifyParam);
382        int maxDiff = maxDiff(bi.getRaster(), bi2.getRaster());
383        long end = System.currentTimeMillis();
384        if (LOG.isDebugEnabled())
385            LOG.debug("Verified compressed frame #{} in {} ms - max pixel value error: {}",
386                    new Object[] { index + 1, end - start, maxDiff });
387        if (maxDiff > maxPixelValueError)
388            throw new CompressionVerificationException(maxDiff);
389
390    }

392    private int maxDiff(WritableRaster raster, WritableRaster raster2) {
393        ComponentSampleModel csm = 
394                (ComponentSampleModel) raster.getSampleModel();
395        ComponentSampleModel csm2 = 
396                (ComponentSampleModel) raster2.getSampleModel();
397        DataBuffer db = raster.getDataBuffer();
398        DataBuffer db2 = raster2.getDataBuffer();
399        int blockSize = avgPixelValueBlockSize;
400        if (blockSize > 1) {
401            int w = csm.getWidth();
402            int h = csm.getHeight();
403            int maxY = (h / blockSize - 1) * blockSize;
404            int maxX = (w / blockSize - 1) * blockSize;
405            int[] samples = new int[blockSize * blockSize];
406            int diff, maxDiff = 0;
407            for (int b = 0; b < csm.getNumBands(); b++)
408                for (int y = 0; y < maxY; y += blockSize) {
409                    for (int x = 0; x < maxX; x += blockSize) {
410                        if (maxDiff < (diff = Math.abs(
411                                sum(csm.getSamples(
412                                    x, y, blockSize, blockSize, b, samples, db))
413                              - sum(csm2.getSamples(
414                                    x, y, blockSize, blockSize, b, samples, db2)))))
415                            maxDiff = diff;
416                    }
417                }
418            return maxDiff / samples.length;
419        }
420        switch (db.getDataType()) {
421        case DataBuffer.TYPE_BYTE:
422            return maxDiff(csm, ((DataBufferByte) db).getBankData(),
423                  csm2, ((DataBufferByte) db2).getBankData());
424        case DataBuffer.TYPE_USHORT:
425            return maxDiff(csm, ((DataBufferUShort) db).getData(),
426                  csm2, ((DataBufferUShort) db2).getData());
427        case DataBuffer.TYPE_SHORT:
428            return maxDiff(csm, ((DataBufferShort) db).getData(),
429                  csm2, ((DataBufferShort) db2).getData());
430        default:
431            throw new UnsupportedOperationException(
432                    "Unsupported Datatype: " + db.getDataType());
433        }
434    }

436    private int sum(int[] samples) {
437        int sum = 0;
438        for (int sample : samples)
439            sum += sample;
440        return sum;
441    }

443    private int maxDiff(ComponentSampleModel csm, short[] data,
444                        ComponentSampleModel csm2, short[] data2) {
445        int w = csm.getWidth() * csm.getPixelStride();
446        int h = csm.getHeight();
447        int stride = csm.getScanlineStride();
448        int stride2 = csm2.getScanlineStride();
449        int diff, maxDiff = 0;
450        for (int y = 0; y < h; y++) {
451            for (int j = w, i = y * stride, i2 = y * stride2; j-- > 0; i++, i2++) {
452                if (maxDiff < (diff = Math.abs(data[i] - data2[i2])))
453                    maxDiff = diff;
454            }
455        }
456        return maxDiff;
457    }

459    private int maxDiff(ComponentSampleModel csm, byte[][] banks,
460                        ComponentSampleModel csm2, byte[][] banks2) {
461        int w = csm.getWidth();
462        int h = csm.getHeight();
463        int bands = csm.getNumBands();
464        int stride = csm.getScanlineStride();
465        int pixelStride = csm.getPixelStride();
466        int[] bankIndices = csm.getBankIndices();
467        int[] bandOffsets = csm.getBandOffsets();
468        int stride2 = csm2.getScanlineStride();
469        int pixelStride2 = csm2.getPixelStride();
470        int[] bankIndices2 = csm2.getBankIndices();
471        int[] bandOffsets2 = csm2.getBandOffsets();
472        int diff, maxDiff = 0;
473        for (int b = 0; b < bands; b++) {
474            byte[] bank = banks[bankIndices[b]];
475            byte[] bank2 = banks2[bankIndices2[b]];
476            int off = bandOffsets[b];
477            int off2 = bandOffsets2[b];
478            for (int y = 0; y < h; y++) {
479                for (int x = w, i = y * stride + off, i2 = y * stride2 + off2;
480                        x-- > 0; i += pixelStride, i2 += pixelStride2) {
481                    if (maxDiff < (diff = Math.abs(bank[i] - bank2[i2])))
482                        maxDiff = diff;
483                }
484            }
485        }
486        return maxDiff;
487    }

489    private void nullifyUnusedBits(int bitsStored, BufferedImage bi) {
490        DataBuffer db = bi.getRaster().getDataBuffer();
491        switch (db.getDataType()) {
492        case DataBuffer.TYPE_USHORT:
493            nullifyUnusedBits(bitsStored, ((DataBufferUShort) db).getData());
494            break;
495        case DataBuffer.TYPE_SHORT:
496            nullifyUnusedBits(bitsStored, ((DataBufferShort) db).getData());
497            break;
498        }
499    }
500
501    private void nullifyUnusedBits(int bitsStored, short[] data) {
502        int mask = (1<<bitsStored)-1;
503        for (int i = 0; i < data.length; i++)
504            data[i] &= mask;
505    }

507    private void extractEmbeddedOverlays(int frameIndex, BufferedImage bi) {
508        for (int gg0000 : embeddedOverlays) {
509            int ovlyRow = dataset.getInt(Tag.OverlayRows | gg0000, 0);
510            int ovlyColumns = dataset.getInt(Tag.OverlayColumns | gg0000, 0);
511            int ovlyBitPosition = dataset.getInt(Tag.OverlayBitPosition | gg0000, 0);
512            int mask = 1 << ovlyBitPosition;
513            int ovlyLength = ovlyRow * ovlyColumns;
514            byte[] ovlyData = dataset.getSafeBytes(Tag.OverlayData | gg0000);
515            if (ovlyData == null) {
516                ovlyData = new byte[(((ovlyLength*frames+7)>>>3)+1)&(~1)];
517                dataset.setBytes(Tag.OverlayData | gg0000, VR.OB, ovlyData);
518            }
519            Overlays.extractFromPixeldata(bi.getRaster(), mask, ovlyData,
520                    ovlyLength * frameIndex, ovlyLength);
521            LOG.debug("Extracted embedded overlay #{} from bit #{} of frame #{}",
522                    new Object[]{(gg0000 >>> 17) + 1, ovlyBitPosition, frameIndex + 1});
523        }
524    }
