112    public Decompressor(Attributes dataset, String tsuid) {
113        if (tsuid == null)
114            throw new NullPointerException("tsuid");
115
116        this.dataset = dataset;
117        this.tsuid = tsuid;
118        this.tstype = TransferSyntaxType.forUID(tsuid);
119        Object pixeldata = dataset.getValue(Tag.PixelData);
120        if (pixeldata == null)
121            return;
122
123        if (tstype == null)
124            throw new IllegalArgumentException("Unknown Transfer Syntax: " + tsuid);
125        this.file = fileOf(pixeldata);
126        this.rows = dataset.getInt(Tag.Rows, 0);
127        this.cols = dataset.getInt(Tag.Columns, 0);
128        this.samples = dataset.getInt(Tag.SamplesPerPixel, 0);
129        this.pmi = PhotometricInterpretation.fromString(
130                dataset.getString(Tag.PhotometricInterpretation, "MONOCHROME2"));
131        this.bitsAllocated = dataset.getInt(Tag.BitsAllocated, 8);
132        this.bitsStored = dataset.getInt(Tag.BitsStored, bitsAllocated);
133        this.banded = dataset.getInt(Tag.PlanarConfiguration, 0) != 0;
134        this.signed = dataset.getInt(Tag.PixelRepresentation, 0) != 0;
135        this.frames = dataset.getInt(Tag.NumberOfFrames, 1);
136        this.frameLength = rows * cols * samples * (bitsAllocated>>>3);
137        this.length = frameLength * frames;
138        
139        if (pixeldata instanceof Fragments) {
140            if (!tstype.isPixeldataEncapsulated())
141                throw new IllegalArgumentException("Encapusulated Pixel Data"
142                        + "with Transfer Syntax: " + tsuid);
143            this.pixeldataFragments = (Fragments) pixeldata;
144
145            int numFragments = pixeldataFragments.size();
146            if (frames == 1 ? (numFragments < 2)
147                            : (numFragments != frames + 1))
148                throw new IllegalArgumentException(
149                        "Number of Pixel Data Fragments: "
150                        + numFragments + " does not match " + frames);
151
152            ImageReaderFactory.ImageReaderParam param =
153                    ImageReaderFactory.getImageReaderParam(tsuid);
154            if (param == null)
155                throw new UnsupportedOperationException(
156                        "Unsupported Transfer Syntax: " + tsuid);
157
158            this.decompressor = ImageReaderFactory.getImageReader(param);
159            LOG.debug("Decompressor: {}", decompressor.getClass().getName());
160            this.readParam = decompressor.getDefaultReadParam();
161            this.patchJpegLS = param.patchJPEGLS;
162        }
163    }
164
165    public void dispose() {
166        if (decompressor != null)
167            decompressor.dispose();
168
169        decompressor = null;
170    }

172    public boolean decompress() {
173        if (decompressor == null)
174            return false;
175 
176        if (tstype == TransferSyntaxType.RLE)
177            bi = createBufferedImage(bitsStored, true, signed);
178
179        dataset.setValue(Tag.PixelData, VR.OW, new Value() {
180
181            @Override
182            public boolean isEmpty() {
183                return false;
184            }
185
186            @Override
187            public byte[] toBytes(VR vr, boolean bigEndian) throws IOException {
188                ByteArrayOutputStream out = new ByteArrayOutputStream();
189                Decompressor.this.writeTo(out);
190                return out.toByteArray();
191            }
192
193            @Override
194            public void writeTo(DicomOutputStream out, VR vr) throws IOException {
195                Decompressor.this.writeTo(out);
196            }
197
198            @Override
199            public int calcLength(DicomEncodingOptions encOpts, boolean explicitVR, VR vr) {
200                return getEncodedLength(encOpts, explicitVR, vr);
201            }
202
203            @Override
204            public int getEncodedLength(DicomEncodingOptions encOpts, boolean explicitVR, VR vr) {
205                return (length + 1) & ~1;
206            }
207        });
208        if (samples > 1) {
209            dataset.setString(Tag.PhotometricInterpretation, VR.CS, 
210                    pmi.decompress().toString());
211
212            dataset.setInt(Tag.PlanarConfiguration, VR.US,
213                    tstype.getPlanarConfiguration());
214        }
215        return true;
216    }

218    public static boolean decompress(Attributes dataset, String tsuid) {
219        return new Decompressor(dataset, tsuid).decompress();
220    }

222    protected BufferedImage createBufferedImage(int bitsStored,
223            boolean banded, boolean signed) {
224        int dataType = bitsAllocated > 8 
225                ? (signed ? DataBuffer.TYPE_SHORT : DataBuffer.TYPE_USHORT)
226                : DataBuffer.TYPE_BYTE;
227        ComponentColorModel cm = samples == 1
228                    ? new ComponentColorModel(
229                        ColorSpace.getInstance(ColorSpace.CS_GRAY),
230                        new int[] { bitsStored },
231                        false, // hasAlpha
232                        false, // isAlphaPremultiplied,
233                        Transparency.OPAQUE,
234                        dataType)
235                    :  new ComponentColorModel(
236                        ColorSpace.getInstance(ColorSpace.CS_sRGB),
237                        new int[] { bitsStored, bitsStored, bitsStored },
238                        false, // hasAlpha
239                        false, // isAlphaPremultiplied,
240                        Transparency.OPAQUE,
241                        dataType);
242
243        SampleModel sm = banded
244                ? new BandedSampleModel(dataType, cols, rows, samples)
245                : new PixelInterleavedSampleModel(dataType, cols, rows,
246                        samples, cols * samples, bandOffsets());
247        WritableRaster raster = Raster.createWritableRaster(sm, null);
248        return new BufferedImage(cm, raster, false, null);
249    }
250
251    private int[] bandOffsets() {
252        int[] offsets = new int[samples];
253        for (int i = 0; i < samples; i++)
254            offsets[i] = i;
255        return offsets;
256    }

258    static File fileOf(Object o) {
259        if (o instanceof Fragments)
260            return fileOf(((Fragments) o).get(1));
261
262        try {
263            return new File(new URI(((BulkDataLocator) o).uri));
264        } catch (ClassCastException e) {
265            throw new IllegalArgumentException("Unsupported Pixel Data : "
266                    + o.getClass());
267        } catch (URISyntaxException e) {
268            throw new IllegalArgumentException("Invalid URI: "
269                    + ((BulkDataLocator) o).uri);
270        }
271    }

273    private void writeTo(OutputStream out) throws IOException {
274        ImageInputStream iis = new FileImageInputStream(file);
275        try {
276            for (int i = 0; i < frames; ++i)
277                writeTo(decompressFrame(iis, i).getRaster(), out);
278            if ((length & 1) != 0)
279                out.write(0);
280        } finally {
281            try { iis.close(); } catch (IOException ignore) {}
282            decompressor.dispose();
283        }
284    }
285
286    @SuppressWarnings("resource")
287    protected BufferedImage decompressFrame(ImageInputStream iis, int index)
288            throws IOException {
289        SegmentedInputImageStream siis =
290                new SegmentedInputImageStream(iis, pixeldataFragments, index);
291        decompressor.setInput(patchJpegLS != null
292                ? new PatchJPEGLSImageInputStream(siis, patchJpegLS)
293                : siis);
294        readParam.setDestination(bi);
295        long start = System.currentTimeMillis();
296        bi = decompressor.read(0, readParam);
297        long end = System.currentTimeMillis();
298        if (LOG.isDebugEnabled())
299            LOG.debug("Decompressed frame #{} 1:{} in {} ms", 
300                    new Object[] {index + 1,
301                    (float) sizeOf(bi) / siis.getStreamPosition(),
302                    end - start });
303        return bi;
304    }

306    static int sizeOf(BufferedImage bi) {
307        DataBuffer db = bi.getData().getDataBuffer();
308        return db.getSize() * db.getNumBanks()
309                * (DataBuffer.getDataTypeSize(db.getDataType()) >>> 3);
310    }
311
312    private static void writeTo(Raster raster, OutputStream out) throws IOException {
313        SampleModel sm = raster.getSampleModel();
314        DataBuffer db = raster.getDataBuffer();
315        switch (db.getDataType()) {
316        case DataBuffer.TYPE_BYTE:
317            writeTo(sm, ((DataBufferByte) db).getBankData(), out);
318            break;
319        case DataBuffer.TYPE_USHORT:
320            writeTo(sm, ((DataBufferUShort) db).getData(), out);
321            break;
322        case DataBuffer.TYPE_SHORT:
323            writeTo(sm, ((DataBufferShort) db).getData(), out);
324            break;
325        case DataBuffer.TYPE_INT:
326            writeTo(sm, ((DataBufferInt) db).getData(), out);
327            break;
328        default:
329            throw new UnsupportedOperationException(
330                    "Unsupported Datatype: " + db.getDataType());
331        }
332    }

334    private static void writeTo(SampleModel sm, byte[][] bankData, OutputStream out)
335            throws IOException {
336        int h = sm.getHeight();
337        int w = sm.getWidth();
338        ComponentSampleModel csm = (ComponentSampleModel) sm;
339        int len = w * csm.getPixelStride();
340        int stride = csm.getScanlineStride();
341        if (csm.getBandOffsets()[0] != 0)
342            bgr2rgb(bankData[0]);
343        for (byte[] b : bankData)
344            for (int y = 0, off = 0; y < h; ++y, off += stride)
345                out.write(b, off, len);
346    }
347
348    private static void bgr2rgb(byte[] bs) {
349        for (int i = 0, j = 2; j < bs.length; i += 3, j += 3) {
350            byte b = bs[i];
351            bs[i] = bs[j];
352            bs[j] = b;
353        }
354    }
355
356    private static void writeTo(SampleModel sm, short[] data, OutputStream out)
357            throws IOException {
358        int h = sm.getHeight();
359        int w = sm.getWidth();
360        int stride = ((ComponentSampleModel) sm).getScanlineStride();
361        byte[] b = new byte[w * 2];
362        for (int y = 0; y < h; ++y) {
363            for (int i = 0, j = y * stride; i < b.length;) {
364                short s = data[j++];
365                b[i++] = (byte) s;
366                b[i++] = (byte) (s >> 8);
367            }
368            out.write(b);
369        }
370    }
371
372    private static void writeTo(SampleModel sm, int[] data, OutputStream out)
373            throws IOException {
374        int h = sm.getHeight();
375        int w = sm.getWidth();
376        int stride = ((SinglePixelPackedSampleModel) sm).getScanlineStride();
377        byte[] b = new byte[w * 3];
378        for (int y = 0; y < h; ++y) {
379            for (int i = 0, j = y * stride; i < b.length;) {
380                int s = data[j++];
381                b[i++] = (byte) (s >> 16);
382                b[i++] = (byte) (s >> 8);
383                b[i++] = (byte) s;
384            }
385            out.write(b);
386        }
387    }
